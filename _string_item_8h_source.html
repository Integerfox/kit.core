24/** This partially concrete provides a wrapper class that allows a IString
25 object to be stored directly in one the KIT Containers. A StringItem
26 can be stored in a SList, DList, and Map.
27
28 Notes:
29
30 o The allocation of the actual IString is deferred to child classes.
31
32 o When storing a StringItem in a 'keyed' container (i.e. Map) the
33 application MUST NOT change the value of the internal string since it
34 is the 'key' value that is used to determine its placement in the Map.
35 Modifying the internal IString while the StringItem is in a Map will
36 CORRUPT the Container.
37
38 o StringItem instances CANNOT be copied. This is to avoid potential
39 Container corruption since the individual Items contain the memory
40 for the Containers' linkage, i.e. a copied container Item is
41 analogous to storing a single Item in more than one Container at the
42 same time.
43 */
44class StringItem : public Kit::Container::MapItem
45{
46protected:
47 /// Constructor
48 StringItem( IString& string ) noexcept :m_myString( string ) {}
49
50public:
51 /// Returns a reference to the actual IString
52 inline IString& get() const noexcept { return m_myString; }
53
54 /// Returns a pointer to the internal C string
55 inline const char* getString() const noexcept { return m_myString.getString(); }
56
57 /// Cast to a IString reference
58 inline operator IString& ( ) const noexcept { return m_myString; }
59
60 /// Cast to read-only character string pointer.
61 inline operator const char* ( ) const noexcept { return m_myString.getString(); }
62
63 /// Returns a Read-only pointer to the "raw" (short-hand for getString())
64 inline const char* operator()() const noexcept { return m_myString.getString(); }
65
66
67public:
68 /// Return Kit::Container::Key (A IString is-a Key)
69 const Kit::Container::Key& getKey() const noexcept { return m_myString; }
70
71
72protected:
73 /// Reference to the actual IString/storage
74 IString & m_myString;
75
76
77private:
78 /// Prevent access to the copy constructor -->Container Items should not be copied!
79 StringItem( const StringItem& m ) noexcept = delete;
80
81 /// Prevent access to the assignment operator -->Container Items should not be copied!

83
84 /// Prevent access to the move constructor -->Container Items should not be moved!
85 StringItem( StringItem&& m ) noexcept = delete;
