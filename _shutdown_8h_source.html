40/** This class defines methods for forcibly terminating the application.
41 How gracefully (or not) the shutdown is dependent on the actual platform
42 implementation. In addition, what 'shutdown' means is also platform
43 dependent, e.g. on an embedded platform it may invoke a reboot and/or
44 restart of the application.
45
46 Since this is a forcibly shutdown there are no guaranties with respect to
47 releasing resources.
48
49 The interface does provide for callback method(s) to be called on shutdown.
50 These methods are guaranteed to be called (or at least attempted to be
51 called in the case of true code failure).
52 */
53class Shutdown
54{
55public:
56 /** This call defines the callback interface that is used when the
57 application is shutdown
58 */
59 class Handler : public Kit::Container::ListItem
60 {
61 public:
62 /** This method is called when the application is shutting down. The
63 method is passed 'exit_code' which is the exit code provided by the
64 application when called the Shutdown interface. The return value
65 from the handler will be used as the application shutdown exit code.
66 To leave the exit code unaltered - return the passed in 'exit_code'.
67
68 Notes:
69 o The final exit code is the serialized 'sum' of the all of the
70 shutdown handlers.
71 o The notify() method is called in the thread context that
72 initiated the shutdown.
73 */
74 virtual int notify( int exitCode ) noexcept = 0;
75
76 public:
77 /// Ensure the destructor is virtual
78 virtual ~Handler() {}
79 };
80
81
82public:
83 /** This function will force a shutdown of the application with a
84 'success' exit code. What 'forced' means is dependent on the
85 actual platform. All registered callback methods will be called
86 before exiting the application.
87 */
88 static int success() noexcept;
89
90 /** This function will force a shutdown of the application with a 'failure'
91 exit code. The caller can optional specify an exit code. What 'forced'
92 means is dependent on the actual platform. All registered callback
93 methods will be called before exiting the application. The returned
94 value - assuming the method actually returns - will be the exit code
95 returned by the last shutdown handler.
96
97 Note: The recommended approach for exiting the application due to an
98 error is to use the Cpl::System::FatalError interface so that
99 the 'why' of the failure has the potential for being captured.
100 */
101 static int failure( int exitCode = OPTION_KIT_SYSTEM_SHUTDOWN_FAILURE_ERROR_CODE ) noexcept;
102
103
104public:
105 /** This method is used to register a callback method that will be called
106 when success() or failure() is called and before the application exits.
107
108 A LIFO is used when registering handlers. This means that the first
109 handler registered, is the last handler run.
110
111 During shutdown, the 'notify()' method of 'instanceToRegister'
112 is called.
113 */
114 static void registerHandler( Shutdown::Handler& instanceToRegister ) noexcept;
115
116
117public:
118 /** This COMPONENT Scoped method. The application SHOULD not directly
119 access this method (unless you are REALLY REALLY REALLY sure you
120 know what you are doing).
121
122 Helper method used to notify registered shutdown handlers
123 */
124 static int notifyShutdownHandlers( int exitCode ) noexcept;
125};
126
127
128}; // end namespaces
129};
130#endif // end header latch
