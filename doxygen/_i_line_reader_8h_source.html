22/** This abstract class defines a interface for a stream line reader. A
23 line reader allows the client to read one line at time from the stream.
24 A line is consider a stream of ASCII character terminated by '\n' a.k.a
25 newline. The actual newline character(s) are handled by the interface
26 and the client is not aware of the actual newline implementation.
27 */
28class ILineReader
29{
30public:
31 /** Reads a single line from the stream. If the line is greater than the
32 will fit in 'destString', the line content is truncated. The next
33 subsequent readln() reads the next line (NOT the truncated characters!).
34
35 The newline character(s) are NOT copied into 'destString' but is
36 replaced with '\0'. Returns true if successful, or false if End-of-Stream
37 was encountered.
38
39 NOTE: This call will not return until a newline character is encounter!
40 */
41 virtual bool readln( Kit::Text::IString& destString ) noexcept = 0;
42
43 /** Returns true if there is data available to be read from the stream.
44
45 NOTE: The implementation of this method is VERY PLATFORM dependent! If
46 your code uses it - it may not be portable to all platforms.
47 If a platform does not/can not support this method it is
48 guaranteed to return 'true'
49 */
50 virtual bool available() noexcept = 0;
51
52 /** Closes the reader and the underlying input stream.
53 */
54 virtual void close() noexcept = 0;
55
56
57public:
58 /// Lets the make the destructor virtual
59 virtual ~ILineReader() noexcept = default;
60};
