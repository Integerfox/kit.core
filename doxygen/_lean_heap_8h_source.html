24/** This template class manages an allocate-only-heap with statically allocated
25 memory. Memory allocated by this class cannot be freed. All of the memory
26 is aligned to the alignment requirement of the specified word type.
27
28 Note: The heap has a 'reset' method that can be used to free all of the
29 memory allocated by the instance - HOWEVER it is the responsibility
30 of the Application to ensure the proper behavior when this
31 method is used (i.e. that is okay to 'free everything')
32
33 NOTE: In C++ it is possible that sizeof(T) does NOT equal alignof(T). For
34 example on my Ubuntu x86_64 system, sizeof(std::max_align_t) is 32 bytes
35 but alignof(std::max_align_t) is 16 bytes. Why do I care? Because
36 memory is allocated in 'words' of sizeof(T) bytes, not alignof(T) bytes.
37
38 Template args:
39 MAXWORDS - The maximum number of words in the heap. Must be > 0.
40 WORDTYPE - The type that defines the word size and alignment.
41 Defaults to std::max_align_t.
42 */
43
44template <size_t MAXWORDS, typename WORDTYPE = std::max_align_t>
45class LeanHeap : public ContiguousAllocator
46{
47 static_assert( MAXWORDS > 0, "MAXWORDS must be greater than 0" );
48
49 /// Total size of the heap in bytes
50 static constexpr size_t HEAP_SIZE_IN_BYTES = MAXWORDS * sizeof( WORDTYPE );
51
52public:
53 /** Constructor for statically-allocated heap memory.
54 */
55 LeanHeap() noexcept
56 : m_ptrBase( m_heapStorage )
57 , m_ptr( m_heapStorage )
58 , m_unitsRemaining( MAXWORDS )
59 {
60 // Zero out all of memory
61 std::memset( m_heapStorage, 0, sizeof(m_heapStorage) );
62 }
63
64public:
65 /// See Kit::Memory::Allocator
66 void* allocate( size_t numbytes ) noexcept override
67 {
68 // Calculate the actual size needed (rounded up to word boundary)
69 size_t alignedBytes = allocatedSizeForNBytes( numbytes );
70 size_t numUnits = alignedBytes / sizeof( WORDTYPE );
71
72 if ( numUnits > m_unitsRemaining )
73 {
74 return nullptr;
75 }
76
77 uint8_t* result = m_ptr;
78 m_unitsRemaining -= numUnits;
79 m_ptr += alignedBytes;
80 return result;
81 }
82
83 /// See Kit::Memory::Allocator
84 size_t wordSize() const noexcept override { return sizeof( WORDTYPE ); }
85
86 /// See Kit::Memory::ContiguousAllocator
87 void reset() noexcept override
88 {
89 m_unitsRemaining = MAXWORDS;
90 m_ptr = m_ptrBase;
91 std::memset( m_ptr, 0, MAXWORDS * sizeof( WORDTYPE ) );
92 }
93
94 /// See Kit::Memory::ContiguousAllocator.
95 MemoryInfo_T& getMemoryStart( MemoryInfo_T& dstInfo ) noexcept override
96 {
97 dstInfo.m_memoryStartPtr = m_ptrBase;
98 dstInfo.m_totalLenInBytes = HEAP_SIZE_IN_BYTES;
99 dstInfo.m_maxWords = MAXWORDS;
100 dstInfo.m_freeWords = m_unitsRemaining;
101 return dstInfo;
102 }
103
104private:
105 /// Prevent access to the copy constructor -->LeanHeap can not be copied!
106 LeanHeap( const LeanHeap& m ) = delete;
107
108 /// Prevent access to the assignment operator -->LeanHeap can not be copied!
109 const LeanHeap& operator=( const LeanHeap& m ) = delete;
110
111protected:
112 /// Pointer to the base of the original heap memory
113 uint8_t* m_ptrBase;
114
115 /// Pointer to the next available free byte of memory
116 uint8_t* m_ptr;
117
118 /// The amount of remaining free memory in alignment units
119 size_t m_unitsRemaining;
120
121 /// Statically allocated heap storage
122 alignas( WORDTYPE ) uint8_t m_heapStorage[HEAP_SIZE_IN_BYTES];
123};
