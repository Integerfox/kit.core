35/** This class provides a centralized supervisor for monitoring all watched threads
36 in the system. It maintains a list of threads being monitored and provides
37 thread-safe methods for managing the watchdog system.
38
39 All methods are static and thread-safe to allow access from multiple threads.
40 The class uses a composition relationship with SList<WatchedThread> to manage
41 the collection of monitored threads.
42 */
43class Supervisor
44{
45public:
46 /** Enables the watchdog system. Must be called before any threads
47 can be monitored.
48
49 @return true if watchdog was successfully enabled, false otherwise
50 */
51 static bool enableWdog() noexcept;
52
53 /** Begins monitoring a thread. The thread will be added to the list
54 of monitored threads. Once a thread has been added to the Supervisor,
55 this method can NOT be called again until after endWatching() has
56 been called on the same thread.
57 @param threadToMonitor Reference to the WatchedThread to begin monitoring
58 */
59 static void beginWatching( WatchedThread& threadToMonitor ) noexcept;
60
61 /** Stops monitoring a thread. The thread will be removed from the list
62 of monitored threads.
63 @param threadBeingMonitored Reference to the WatchedThread to stop monitoring
64 */
65 static void endWatching( WatchedThread& threadBeingMonitored ) noexcept;
66
67 /** Monitors all registered threads. This method should be called periodically
68 (typically every tick/event cycle) by the supervisor thread. It includes
69 a tick divider to reduce the performance impact of being called frequently.
70
71 This method checks each monitored thread's currentCountMs against its
72 wdogTimeoutMs and triggers the watchdog if any thread has timed out.
73 */
74 static void monitorThreads() noexcept;
75
76 /** Reloads/resets the watchdog counter for a specific thread. This is
77 called by the thread to indicate it is still alive and processing.
78
79 @param thread Reference to the WatchedThread to reload
80 */
81 static void reloadThread( WatchedThread& thread ) noexcept;
82
83 /** Trips the watchdog to force a system reset. This method ensures the
84 watchdog is enabled (calling enableWdog() if necessary) before tripping
85 to guarantee a system reset occurs.
86 */
87 static void tripWdog() noexcept;
88
89protected:
90 /// List of threads being monitored
91 static Kit::Container::SList<WatchedThread> m_watchedThreads;
92
93 /// Mutex for thread-safe access to the watched threads list
94 static Mutex m_mutex;
95
96 /// Current tick count for the divider
97 static uint32_t m_currentTick;
98
99 /// Time marker for elapsed time calculations
100 static uint32_t m_timeMarker;
101
102 /// Flag indicating if watchdog system is enabled
103 static bool m_isEnabled;
104};
