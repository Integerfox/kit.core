20/** This mostly concrete interface defines the operations that can be performed
21 on a software timer. Software timers execute 'in-thread' in that all
22 operations (start, timer expired callbacks, etc.) are performed in a single
23 thread.
24
25 Because the timer context (i.e. the timer owner), timer methods and callbacks
26 all occur in the same thread, the timer context will never receive a timer
27 expired callback AFTER the timer's stop() method has been called.
28
29 NOTES:
30 o The timer context must implement the following method:
31 virtual void expired() noexcept;
32
33 o Because the timing source of an individual thread may NOT be a clean
34 divider of the timer duration, the timer duration is taken as the
35 minimum. For example: if the timing source has a resolution of 20msec
36 per count, and the timer duration on the start() timer call is 5 msec,
37 then the timer will expire after the next full count, i.e. after 20msec,
38 not 5msec. IT IS THE APPLICATION'S RESPONSIBILITY TO MANAGE THE
39 RESOLUTION OF THE TIMING SOURCES.
40 */
41class Timer : public ICounter
42{
43public:
44 /// Constructor
45 Timer( TimerManager& timingSource ) noexcept;
46
47 /// Constructor. Alternate constructor - that optionally defers the assignment of the timing source
48 Timer( TimerManager* timingSource = nullptr ) noexcept;
49
50public:
51 /** Starts the timer with an initial count down count duration of
52 'timerDurationInMilliseconds'. If the timer is currently running, the
53 timer is first stopped, and then restarted.
54 */
55 virtual void start( uint32_t timerDurationInMilliseconds ) noexcept;
56
57 /** Stops the timer. It is okay to call stop() even after the timer has
58 previously expired or explicitly stopped.
59 */
60 virtual void stop() noexcept;
61
62public:
63 /** Sets the timing source. This method CAN ONLY BE CALLED when the timer
64 has never been started or it has been stopped
65 */
66 virtual void setTimingSource( TimerManager& timingSource ) noexcept;
67
68protected:
69 /// See Cpl::System::ICounter
70 void decrement( uint32_t milliseconds = 1 ) noexcept override;
71
72 /// See Cpl::System::ICounter
73 void increment( uint32_t milliseconds = 1 ) noexcept override;
74
75 /// See Cpl::System::ICounter
76 uint32_t count() const noexcept override;
77
78protected:
79 /// The timer's tick source
80 TimerManager* m_timingSource;
81
82 /// Current count
83 uint32_t m_count;
84};

87/** This template class implements a Software Timer that is context
88 independent and allows for a single context to contain many Timers.
89
90 Template args:
91 CONTEXT Type of the Class that implements the context for the timer
92 */
93
94template <class CONTEXT>
95class TimerComposer : public Timer
96{
97public:
98 /** Definition of the call-back method that notifies the
99 context/client when the timer expires.
100 */
101 typedef void ( CONTEXT::*TimerExpiredFunction_T )();
102
103public:
104 /// Constructor
105 TimerComposer( CONTEXT& timerContextInstance,
106 TimerExpiredFunction_T expiredCallbackFunc,
107 TimerManager& timingSource ) noexcept
108 : Timer( timingSource )
109 , m_context( timerContextInstance )
110 , m_expiredFuncPtr( expiredCallbackFunc )
111 {
112 }
113
114 /** Alternate Constructor that is used to defer the assignment of the time source.
115 When using this constructor - the Application logic is REQUIRED to use
116 the setTimingSource() method to set the timing source BEFORE the timer
117 is used.
118 */
119 TimerComposer( CONTEXT& timerContextInstance,
120 TimerExpiredFunction_T expiredCallbackFunc,
121 TimerManager* timingSourcePtr = nullptr ) noexcept
122 : Timer( timingSourcePtr )
123 , m_context( timerContextInstance )
124 , m_expiredFuncPtr( expiredCallbackFunc )
125 {
126 }
127
128protected:
129 /// See Cpl::System::ICounter
130 void expired() noexcept
131 {
132 ( m_context.*m_expiredFuncPtr )();
133 }
134
135protected:
136 /// The callback context, i.e. the class that implements the 'expiredCallbackFunc' method
137 CONTEXT& m_context;
138
139 /// The context's method for 'expiredCallbackFunc' method, i.e. this method gets called when the timer expires
140 TimerExpiredFunction_T m_expiredFuncPtr;
141};
