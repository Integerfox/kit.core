28/** This concrete class monitors the a Queue of LED Flash codes and the
29 consumes-and-executes the Flash codes from the Queue.
30
31 This class is thread safe
32 */
33class LedServer : public Kit::Itc::OpenCloseSync
34{
35public:
36 /// Constructor
37 LedServer( Kit::EventQueue::IQueue& myEventQueue,
38 Kit::Container::RingBufferMP<FlashCode_T>& flashCodeQueue )
39 : Kit::Itc::OpenCloseSync( myEventQueue )
40 , m_flashCodeQueue( flashCodeQueue )
41 , m_obFlashCodeQueue( myEventQueue )
42 , m_opened( false )
43 {
44 m_obFlashCodeQueue.setCallback<LedServer, &LedServer::queueElementCountChanged>( this );
45 }
46
47public:
48 /// See Kit::Itc::OpenCloseSync
49 void request( OpenMsg& msg ) noexcept override
50 {
51 // Only 'open' when not already opened
52 if ( !m_opened )
53 {
54 // Housekeeping
55 m_opened = true;
56
57 // Subscribe for the Flash Code Queue change notifications
58 m_flashCodeQueue.m_mpElementCount.attach( m_obFlashCodeQueue );
59 }
60
61 // Return the message to the sender since we are done processing the request
62 msg.returnToSender();
63 }
64
65 /// See Kit::Itc::OpenCloseSync
66 void request( CloseMsg& msg ) noexcept override
67 {
68 // Only 'close' when already opened
69 if ( m_opened )
70 {
71 // Housekeeping
72 m_opened = false;
73
74 // Make sure the all of the potential callbacks get cancelled
75 m_flashCodeQueue.m_mpElementCount.detach( m_obFlashCodeQueue );
76 }
77
78 // Return the message to the sender since we are done processing the request
79 msg.returnToSender();
80 }
81
82
83protected:
84 /// Callback when the number of items in the Flash Code Queue changes
85 void queueElementCountChanged( Kit::Dm::Mp::Uint32& mpElementCount, Kit::Dm::IObserver& observer ) noexcept
86 {
87 uint32_t numElementsInQueue;
88 if ( mpElementCount.readAndSync( numElementsInQueue, observer ) && numElementsInQueue > 0 )
89 {
90 // Temporarily cancel the change notification to skip receiving 'removed' notifications
91 mpElementCount.detach( m_obFlashCodeQueue );
92
93 // Drain ALL flash code in the Queue and execute them.
94 FlashCode_T flashCode;
95 uint16_t seqNum;
96 while ( m_flashCodeQueue.remove( flashCode, seqNum ) )
97 {
98 executeFlashCode( flashCode );
99 }
100
101 // Re-subscribe for the Flash Code Queue change notifications.
102 // Use the sequence number of the last 'used-by-the-server-state'
103 // so as to NOT get an immediate callback when no new item has been added
104 // while NOT missing a change notification that could have occurred
105 // between when the server called remove() and and server re-subscribing
106 mpElementCount.attach( m_obFlashCodeQueue, seqNum );
107 }
108 }
109
110protected:
111 /// Helper method to execute a flash code
112 void executeFlashCode( const FlashCode_T& flashCode ) noexcept
113 {
114 KIT_SYSTEM_TRACE_MSG( "main",
115 "SERVER: Executing a flash code. on=%" PRIu32 " ms, off=%" PRIu32 " ms, numFlashes=%u, totalTime=%" PRIu32 " ms",
116 flashCode.onPeriodMs,
117 flashCode.offPeriodMs,
118 flashCode.numFlashes,
119 ( flashCode.onPeriodMs + flashCode.offPeriodMs ) * flashCode.numFlashes );
120
121 for ( unsigned i = 0; i < flashCode.numFlashes; ++i )
122 {
123 Bsp_turn_on_debug1();
124 Kit::System::sleep( flashCode.onPeriodMs );
125 Bsp_turn_off_debug1();
126 Kit::System::sleep( flashCode.offPeriodMs );
127 }
128 }
129
130protected:
131 /// Flash codes Queue
132 Kit::Container::RingBufferMP<FlashCode_T>& m_flashCodeQueue;
133
134 /// Observer callback for when flash code(s) are added to the Queue
135 Kit::Dm::ObserverCallback<Kit::Dm::Mp::Uint32> m_obFlashCodeQueue;
136
137 /// Track my open/closed
138 bool m_opened;
139};
