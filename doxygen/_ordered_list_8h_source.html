20/** This template class implements a double linked list which maintains the
21 ordering imposed on it by the application. It is type-safe wrapper around
22 the OrderedListBase class.
23
24 NOTE: KEYED_ITEM must be a subclass of Kit::Container::KeyedItem.
25 */
26template <class KEYED_ITEM>
27class OrderedList : public OrderedListBase
28{
29public:
30 /// Public constructor initializes head and tail pointers.
31 OrderedList() noexcept
32 : OrderedListBase() {}
33
34 /** This is a special constructor for when the list is statically declared

36 executed. See Kit::Container::OrderedListBase for details.
37 */
38 OrderedList( const char* ignoreThisParameter_usedToCreateAUniqueConstructor ) noexcept
39 : OrderedListBase( ignoreThisParameter_usedToCreateAUniqueConstructor ) {}
40
41public:
42 /// Moves the content of the this queue to the specified queue.
43 void move( OrderedList<KEYED_ITEM>& dst ) noexcept { OrderedListBase::move( dst ); }
44
45 /// Empties the list. All references to the item(s) in the list are lost.
46 void clearTheList() noexcept { OrderedListBase::clearTheList(); }
47
48public:
49 /** Removes the first item in the list. Returns nullptr if the list
50 is empty.
51 */
52 KEYED_ITEM* getFirst() noexcept { return static_cast<KEYED_ITEM*>(OrderedListBase::getFirst()); }
53
54 /** Removes the last item in the list. Returns nullptr if the list

59 /** Remove specified KEYED_ITEM element from the list. Returns true if the

62 bool remove( KEYED_ITEM& item ) noexcept { return OrderedListBase::remove( item ); }
63
64 /** Insert the KEYED_ITEM "item" in 'order' into the list
65 */
66 void insert( KEYED_ITEM& item ) noexcept { OrderedListBase::insert( item ); }
67
68 /// Returns true if the specified item is already in the list, else false.

70
71 /** Return a pointer to the first item in the list. The returned item

79 KEYED_ITEM* last() const noexcept { return static_cast<KEYED_ITEM*>(OrderedListBase::last()); }
80
81 /** Return a pointer to the item after the "item". Both items remain in

86 /** Return a pointer to the item before the "item". Both items remain in

90
91private:
