29/** This concrete class is responsible for generating flash codes (i.e. adding
30 to the Flash Code RingBuffer. The class fills the entire Ring Buffer and
31 then waits for the Ring Buffer to be empty before repeating the fill process
32 */
33class Producer : public Kit::Itc::OpenCloseSync
34{
35public:
36 /// Constructor
37 Producer( Kit::EventQueue::IQueue& myEventQueue,
38 Kit::Container::RingBufferMP<FlashCode_T>& flashCodeQueue,
39 unsigned maxRefills = UINT_MAX )
40 : Kit::Itc::OpenCloseSync( myEventQueue )
41 , m_flashCodeQueue( flashCodeQueue )
42 , m_obFlashCodeQueue( myEventQueue )
43 , m_maxRefills( maxRefills )
44 , m_opened( false )
45 {
46 m_obFlashCodeQueue.setCallback<Producer, &Producer::queueElementCountChanged>( this );
47 }
48
49public:
50 /// See Kit::Itc::OpenCloseSync
51 void request( OpenMsg& msg ) noexcept override
52 {
53 // Only 'open' when not already opened
54 if ( !m_opened )
55 {
56 // Housekeeping
57 m_opened = true;
58 m_fillCount = 0;
59
60 // Subscribe for the Queue change notifications
61 m_flashCodeQueue.m_mpElementCount.attach( m_obFlashCodeQueue );
62
63 // Fill the Flash Code Queue with flash codes to be executed by the Server
64 fillFlashCodeQueue();
65 }
66
67 // Return the message to the sender since we are done processing the request
68 msg.returnToSender();
69 }
70
71 /// See Kit::Itc::OpenCloseSync
72 void request( CloseMsg& msg ) noexcept override
73 {
74 // Only 'close' when already opened
75 if ( m_opened )
76 {
77 // Housekeeping
78 m_opened = false;
79
80 // Make sure the all of the potential callbacks get cancelled
81 m_flashCodeQueue.m_mpElementCount.detach( m_obFlashCodeQueue );
82 }
83
84 // Return the message to the sender since we are done processing the request
85 msg.returnToSender();
86 }
87
88
89protected:
90 /// Callback when the number of items in the Flash Code Queue changes
91 void queueElementCountChanged( Kit::Dm::Mp::Uint32& mpElementCount, Kit::Dm::IObserver& observer ) noexcept
92 {
93 uint32_t numElementsInQueue;
94 if ( mpElementCount.readAndSync( numElementsInQueue, observer ) && numElementsInQueue == 0 )
95 {
96 // Temporarily cancel the change notification to skip receiving 'adding' notifications
97 mpElementCount.detach( m_obFlashCodeQueue );
98
99 // Re-fill the Flash Code Queue with flash codes to be executed by the LedServer
100 fillFlashCodeQueue();
101
102 // Shutdown after N 'fill cycles'
103 if ( m_fillCount >= m_maxRefills )
104 {
105 Kit::System::Shutdown::success();
106 return;
107 }
108
109 // Re-subscribe for the Flash Code Queue change notifications.
110 mpElementCount.attach( m_obFlashCodeQueue );
111 }
112 }
113
114protected:
115 /// Helper method that produces flash codes and adds them to the Flash Code Queue
116 void fillFlashCodeQueue() noexcept
117 {
118 m_fillCount++;
119 unsigned idx = 0;
120 unsigned numAdded = 0;
121 while ( m_flashCodeQueue.add( g_fcodes[idx] ) )
122 {
123 idx = ( idx + 1 ) % ( sizeof( g_fcodes ) / sizeof( g_fcodes[0] ) );
124 numAdded++;
125 }
126 KIT_SYSTEM_TRACE_MSG( "main", "Producer[%u]: Added %u Flash Codes to the Queue", m_fillCount, numAdded );
127 }
128
129protected:
130 /// Flash codes Queue
131 Kit::Container::RingBufferMP<FlashCode_T>& m_flashCodeQueue;
132
133 /// Observer callback for when flash code(s) are removed from the Queue
134 Kit::Dm::ObserverCallback<Kit::Dm::Mp::Uint32> m_obFlashCodeQueue;
135
136 /// Maximum number of times to fill/refill the Flash Code Queue.
137 unsigned m_maxRefills;
138
139 /// Number of fill/refill attempts
140 unsigned m_fillCount;
141
142 /// Track my open/closed
143 bool m_opened;
144
145 /// Array of flash codes
146 static FlashCode_T g_fcodes[4];
147};
