20/** The maximum number of consecutive corrupt entries to scan during recovery
21 of the Record. The larger this number, the more resilient the recovery
22 process is to consecutive corrupt entries, but it may increase the recovery
23 time - this includes the initial 'virgin' boot of the persistent storage.
24 */
25#ifndef OPTION_KIT_PERSISTENCE_JOURNAL_ENTRY_RECORD_MAX_CORRUPT_SCAN
26#define OPTION_KIT_PERSISTENCE_JOURNAL_ENTRY_RECORD_MAX_CORRUPT_SCAN 8
27#endif
28
29/** Number of consecutive corrupt entries skip over when using getNext/getPrevious
30 to traverse the entries before declaring an error. This provides fault tolerance
31 against entries that may have been corrupted when a power-loss occurs while writing
32 the entry to persistent storage.
33 */
34#ifndef OPTION_KIT_PERSISTENCE_JOURNAL_ENTRY_RECORD_MAX_CONSECUTIVE_CORRUPT_SKIP
35#define OPTION_KIT_PERSISTENCE_JOURNAL_ENTRY_RECORD_MAX_CONSECUTIVE_CORRUPT_SKIP 3
36#endif

47/** This concrete class implements the IRecord interface that manages a collection
48 of Entries that are timestamp and oldest entries are silently lost when the
49 IMedia storage is 'full'
50
51 This class is responsible for managing the 'head pointer' for the logical
52 RingBuffer that is used to manage multiple entries.
53*/

57 /** Constructor.

59 @param singleEntrySizeInBytes Size of the application payload for one entry, in bytes

61 @param headRecord IHead instance that persists the ring-buffer head pointer
62 @param maxCorruptScan The maximum number of consecutive corrupt entries to walk
63 back over when recovering the head pointer at startup.
64 @param maxConsecutiveCorruptSkip The maximum number of consecutive corrupt entries to skip
65 over when traversing entries via getNext()/getPrevious()
66 before declaring a failure.
67
68 NOTE: The Application MUST allocate storage for at least 2 entries
69 */
