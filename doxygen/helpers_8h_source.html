27/** This function performs a blocking wait by monitoring a Model Point for
28 a specific value. The method waits a specified amount of time BEFORE
29 checking for the expected value. The method returns true if the specified
30 MP's values matches the expected value; else the method will eventually time
31 out and return false.
32 */
33template <class MPTYPE, class VALTYPE>
34bool minWaitOnModelPoint( MPTYPE& src, VALTYPE expectedVal, unsigned long minWait, VALTYPE* lastVal = nullptr, unsigned long maxWaitMs = 10000, unsigned long waitDelayMs = 10 )
35{
36 Kit::System::sleep( minWait );
37
38 VALTYPE val;
39 bool valid = src.read( val );
40 while ( maxWaitMs )
41 {
42 if ( valid && val == expectedVal )
43 {
44 // Return the last read value
45 if ( lastVal )
46 {
47 *lastVal = val;
48 }
49 return true;
50 }
51
52 Kit::System::sleep( waitDelayMs );
53 maxWaitMs -= waitDelayMs;
54 valid = src.read( val );
55 }
56
57 // Return the last read value
58 if ( lastVal )
59 {
60 *lastVal = val;
61 }
62 return false;
63};

100/** This method is similar to minWaitOnModelPoint<>(), except that it waits on
101 valid/invalid state of the MP
102 */
103template <class MPTYPE>
104bool minWaitOnModelPointValidState( MPTYPE& src, bool expectedValidState, unsigned long minWait, int8_t* lastVal = nullptr, unsigned long maxWaitMs = 10000, unsigned long waitDelayMs = 10 )
105{
106 Kit::System::sleep( minWait );
107
108 bool state = !src.isNotValid();
109 while ( maxWaitMs )
110 {
111 if ( state == expectedValidState )
112 {
113 // Return the last read state
114 if ( lastVal )
115 {
116 *lastVal = state;
117 }
118 return true;
119 }
120
121 Kit::System::sleep( waitDelayMs );
122 maxWaitMs -= waitDelayMs;
123 state = !src.isNotValid();
124 }
125
126 // Return the last read value
127 if ( lastVal )
128 {
129 *lastVal = state;
130 }
131 return false;
132};

139{

158 ;
159 }
160
161 // Return the last read value

163 {
164 *newSeqNum = seqNum;
165 }

169/** This class is a EventQueue Server that signals the provided semaphore when
170 it begins execution. This is helpful when you need synchronize your test
171 with the EventQueue Server thread actually running
172
173 The class also supports 'Pausing' the EventQueue Server (see freeze() and thaw()
174 methods). When the server is paused the thread essentially stops executing -
175 providing a pseudo thread safe mechanism for the test manager to operate on
176 objects/data/methods that execute in the paused thread.
177 */
178class TestEventQueueServer : public Kit::EventQueue::Server
179{
180public:
181 /// Constructor
182 TestEventQueueServer( Kit::System::Semaphore& signalToNotify,
183 uint32_t timeOutPeriodInMsec = OPTION_KIT_SYSTEM_EVENT_LOOP_TIMEOUT_PERIOD,
184 Kit::Container::SList<Kit::System::IEventFlag>* eventFlagsList = nullptr,
185 Kit::System::IWatchedEventLoop* watchdog = nullptr )
186 : Server( timeOutPeriodInMsec, eventFlagsList, watchdog )
187 , m_sema( signalToNotify )
188 {
189 }
190
191public:
192 /// See Kit::System::IRunnable
193 void entry() noexcept override
194 {
195 m_sema.signal();
196 Kit::EventQueue::Server::entry();
197 }
198
199
200public:
201 /// See Kit::EventQueue::Server
202 void serverProcessEvents() noexcept override
203 {
204 if ( m_frozen )
205 {
206 // Wait to be signaled to resume the thread (i.e. thaw the thread)
207 m_semaPauseTread.wait();
208 m_frozen = false;
209
210 // Signal that I have resumed and processed events
211 m_sema.signal();
212 return;
213 }
214
215 // Call the base class to process events as normal
216 Kit::EventQueue::Server::serverProcessEvents();
217 }
218
219 /// This method freezes/Pauses the Mailbox
220 void freeze()
221 {
222 m_frozen = true;
223 this->signal(); // Wake up the thread so that it can process the frozen state
224 }
225
226 /** This method resumes/unfreezes the Mailbox. The method does NOT return
227 until the Server has resumed
228 */
229 void thaw()
230 {
231 // Signal the paused thread to wake up and then wait for it to actually wake up
232 m_semaPauseTread.signal();
233 m_sema.wait();
234 }
235
236public:
237 /// Semaphore to signal when I run
238 Kit::System::Semaphore& m_sema;
239
240 /// Semaphore used to pause the Mailbox
241 Kit::System::Semaphore m_semaPauseTread;
242
243 /// Freeze state
244 volatile bool m_frozen;
245};

248/** This class is used to 'convert' the synchronous ITC semantics of open/close
249 requests to asynchronous. The initial use case is when using Simulate
250 time and the thread calling the open/close is also the thread that is
251 responsible for advancing simulated time. There use case is essentially
252 a dead-lock scenario. By changing the open/close semantics to asynchronous
253 the dead-lock is avoid (albeit polling and an extra thread is required).
254 */
255class AsyncOpenClose : public Kit::System::IRunnable
256{
257public:
258 /// Synchronize open/close calls
259 Kit::System::Semaphore m_sema;
260
261 /// Synchronize the thread is ready (for an open call)
262 Kit::System::Semaphore m_semaReady;
263
264 /// Critical section for flags
265 Kit::System::Mutex m_lock;
266
267 /// The instance to be opened/closed
268 Kit::Itc::OpenCloseSync& m_target;
269
270 /// Optional open argument
271 void* m_openArgs;
272
273 /// Open/close state
274 volatile bool m_opened;
275
276 /// Option to exit the runnable object/thread when closeSubject() is called
277 volatile bool m_exitOnClose;
278
279public:
280 /// Constructor
281 AsyncOpenClose( Kit::Itc::OpenCloseSync& target )
282 : m_target( target )
283 , m_openArgs( nullptr )
284 , m_opened( false )
285 , m_exitOnClose( true )
286 {
287 }
288
289public:
290 /// Used to invoke the open() request using asynchronous semantics (with respect to the caller)
291 void openSubject( void* args = nullptr )
292 {
293 m_semaReady.wait();
294 m_openArgs = args;
295 m_sema.signal();
296 }
297
298 /// Used to invoke the close() request using asynchronous semantics (with respect to the caller)
299 void closeSubject( bool exitThreadOnClose = false )
300 {
301 m_exitOnClose = exitThreadOnClose;
302 m_sema.signal();
303 }
304
305 /// Test executive calls this method to get the status of open/close class
306 bool isOpened()
307 {
308 Kit::System::Mutex::ScopeLock criticalSection( m_lock );
309 return m_opened;
310 }
311
312protected:
313 /// See Kit::System::IRunnable
314 void entry() noexcept override
315 {
316 for ( ;; )
317 {
318 m_semaReady.signal();
319 m_sema.wait();
320 m_target.open( m_openArgs );
321 setState( true );
322 m_sema.wait();
323 m_target.close();
324 setState( false );
325 if ( m_exitOnClose )
326 {
327 break;
328 }
329 }
330 }
331
332 /// Helper method
333 inline void setState( bool opened )
334 {
335 Kit::System::Mutex::ScopeLock criticalSection( m_lock );
336 m_opened = opened;
337 }
338};

341/** This function performs a blocking wait by monitoring an input stream for
342 available data. The method waits a specified amount of time BEFORE
343 checking for available data. The method returns true if there is data
344 available; else the method will eventually time out and return false.
345 */
346inline bool minWaitOnStreamData( Kit::Io::IInput& stream, unsigned long minWait = 10, unsigned long maxWaitMs = 10000, unsigned long waitDelayMs = 10 )
347{
348 Kit::System::sleep( minWait );
349
350 bool avail = false;
351 while ( maxWaitMs && !avail )
352 {
353 Kit::System::sleep( waitDelayMs );
354 maxWaitMs -= waitDelayMs;
355 avail = stream.available();
356 }
357 return avail;
358};
