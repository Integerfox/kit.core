38/** This concrete template class implements the ITC messaging and Model Point
39 monitoring needed to provide a thread-safe/asynchronous interface for
40 Journal Records. A single Server instance manages a single EntryRecord
41 instance. Multiple EntryRecords can be managed by creating multiple Server
42 instances
43
44 NOTE: The EntryRecords and HeadRecords are NOT managed by the Kit::Persistence::Record::Server.
45 They are wholly managed by this class (Kit::Persistence::Record::Journal::Server)
46
47 Template Args:
48 ENTRY:= The Application specific class that defines an 'entry'
49
50 */
51template <class ENTRY>
52class Server : public Kit::Itc::OpenCloseSync,
53 public ReaderSync,
54 public ResetSync
55{
56public:
57 /** Constructor. The argument 'timingTickInMsec' specifies the timing
58 resolution that will be used for Kit::Timer::Local Timers.
59
60 NOTE: When the RingBufferMP 'incomingEntriesBuffer' is created it MUST
61 to be created with the 'initializeMemory' constructor flags set to
62 false - which is NOT the default behavior. This is because ENTRY
63 is not struct - but a class with a vtable pointer.
64 */
65 Server( Kit::EventQueue::IQueue& myEventQueue,
66 IEntry& entryRecord,
67 Kit::Container::RingBufferMP<ENTRY>& incomingEntriesBuffer ) noexcept
68 : OpenCloseSync( myEventQueue )
69 , ReaderSync( myEventQueue )
70 , ResetSync( myEventQueue )
71 , m_record( entryRecord )
72 , m_obBuffer( myEventQueue )
73 , m_buffer( incomingEntriesBuffer )
74 , m_opened( false )
75 {
76 m_obBuffer.setCallback<Server, &Server::bufferElementCountChanged>( this );
77 }
78
79public:
80 /// See Kit::Itc::IOpenRequest
81 void request( OpenMsg& msg ) noexcept override
82 {
83 if ( !m_opened )
84 {
85 m_opened = m_record.start( m_eventQueue );
86 if ( m_opened )
87 {
88 m_buffer.m_mpElementCount.attach( m_obBuffer );
89 }
90 }
91
92 msg.returnToSender();
93 }
94
95 /// See Kit::Itc::ICloseRequest
96 void request( CloseMsg& msg ) noexcept override
97 {
98 if ( m_opened )
99 {
100 m_opened = false;
101 m_buffer.m_mpElementCount.detach( m_obBuffer );
102 m_record.stop();
103 }
104
105 msg.returnToSender();
106 }
107
108public:

110 void request( RetrieveLatestMsg& msg ) noexcept override
111 {
112 ReadRequest::LatestPayload& payload = msg.getPayload();
113 //
114 payload.m_success = m_record.getLatest( payload.m_entryDst,
115 payload.m_markerEntryRetrieved );
116 msg.returnToSender();
117 }

129 }

170protected:
171 /// Callback when the 'trigger' MP changes
172 void bufferElementCountChanged( Kit::Dm::Mp::Uint32& mp, Kit::Dm::IObserver& observer ) noexcept
173 {
174 uint32_t count;
175 uint16_t seqNum = 0;
176 if ( mp.readAndSync( count, seqNum, observer ) && count > 0 )
177 {
178 // Temporarily disable callbacks to avoid un-necessary callbacks
179 mp.detach( m_obBuffer );
180
181 // Drain the buffer (but limit how many adds at one time) and write the entries to persistent storage
182 unsigned iterations = 0;
183 ENTRY entry;
184 while ( iterations < OPTION_KIT_PERSISTENCE_JOURNAL_SERVER_MAX_BATCH_WRITE && m_buffer.remove( entry ) )
185 {
186 hookAddingEntry( entry );
187 m_record.addEntry( entry );
188 iterations++;
189 count--;
190 }
191
192 // Re-enable callbacks to be notified of new (or still pending) entries
193 if ( count > 0 )
194 {
195 // If there are still pending entries, use SEQUENCE_NUMBER_UNKNOWN to get an immediate callback
196 seqNum = Kit::Dm::IModelPoint::SEQUENCE_NUMBER_UNKNOWN;
197 }
198 mp.attach( m_obBuffer, seqNum );
199 }
200 }
201
202protected:
203 /// Hook function called for each entry BEFORE it is added to the record.
204 virtual void hookAddingEntry( const ENTRY& entry ) noexcept
205 {
206 // Default does nothing, but child class can override to add functionality
207 }
208
209protected:
210 /// Indexed Entry Record that handles the actual work to read/write the data
211 IEntry& m_record;
212
213 /// Observer for change notification (to the RingBuffer)
214 Kit::Dm::ObserverCallback<Dm::Mp::Uint32> m_obBuffer;
215
216 /// The incoming entries RingBuffer
217 Kit::Container::RingBufferMP<ENTRY>& m_buffer;
218
219 /// Track my open state
220 bool m_opened;
221};
