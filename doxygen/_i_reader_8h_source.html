24/** This abstract class defines interface for reading 'entries' from a
25 a Journal Entry Record.
26
27 See the README.md file for more details about "Journal Records".
28
29 NOTE: This interface/class IS THREAD SAFE and can be called from any thread,
30 EXCEPT for the thread that the Journal::Server executes in.
31*/
32class IReader
33{
34public:
35 /** This method reads/retrieves the latest entry (from the list of Journal
36 Entries) stored in the persistent media. The method is synchronous in
37 that the method does not return until the entry has been 'read' from
38 the persistent media.
39
40 Returns true if the latest entry is valid and has been read into 'dst';
41 else false is returned.
42
43 NOTE: When retrieveLatest() returns false, this means there are NO entries
44 stored.
45
46 NOTE: 'dst' is ALWAYS updated EVEN if no entry was 'found', basically 'dst'
47 is used as a work buffer when traversing the list.
48 */
49 virtual bool retrieveLatest( IPayload& dst, IEntry::Marker_T& entryMarker ) noexcept = 0;
50
51public:
52 /** This method walks the entire 'list of entries' and returns the next
53 newer entry as specified by the 'newerThan' value. The traversal
54 starts with the 'beginHereMarker' The method is synchronous in that the
55 method does not return until the entry has been 'read' from the persistent
56 media.
57
58 Returns true if the 'next' entry was found and has been read into 'dst';
59 else false is returned.
60
61 NOTE: 'dst' is ALWAYS updated EVEN if no entry was 'found', basically 'dst'
62 is used as a work buffer when traversing the list.
63
64 */
65 virtual bool retrieveNext( uint64_t newerThanTimestamp,
66 const IEntry::Marker_T beginHereMarker,
67 IPayload& dst,
68 IEntry::Marker_T& entryMarker ) noexcept = 0;
69
70 /** This method is similar to retrieveNext(), except that it returns the next
71 oldest entry.
72 */
73 virtual bool retrievePrevious( uint64_t olderThanTimestamp,
74 const IEntry::Marker_T beginHereMarker,
75 IPayload& dst,
76 IEntry::Marker_T& entryMarker ) noexcept = 0;
77
78public:
79 /** This method can be used to read an entry by its 'entry index'. The
80 entry index is a zero based index.
81
82 Returns true if the entry at 'entryIndex' is a valid entry; else false
83 is returned. Note: if 'entryIndex' is out of range, false is returned.
84
85 NOTE: 'dst' is ALWAYS updated EVEN if no valid entry was 'found',
86 basically 'dst' is used as a work buffer when traversing the list.
87 */
88 virtual bool retrieveByEntryIndex( Size_T entryIndex,
89 IPayload& dst,
90 IEntry::Marker_T& entryMarker ) noexcept = 0;
91
92
93 /** This method returns the maximum allowed 'entryIndex' when calling
94 retrieveByEntryIndex().
95 */
96 virtual Size_T maxIndex() const noexcept = 0;
97};
