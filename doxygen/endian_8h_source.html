11/** @file
12
13 This file provides a set of macros/method to convert between the 'host'
14 platform and Big/Little endian multi-byte layouts. NOTE: The application
15 should ALWAYS use the macros, i.e. never directly call the internal
16 methods.
17
18 Usage:
19 The application defines one of the following macros based on the endianess
20 of the platform that the application executes on:
21 USE_KIT_TYPE_ENDIAN_LE_HOST
22 USE_KIT_TYPE_ENDIAN_BE_HOST
23
24 The code is derived from the public domain source in the repo:
25 https://github.com/rofl0r/endianness.h/tree/master
26*/

129/// Internal method. Do not call directly - always use the macros above
130inline uint16_t endianBswap16( uint16_t x )
131{
132 return ( x << 8 ) | ( x >> 8 );
133}
134
135/// Internal method. Do not call directly - always use the macros above
136inline uint32_t endianBswap32( uint32_t x )
137{
138 return ( x >> 24 ) | ( x >> 8 & 0xff00 ) | ( x << 8 & 0xff0000 ) | ( x << 24 );
139}
140
141/// Internal method. Do not call directly - always use the macros above
142inline float endianBswapF32( float x )
143{
144 uint32_t tmp;
145 memcpy( &tmp, &x, sizeof( tmp ) );
146 tmp = endianBswap32( tmp );
147 float result;
148 memcpy( &result, &tmp, sizeof( result ) );
149 return result;
150}
151
152/// Internal method. Do not call directly - always use the macros above
153inline uint64_t endianBswap64( uint64_t x )
154{
155 return ( ( endianBswap32( (uint32_t)x ) + 0ULL ) << 32 ) | ( endianBswap32( x >> 32 ) );
156}
157
158/// Internal method. Do not call directly - always use the macros above
159inline double endianBswapF64( double x )
160{
161 uint64_t tmp;
162 memcpy( &tmp, &x, sizeof( tmp ) );
163 tmp = endianBswap64( tmp );
164 double result;
165 memcpy( &result, &tmp, sizeof( result ) );
166 return result;
167}
