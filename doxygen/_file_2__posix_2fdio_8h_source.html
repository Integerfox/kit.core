34/** This static class provides a collection of functions for operating on POSIX
35 file descriptors.
36 NOTE: All file name/paths MUST be in native format (i.e. use '/' as the
37 directory separator)
38 */
39class PosixFileIO : public Kit::Io::Stdio::PosixIO
40{
41public:
42 /** Opens a file. The default arguments are set for:
43 o Read/Write access
44 o Do NOT create the file if the file does not exist
45 o Do NOT truncate the file contents
46 */
47 static KitIoFileHandle_T open( const char* fileEntryName, bool readOnly = true, bool forceCreate = false, bool forceEmptyFile = false ) noexcept
48 {
49 KIT_SYSTEM_ASSERT( fileEntryName != nullptr );
50
51 // Set open flags as requested
52 int flags = readOnly ? O_RDONLY : O_RDWR;
53 int mode = 0666;
54 if ( forceCreate )
55 {
56 flags |= O_CREAT;
57 }
58 if ( forceEmptyFile )
59 {
60 flags |= O_TRUNC;
61 }
62
63
64 // Open the file
65 errno = 0;
66 KitIoFileHandle_T fd( ::open( fileEntryName, flags, mode ) );
67 return fd;
68 }
69
70public:
71 /** Returns the length, in bytes, of the file. If there is an error than
72 false is returned. Note: the implementation of this method is brute force
73 in that it sets the file position indicator to EOF and uses the EOF byte
74 offset as the length. The file position indicator is restored to its
75 original position before returning.
76 */
77 static bool length( KitIoFileHandle_T fd, ByteCount_T& length ) noexcept
78 {
79 if ( fd == INVALID_FD )
80 {
81 return false;
82 }
83
84 off_t cur = lseek( fd, 0, SEEK_CUR );
85 off_t eof = lseek( fd, 0, SEEK_END );
86 off_t restore = lseek( fd, cur, SEEK_SET );
87 length = (ByteCount_T)eof;
88 return ( cur != (off_t)-1 ) && ( eof != (off_t)-1 ) && ( restore != (off_t)-1 );
89 }
90
91public:
92 /** Returns the current file pointer offset, in bytes, from the top of the
93 file. If there is an error than false is returned.
94 */
95 inline static bool currentPos( KitIoFileHandle_T fd, ByteCount_T& currentPos ) noexcept
96 {
97 if ( fd == INVALID_FD )
98 {
99 return false;
100 }
101
102 off_t pos = lseek( fd, 0, SEEK_CUR );
103 currentPos = (ByteCount_T)pos;
104 return pos != (off_t)-1;
105 }
106
107 /** Adjusts the current pointer offset by the specified delta (in bytes).
108 Returns true if successful, else false (i.e. setting the pointer
109 past/before the file boundaries).
110 */
111 inline static bool setRelativePos( KitIoFileHandle_T fd, ByteCount_T deltaOffset ) noexcept
112 {
113 if ( fd == INVALID_FD )
114 {
115 return false;
116 }
117
118 off_t pos = lseek( fd, (off_t)deltaOffset, SEEK_CUR );
119 return pos != (off_t)-1;
120 }
121
122 /** Sets the file pointer to the absolute specified offset (in bytes).
123 Returns true if successful, else false (i.e. setting the
124 pointer past the end of the file).
125 */
126 inline static bool setAbsolutePos( KitIoFileHandle_T fd, ByteCount_T newoffset ) noexcept
127 {
128 if ( fd == INVALID_FD )
129 {
130 return false;
131 }
132
133 off_t pos = lseek( fd, (off_t)newoffset, SEEK_SET );
134 return pos != (off_t)-1;
135 }
136
137 /** Sets the file pointer to End-Of-File. Returns true if successful, else
138 false if an error occurred.
139 */
140 inline static bool setToEof( KitIoFileHandle_T fd ) noexcept
141 {
142 if ( fd == INVALID_FD )
143 {
144 return false;
145 }
146
147 return lseek( fd, 0, SEEK_END ) != (off_t)-1;
148 }

151 /** Returns information about the file system entry. If there is any
152 error, the function returns false; else true is returned.
153 */
154 inline static bool getInfo( const char* fsEntryName, struct stat& statOut ) noexcept
155 {
156 KIT_SYSTEM_ASSERT( fsEntryName != nullptr );
157 return stat( fsEntryName, &statOut ) == 0;
158 }
159
160 /** Creates a new, empty file. If the file already exists, the call
161 fails. Returns true if successful.
162 */
163 static bool createFile( const char* fileName ) noexcept
164 {
165 KIT_SYSTEM_ASSERT( fileName != nullptr );
166
167 bool result = false;
168 int fd = ::open( fileName, O_RDWR | O_CREAT | O_EXCL, 0666 );
169 if ( fd != -1 )
170 {
171 result = true;
172 ::close( fd );
173 }
174
175 return result;
176 }
177
178 /** Creates a new directory. If the directory already exists, the call fails.
179 Returns true if successful.
180 */
181 static bool createDirectory( const char* dirName ) noexcept
182 {
183 KIT_SYSTEM_ASSERT( dirName != nullptr );
184 return mkdir( dirName, 0755 ) == 0;
185 }
186
187public:
188 /** Renames and/or moves a file. Returns true if successful.
189 */
190 static bool move( const char* oldName, const char* newName ) noexcept
191 {
192 KIT_SYSTEM_ASSERT( oldName != nullptr );
193 KIT_SYSTEM_ASSERT( newName != nullptr );
194 return ::rename( oldName, newName ) == 0;
195 }
196
197 /** Removes a file or directory. The directory must be empty to be removed.
198 Returns true if successful.
199 */
200 static bool remove( const char* fsEntryName )
201 {
202 KIT_SYSTEM_ASSERT( fsEntryName != nullptr );
203 return ::remove( fsEntryName ) == 0;
204 }

206public:
207 /** Opens a directory - for 'walking' its contents. Each successful 'open'
208 call must be followed (at some point) by a call to closeDirectory()
209 */
210 static bool openDirectory( KitIoFileDirectory_T& hdl, const char* dirName ) noexcept
211 {
212 KIT_SYSTEM_ASSERT( dirName != nullptr );
213 hdl = opendir( dirName );
214 return hdl != nullptr;
215 }
216
217 /** Closes a directory. Can only be called after a successful call to
218 openDirectory()
219 */
220 static void closeDirectory( KitIoFileDirectory_T& hdl ) noexcept
221 {
222 closedir( hdl );
223 }
224
225 /** Reads 'next' entry in the directory. The entry name is copied into
226 'dstEntryName'. If there are no more entries in the directory, then
227 'dstEntryName' is set to an empty string. The 'maxEntryNameSize'
228 argument specifies the size - which needs to include space for the null
229 terminator - of the 'dstEntryName' buffer.
230
231 Returns true if successful; else false if a file system error was
232 encountered.
233 */
234 static bool readDirectory( KitIoFileDirectory_T& hdl, NameString& dstEntryName ) noexcept
235 {
236
237 errno = 0;
238 struct dirent* entryPtr = readdir( hdl );
239 if ( entryPtr != nullptr )
240 {
241 dstEntryName = entryPtr->d_name;
242 return true;
243 }
244 else
245 {
246 // No more entries
247 if ( errno == 0 )
248 {
249 dstEntryName.clear();
250 return true;
251 }
252 }
253
254 // File system error
255 return false;
256 }
