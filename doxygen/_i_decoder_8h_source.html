21/** This abstract class defines an interface for 'receiving' (decoding) a frame.
22 See the README.md file for details about a what a Frame is
23 */
24class IDecoder
25{
26public:
27 /** This method reads from an Input source (which is defined/provided by
28 the concrete implementation) until a valid frame is found or an error
29 occurred (i.e. a blocking call). If a valid frame was found, true will
30 be returned and the frame will be stored in 'frameBuffer'. The length,
31 in bytes, of the frame found is returned via 'frameSize'. False is
32 returned if a error was encountered while reading the Input source.
33 */
34 virtual bool scan( Kit::Type::SSize_T maxSizeOfFrameBuffer,
35 uint8_t* frameBuffer,
36 Kit::Type::SSize_T& frameSize ) noexcept = 0;
37
38 /** This method is similar to the above scan() method, except that it does
39 NOT block till a 'frame' has found, instead it indicates when a 'frame'
40 has been found by setting the 'isEof' flag to true.
41
42 False is returned if a error was encountered while reading the Input
43 source.
44 */
45 virtual bool scan( Kit::Type::SSize_T maxSizeOfFrameBuffer,
46 uint8_t* frameBuffer,
47 Kit::Type::SSize_T& frameSize,
48 bool& isEof ) noexcept = 0;
49
50public:
51 /** This method allows 'out-of-band' reading of the input source. If the
52 scanner is in-a-frame the method return false and does nothing.
53
54 CAUTION: Most client/consumers of the decoder should never use this
55 method. This method only has meaning/usefulness when the
56 application KNOWS when the input source is NOT in a frame AND
57 that there is 'non-framed' data that can be consumed.
58
59 Attempts to read the specified number of bytes from the stream into the
60 supplied buffer. The caller is responsible for ensuring that the buffer
61 is large enough to hold 'numBytes'. The actual number of bytes read is
62 returned via 'bytesRead'.
63
64 Returns true if successful, or false if End-of-Stream was encountered or
65 if the input source is in a frame.
66 */
67 virtual bool oobRead( uint8_t* dstBuffer,
68 Kit::Type::SSize_T numBytes,
69 Kit::Type::SSize_T& bytesRead ) noexcept = 0;
70
71public:
72 /// Virtual Destructor
73 virtual ~IDecoder() = default;
74};
