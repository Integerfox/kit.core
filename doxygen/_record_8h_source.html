30/** This mostly concrete class implements the Kit::Persistence::Record::IRecord
31 interface where a Record instance contains the data from N model points. A
32 final child class is needed to provide the specifics of 'resetting' the
33 Record's data.
34
35 NOTE: There can be multiple MPs that make up the Record and it is possible
36 to get change notification for all MPs, i.e. change notifications
37 are not atomic to the SET of MPs that make up Record. However, the
38 application can enable the 'settling time' feature of the Record to
39 minimize the number of 'extra' NVRAM updates.
40 */
41class Record : public Kit::Persistence::Record::IRecord,
42 public Kit::Persistence::Record::IPayload,
43 public Kit::Dm::Persistence::IManageRequest,
44 public Kit::System::Timer
45{
46public:
47 /** This data structure associates a Data Model subscriber instance with a
48 Model Point instance.
49 */
50 struct Item_T
51 {
52 Kit::Dm::IModelPoint* mpPtr; //!< Pointer to an allocated Model Point
53 Kit::Dm::ObserverCallback<IModelPoint>* observerPtr; //!< Pointer to an allocated observer, or nullptr if no runtime monitor of changes to the MP is desired to trigger an Record update
54 };
55
56 /// Magic value for no-change notifications for a MP in the Item_T list
57 static constexpr Kit::Dm::IObserver* NO_SUBSCRIBER = nullptr;

59public:
60 /** Constructor. The 'itemList' is array of model points/observer tuples.
61 The application is responsible for ALLOCATING all of the Model Point
62 and Observer instances in the itemList.
63
64
65 If the schema indexes does not match when reading data from persistent
66 storage, THEN the `schemaChange()` method is called. If the `schemaChange()`
67 method returns false, then the record is defaulted.
68
69 The 'writeDelayMs' and 'maxWriteDelayMs' arguments are used to impose
70 a 'settling time' after an model point is updated before writing the
71 change(s) to NVRAM. This is useful when multiple MP changes at the
72 'same' time - so that there is single update to NVRAM instead of
73 attempting multiple NVRAM writes for each MP change notification.
74 */
75 Record( Item_T itemList[],
76 unsigned numItems,
77 Kit::Persistence::Record::IChunk& chunkHandler,
78 uint8_t schemaMajorIndex,
79 uint8_t schemaMinorIndex,
80 uint32_t writeDelayMs = 0,
81 uint32_t maxWriteDelayMs = 0 ) noexcept
82 : Kit::System::Timer( nullptr )
83 , m_myEventQueuePtr( nullptr )
84 , m_items( itemList )
85 , m_chunkHandler( chunkHandler )
86 , m_delayMs( writeDelayMs )
87 , m_maxDelayMs( maxWriteDelayMs )
88 , m_timerMarker( 0 )
89 , m_numItems( numItems )
90 , m_major( schemaMajorIndex )
91 , m_minor( schemaMinorIndex )
92 , m_started( false )
93 {
94 KIT_SYSTEM_ASSERT( itemList );
95 KIT_SYSTEM_ASSERT( numItems > 0 );
96 }

149 /** This method is responsible for updating all of the Model Points in
150 record to their default values. This method is called when there is
151 NO valid data when reading the record's data from persistence storage.
152
153 When the method returns true, the 'reset data' is written to persistence
154 storage; else the persistence storage is NOT updated and NO further updates
155 of the persistence storage will be allowed.
156 */
157 virtual bool resetData() noexcept = 0;
158
159 /** This method is called when the stored record data has different
160 schema indexes. It is up to the child record class on what to do when
161 the schema is different. The default behavior is default all of the
162 data.
163
164 The 'src' argument contains the record data (as stored in persistent
165 storage). The child class can make use of the data as needed. Note:
166 the data in 'src' is AFTER the schema indexes fields.
167
168 The method should return true if the child class has successfully
169 processed in the incoming data. Returning false will cause the
170 record data to be defaulted;
171 */
172 virtual bool schemaChange( uint8_t previousSchemaMajorIndex,
173 uint8_t previousSchemaMinorIndex,
174 const void* src,

177 return false;
