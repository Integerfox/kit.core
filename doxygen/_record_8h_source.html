30/** This mostly concrete class implements the Kit::Persistent::Record interface
31 where a Record instance contains the data from N model points. A final
32 child class is needed to provide the specifics of 'resetting' the Record's
33 data.
34 */
35class Record :
36 public Kit::Persistent::Record,
37 public Kit::Persistent::Payload,
38 public FlushRequest,
39 public EraseRequest,
40 public Kit::System::Timer
41{
42public:
43 /** This data structure associates a Data Model subscriber instance with a
44 Model Point instance.
45 */
46 typedef struct
47 {
48 Kit::Dm::ModelPoint* mpPtr; //!< Pointer to a Model Point
49 Kit::Dm::SubscriberComposer<Record, Kit::Dm::ModelPoint>* observerPtr; //!< Place holder to for dynamically allocated Subscriber Instance. Note: A ZERO value will not allocate a subscriber
50 } Item_T;
51
52public:
53 /** Constructor. The 'itemList' is variable length array where the last item
54 in the list MUST be a 'null' entry, e.g. {0,0}. The application is
55 responsible for ALLOCATING all of the Model Point and Subscriber
56 instances in the itemList.
57
58 If the schema indexes does not match when reading data from persistent
59 storage, THEN the upgrade method is called.record data is defaulted. If the schema minor index from
60 persistent storage does is less than 'schemaMinorIndex' argument, the
61 the record is data loaded and the upgrade() method is called. If the
62 schema minor index from persistent storage is greater than the
63 'schemaMinorIndex' argument, THEN the record data is defaulted.
64
65 The 'writeDelayMs' and 'maxWriteDelayMs' arguments are used to impose
66 a 'settling time' after an model point is updated before writing the
67 change(s) to NVRAM. This is useful when multiple MP changes at the
68 'same' time - so that there is single update to NVRAM instead of
69 attempting multiple NVRAM writes for each MP change notification.
70 */
71 Record( Item_T itemList[],
72 Kit::Persistent::Chunk& chunkHandler,
73 uint8_t schemaMajorIndex,
74 uint8_t schemaMinorIndex,
75 uint32_t writeDelayMs = 0,
76 uint32_t maxWriteDelayMs = 0 ) noexcept;

119protected:
120 /** This method is responsible for updating all of the Model Points in
121 record to their default values. This method is called when there is
122 NO valid data when reading the record's data from persistence storage.
123
124 When the method returns true, the 'reset data' is written to persistence
125 storage; else the persistence storage is NOT updated and NO further updates
126 of the persistence storage will be allowed.
127 */
128 virtual bool resetData() noexcept = 0;
129
130 /** This method is called when the stored record data has different
131 schema indexes. It is up to the child record class on what to do when
132 the schema is different. The default behavior is default all of the
133 data.
134
135 The 'src' argument contains the record data (as stored in persistent
136 storage). The child class can make use of the data as needed. Note:
137 the data in 'src' is AFTER the schema indexes fields.
138
139 The method should return true if the child class has successfully
140 processed in the incoming data. Returning false will cause the
141 record data to be defaulted;
142 */
143 virtual bool schemaChange( uint8_t previousSchemaMajorIndex,
144 uint8_t previousSchemaMinorIndex,
145 const void* src,
146 size_t srcLen ) noexcept {
147 return false;
148 }
