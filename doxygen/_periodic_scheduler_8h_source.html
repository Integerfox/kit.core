28/** This concrete class is a 'policy' object that is used to add polled based,
29 cooperative monotonic scheduling to a Runnable object. Monotonic in this
30 content means that if a 'interval' method is scheduled to execute at a
31 periodic rate of 200Hz, then the method will be called on every 5ms boundary
32 of the system time.
33
34 The scheduler makes its best attempt at being monotonic and deterministic,
35 BUT because the scheduling is polled and cooperative (i.e. its is the
36 application's responsibility to not-overrun/over-allocate the processing
37 done during each interval) the actual the timing cannot be guaranteed.
38 That said, the scheduler will detect and report when the interval timing
39 slips.
40
41 The usage of this class is to implement (or extended) a Cpl::System::Runnable
42 object, i.e. invoked inside the 'forever' loop of the Runnable object's appRun()
43 method. The scheduler can be use within a 'bare' forever loop and it can be
44 used with existing event based Runnable objects to add periodic scheduling.
45 See Cpl::Dm::PeriodicScheduler for example extending an event based Runnable
46 object to include periodic scheduling.
47
48 NOTE: A uint64_t is used for the millisecond time keeping because it
49 provides a larger range and can accommodate long-running applications
50 (i.e. a uint32_t rolls over every 49.7 days; a uint64_t rolls over
51 every 584 million years).
52
53 NOTE: A "traditional C function pointer" paradigm is used to provide
54 application specialization. This is by design on the premise that
55 it simplifies the Application "main" in that 'start-up child classes'
56 are not required to be created for each interval and/or for the
57 scheduler hooks.
58 */
59class PeriodicScheduler
60{
61public:
62 /** Definition for an interval method. This method is called when the
63 period time has expired for the interval.
64
65 Where:
66 currentTick - is current system when the interval method is called
67 currentInterval - is the deterministic interval boundary that is being
68 logically executed.
69 context - Application defined argument/value passed to the
70 interval method.
71
72 Example:
73 Given a interval method is scheduled to execute at 10Hz and the
74 current system time in the Runnable object's forever loop is 10.212
75 seconds when the interval method is called THEN:
76
77 currentTick:= 10212 ms
78 currentInterval:= 10200 ms
79 */
80 typedef void ( *IntervalCallbackFunc_T )( uint64_t currentTick,
81 uint64_t currentInterval,
82 void* context );
83
84 /** Defines an interval. The application should treat this data struct as
85 an opaque structure.
86 */
87 struct Interval_T

89 IntervalCallbackFunc_T callbackFunc; //!< Callback function pointer
90 void* context; //!< Optional Context for the callback. The callback function is required to 'understand' the actual type of the context pointer being passed to it.
91 uint64_t duration; //!< The amount time in the Interval's period.
92 uint64_t timeMarker; //!< Internal Use ONLY: Marks the last time the interval executed
93
94 /// Constructor
95 Interval_T( IntervalCallbackFunc_T callbackFunc,

101 , timeMarker( 0 )

103 }

107 /** Defines the method that is used to report to the Application when an
108 Interval does not execute 'on time'
109
110 Where:
111 intervalThatSlipped - Reference to the interval that did not execute on time
112 currentTick - The system time when the scheduler executed the interval
113 missedInterval - The interval boundary that was missed
114 intervalContext - Application defined argument/value that was
115 passed to the interval method.
116
117 */
118 typedef void ( *ReportSlippageFunc_T )( Interval_T& intervalThatSlipped,

120 uint64_t missedInterval );

136 /** Defines an optional method that is called every time the Runnable's
137 object executes its event/scheduling loop. NOTE: The method is not
138 used by the PeriodicScheduler object directly - if used - it is the
139 responsibility of the Runnable object (that is calling the scheduler) to
140 ensure it is called at the appropriate times.
141 */
142 typedef void ( *IdleFunc_T )( uint64_t currentTick, bool atLeastOneIntervalExecuted );

145public:
146 /** Constructor. The application provides a variable length array of interval
147 definitions that will be scheduled. The last entry in the
148 array must contain a 'null' Interval_T definition (i.e. all fields
149 set to zero). The Scheduler assumes that each Interval_T definition
150 has a unique period time.
151
152 The individual 'intervals' MUST be initialized (either statically or
153 by calling initializeInterval()) before creating the scheduler.
154
155 Notes:
156 - When extending a event based (i.e. inherits from EventLoop) runnable
157 object with scheduling, the Application should define all of the
158 scheduled Intervals to have period times that are multiples of
159 the EventLoop's 'timeOutPeriodInMsec' constructor value.
160 */
161 PeriodicScheduler( Interval_T intervals[],
162 Hook_T beginThreadProcessing = nullptr,
163 Hook_T endThreadProcessing = nullptr,

165 NowFunc_T nowFunc = ElapsedTime::millisecondsEx ) noexcept;
166
167 /// Virtual destructor

171public:
172 /** This method is used to invoke the scheduler. When called - zero or more
173 Interval definitions will be executed. The method returns true if at
174 least one Interval was executed.
175
176 If a scheduled Interval does not execute 'on time', then the reportSlippage()
177 method will called. It is the Application's to decide (what if anything)
178 is done when there is slippage in the scheduling. The slippage is reported
179 AFTER the Interval's IntervalCallbackFunc_T is called.
180 */

190 */

197 */
