26/** This concrete class is a 'server' that responds to requests from a 'client'
27 to generate. See the README.md file for more details
28 */
29class Server : public Kit::Itc::OpenCloseSync
30{
31public:
32 /// Constructor
33 Server( Kit::EventQueue::IQueue& myMbox,
34 MpFoo& mpFoo )
35 : Kit::Itc::OpenCloseSync( myMbox )
36 , m_mpFoo( mpFoo )
37 , m_obFoo( myMbox )
38 , m_opened( false )
39 {
40 m_obFoo.setCallback<Server, &Server::triggerChanged>( this );
41 }
42
43public:
44 /// See Kit::Itc::OpenCloseSync
45 void request( OpenMsg& msg ) noexcept override
46 {
47 // Only 'open' when not already opened
48 if ( !m_opened )
49 {
50 // Housekeeping
51 m_opened = true;
52
53 // Subscribe for the 'trigger' MP change notification.
54 m_mpFoo.attach( m_obFoo );
55 }
56
57 // Return the message to the sender since we are done processing the request
58 msg.returnToSender();
59 }
60
61 /// See Kit::Itc::OpenCloseSync
62 void request( CloseMsg& msg ) noexcept override
63 {
64 // Only 'close' when already opened
65 if ( m_opened )
66 {
67 // Housekeeping
68 m_opened = false;
69
70 // Make sure the all of the potential callbacks get cancelled
71 m_mpFoo.detach( m_obFoo );
72 }
73
74 // Return the message to the sender since we are done processing the request
75 msg.returnToSender();
76 }
77
78
79protected:
80 /// Callback when the 'trigger' MP changes
81 void triggerChanged( MpFoo& mpThatChanged, Kit::Dm::IObserver& observer ) noexcept
82 {
83 Foo_T data;
84 if ( mpThatChanged.readAndSync( data, observer ) )
85 {
86 // 'Alive' feedback when running on hardware
87 Bsp_toggle_debug1();
88
89 // Generate a Random number between the upper and lower limits and write to the MP
90 // TODO: Need a Random Number interface
91 data.randomNumber = ( ( data.upperLimit - data.lowerLimit ) / 2 ) + data.lowerLimit;
92 data.isValid = true; // For this example - the result is always valid (because I am lazy)
93 KIT_SYSTEM_TRACE_MSG( "main",
94 "SERVER: request. upper=%" PRId32 ", lower=%" PRId32 ", result=%" PRId32 ", isValid=%s",
95 data.upperLimit,
96 data.lowerLimit,
97 data.randomNumber,
98 data.isValid ? "true" : "false" );
99
100 // Update the MP with response AND make sure that a change notification
101 // gets generated (i.e. the 'true' argument). Why is the 'force'
102 // required? It's because change notifications are only generated when
103 // the MP value is changed. It is theoretically possible that the
104 // server's random-number/valid-invalid state values do NOT change
105 // the MP actual value. Hence, the 'force' argument is used to
106 // make sure that the Client gets notified of the response.
107 mpThatChanged.detach( m_obFoo ); // Detach so we don't get callback for WRITE
108 uint16_t seqNum = mpThatChanged.write( data, true );
109 mpThatChanged.attach( m_obFoo, seqNum ); // Re-attach for the next request
110 }
111 }
112
113protected:
114 /// MP for the Client/Server transaction
115 MpFoo& m_mpFoo;
116
117 /// Observer callback for the MP change notification
118 Kit::Dm::ObserverCallback<MpFoo> m_obFoo;
119
120 /// Track my open/closed
121 bool m_opened;
122};
