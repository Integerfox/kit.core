22/** This abstract class is used 'extend' the Memory Allocator to have contiguous
23 semantics. This means all memory that is allocated can be viewed as
24 single contiguous block of memory. In addition, the 'release()' method is
25 hidden and replaced with a reset() method that releases/frees ALL memory
26 allocated.
27*/
28class ContiguousAllocator : public Allocator
29{
30public:
31 /// Structure for memory info/status
32 struct MemoryInfo_T
33 {
34 /// Pointer to the start of the contiguous memory block
35 void* m_memoryStartPtr;
36
37 /// Total size of the contiguous block memory used by allocator in bytes
38 size_t m_totalLenInBytes;
39
40 /// Maximum number of 'words' in the block of memory. Note: Unit is 'words' NOT bytes (i.e. word size is defined by wordSize() method)
41 size_t m_maxWords;
42
43 /// Number of remaining/free 'words' in the block of memory.
44 size_t m_freeWords;
45 };
46
47public:
48 /** Resets the allocator, i.e. effectively frees all allocated memory. It
49 is the caller's RESPONSIBILTY to ensure that it is kosher to free all
50 of the memory.
51 */
52 virtual void reset() noexcept = 0;
53
54public:
55 /** This method returns information about Allocator's contiguous memory
56 and allocate/free status. The return value is reference to the 'dstInfo'
57 argument.
58
59 NOTE: This method has GREAT POWER and it is the responsibility of the
60 Application to use it correctly (Uncle Ben 2002).
61 */
62 virtual MemoryInfo_T& getMemoryStart( MemoryInfo_T& dstInfo ) noexcept = 0;
63
64
65private:
66 /** Hide/disable the individual release method (it does not fit the
67 contiguous semantics). If it gets called via the parent 'Allocator'
68 interface - it does NOTHING
69 */
70 void release( void* ptr ) noexcept {};
71};
