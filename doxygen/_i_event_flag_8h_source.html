23/** This abstract class defines the call-back interface when an Event Flag
24 has signaled. Event Flags can be signaled from thread or from interrupt
25 service routines. There is no inherent ordering of Event Flag callbacks
26 when multiple Event Flags are signaled.
27
28 NOTE: Individual Event Flag instances can 'register' to be called back for
29 multiple 'events'. In addition, multiple Event Flag instances can
30 register for the same event.
31
32 The Event Flag interface IS inherently thread safe.
33 */
34class IEventFlag : public Kit::Container::ListItem
35{
36public:
37 /** This method returns a event-flag-mask that indicates which event flag(s)
38 are 'processed' by the event flag instance. At least one bit should be
39 set in the returned mask. The bit positions in the mask correspond to the
40 event flag numbers. For example:
41 o If the event flag instance processes only event flag 0, then the
42 method should return 0x00000001
43 o If the event flag instance processes only event flags 0, 1, and
44 4, then the method should return 0x00000013
45 o If the event flag instance processes all event flags, then the
46 method should return 0xFFFFFFFF
47 */
48 virtual uint32_t getEventFlagsMask() const noexcept = 0;
49
50 /** This method is called when the Event Flag is signaled. It executes
51 in the context of the thread of the IEventManager that was signaled.
52
53 'eventMask' is a bit mask with one or more bits set indicating
54 which events were signaled. The 'eventMask' will only contain 'set'
55 bits for bit positions as returned by getEventFlagsMask().
56 */
57 virtual void notified( uint32_t eventMask ) noexcept = 0;
58
59public:
60 /// Virtual Destructor
61 virtual ~IEventFlag() = default;
62};
63
64/** This is helper class that perform an auto-registration of the Event Flag
65 instance with a list of Event Flags
66
67 The interface is NOT inherently thread safe.
68 */
69class EventFlagBase : public IEventFlag

71protected:

86protected:

89};
