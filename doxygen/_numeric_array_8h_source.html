122 /// See Kit::Dm::Point.
123 bool setJSONVal( JsonDocument& doc ) noexcept override
124 {
125 JsonObject obj = doc.createNestedObject( "val" );
126 JsonArray arr = obj.createNestedArray( "elems" );
127 ELEMTYPE* elemPtr = (ELEMTYPE*)m_dataPtr;
128 for ( size_t i = 0; i < m_numElements; i++ )
129 {
130 arr.add( elemPtr[i] );
131 }
132 return true;
133 }

135public:
136 /// See Kit::Dm::Point.
137 bool fromJSON_( JsonVariant& src, Kit::Dm::IModelPoint::LockRequest_T lockRequest, uint16_t& retSequenceNumber, Kit::Text::IString* errorMsg ) noexcept override
138 {
139
140 // Check for object
141 if ( src.is<JsonObject>() == false )
142 {
143 if ( errorMsg )
144 {
145 *errorMsg = "'val' key/value pair is NOT an JSON object";
146 }
147 return false;
148 }
149
150 // Check for embedded array
151 JsonArray elems = src["elems"];
152 if ( elems.isNull() )
153 {
154 if ( errorMsg )
155 {
156 *errorMsg = "'val' key/value pair is missing the embedded 'elems' array";
157 }
158 return false;
159 }
160
161 // Get starting index (note: if not present a default of '0' will be returned)
162 size_t startIdx = src["start"] | 0;
163
164 // Check for exceeding array limits
165 size_t numElements = elems.size();
166 if ( numElements + startIdx > m_numElements )
167 {
168 if ( errorMsg )
169 {
170 errorMsg->format( "Number of array elements ([%zu+%zu)] exceeds the MP's element count (%zu)", startIdx, numElements, m_numElements );
171 }
172 return false;
173 }
174
175 // Is the element syntactically correct?
176 if ( elems[0].is<ELEMTYPE>() == false )
177 {
178 if ( errorMsg )
179 {
180 errorMsg->format( "Failed parsing element[0]. Content of the MP is suspect!" );
181 }
182 return false;
183 }
184
185 // Update the Model Point in 'M' elements at a time (helps to reduce 'noise' on the MP's sequence number)
186 size_t offset = 0;
187 while ( numElements )
188 {
189 // Attempt to parse the value key/value pair (as a simple numeric)
190 ELEMTYPE tempArray[OPTION_KIT_DM_MP_ARRAY_TEMP_ARRAY_NUM_ELEMENTS];
191 size_t idx;
192 for ( idx = 0; idx < numElements && idx < OPTION_KIT_DM_MP_ARRAY_TEMP_ARRAY_NUM_ELEMENTS; idx++ )
193 {
194 tempArray[idx] = elems[idx + offset].as<ELEMTYPE>();
195 }
196 retSequenceNumber = ArrayBase::writeArrayElements( tempArray, idx, startIdx + offset, lockRequest );
197 offset += idx;
198 numElements -= idx;
199 }
200
201 return true;
202 }
203
204protected:
205 /// The data store the MP
206 ELEMTYPE m_data[NUMELEMS];
207};
