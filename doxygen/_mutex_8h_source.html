25/** This concrete mutex class defines the interface for a mutex that has
26 "recursive" semantics. Recursive semantic allows the thread that owns the
27 mutex to acquire ownership multiple times by calling lock() multiple times.
28 There must be a corresponding number of unlock() calls made before the
29 ownership of the mutex is released.
30 */
31class Mutex
32{
33public:
34 /// Constructor
35 Mutex() noexcept;
36
37 /// Destructor
38 ~Mutex() noexcept;
39
40public:
41 /** This method is invoked prior to entering a critical section. If another
42 thread currently "owns" the mutex, the current thread will "wait" until
43 it can obtain ownership before proceeding.
44 */
45 void lock() noexcept;
46
47 /** This method is invoke at the end of a critical section. This call will
48 release the ownership of the mutex.
49 */
50 void unlock() noexcept;
51
52
53protected:
54 /// Raw Mutex handle/instance/pointer
55 KitSystemMutex_T m_mutex;
56
57
58private:
59 /// Prevent access to the copy constructor -->mutexes can not be copied!
60 Mutex( const Mutex& m ) = delete;
61
62 /// Prevent access to the assignment operator -->mutexes can not be copied!
63 const Mutex& operator=( const Mutex& m ) = delete;
64
65 /// Prevent access to the move constructor -->Containers can not be implicitly moved!
66 Mutex( Mutex&& m ) = delete;
67
68 /// Prevent access to the move assignment operator -->Containers can not be implicitly moved!
69 Mutex& operator=( Mutex&& m ) = delete;
70
71public:
72 /** This concrete class provides a simple mechanism for providing mutex
73 protection for a "scope block". The class is instantiated as
74 a local variable for the scope block it is meant to protected. When
75 this object is created - it calls lock() on its mutex. When this
76 object is destroyed - it calls unlock() on its mutex.
77 */
78 class ScopeLock
79 {
80 private:
81 /// Reference to the mutex to be used for synchronization
82 Mutex& m_mutex;
83
84 public:
85 /// Constructor. This will block until the mutex lock is acquired.
86 ScopeLock( Mutex& mutex ) noexcept
87 : m_mutex( mutex ) { m_mutex.lock(); }
88
89 /// Destructor. This method releases ownership of the mutex
90 ~ScopeLock() { m_mutex.unlock(); }
91
92
93 private:
94 /** This method PREVENTS this object from being created on the Heap.
95 This class can only/should only be instantiated as a local variable to
96 a critical section scope block!
97 */
98 void* operator new( size_t );
99 };
100};
