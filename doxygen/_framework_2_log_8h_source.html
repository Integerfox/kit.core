11/** @file
12
13 This file declares the public facing Logging interface for generating log
14 entries. However, Application or Package code SHOULD NOT directly include
15 this file. Instead it should include their PACKAGE'S SPECIFIC public
16 interface for logging. For example, the Kit Package source code should
17 include: Kit/Logging/Pkg/Log.h
18
19 Logging has two run-time filters - one for Classification ID and one for Package
20 ID. Only log entries that 'pass' BOTH filters are placed into the Log Entry
21 FIFO.
22
23 The interface IS thread safe, i.e Clients can call the vlogf() from any
24 thread.
25*/

72/** This macro 'converts' a logging request into a KIT_SYSTEM_TRACE_RESTRICTED_MSG
73 message. Because the intended use case for this method is when there is NO
74 logging engine the application, i.e. USE_KIT_LOGGING_PKG_LOG_API is NOT
75 defined. This prevents converting numeric IDs to text symbols. The trace section
76 is derived from all of the IDs.
77
78 \code
79 The format is: LOG_<classificationId>.<packageId>.<subSystemId>.<messageId>
80
81 For example, given:
82 classificationId=3, packageId=5, subSystemId=2, messageId=10
83 The trace section would be "LOG_3.5.2.10".
84 \endcode
85
86 The macro always evaluates to LogResult_T::ADDED.
87
88 NOTE: When the Application enables the logging engine using the
89 USE_KIT_LOGGING_PKG_LOG_API switch - it must provide the "logging configuration",
90 i.e. the assignment absolute values for classification and package IDs.
91 */
92#define KitLoggingFramework_logTracef( classificationId, packageId, subSystemId, messageId, ... ) \
93 ( [&]() -> ::Kit::Logging::Framework::LogResult_T { \
94 char sectionBuf[32]; \
95 Kit::Logging::Framework::tracef_buildSection_( sectionBuf, sizeof( sectionBuf ), classificationId, packageId, subSystemId, messageId ); \
96 KIT_SYSTEM_TRACE_RESTRICTED_MSG( sectionBuf, __VA_ARGS__ ); \
97 return ::Kit::Logging::Framework::LogResult_T::ADDED; \
98 }() )
