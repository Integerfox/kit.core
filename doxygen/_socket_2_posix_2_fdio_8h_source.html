37/** This static class provides a collection of functions for operating on POSIX

42public:
43 /// Used to indicate an invalid file descriptor
44 static constexpr int INVALID_FD = -1;
45
46 /** Closes the file descriptor 'fd'. If 'fd' is already closed (i.e. INVALID_FD),
47 the method is a NOP. Upon a successful close, 'fd' is set to INVALID_FD.
48 */
49 inline static void close( int& fd ) noexcept
50 {
51 if ( fd != INVALID_FD )
52 {
53 ::shutdown( fd, SHUT_RDWR );
54 ::close( fd );
55 fd = INVALID_FD;
56 }
57 }
58
59public:
60 /** Attempts to write up to 'maxBytes' from 'buffer' to the file descriptor
61 'fd'. The actual number of bytes written is returned in 'bytesWritten'.
62
63 The 'eosFlag' is set to true if End-of-Stream was encountered, else it is
64 cleared (i.e. set to false)
65
66 The method blocks until at least one byte is written to the file descriptor.
67
68 Returns true if successful; else false on error.
69 */
70 static bool write( int fd, bool& eosFlag, const void* buffer, ByteCount_T maxBytes, ByteCount_T& bytesWritten ) noexcept
71 {
72 KIT_SYSTEM_ASSERT( buffer != nullptr );
73
74 // Throw an error if the socket had already been closed
75 if ( fd == INVALID_FD )
76 {
77 return false;
78 }
79
80 // Ignore write requests of ZERO bytes
81 if ( maxBytes == 0 )
82 {
83 bytesWritten = 0;
84 return true;
85 }
86
87 // perform the write
88 ByteCount_T result = send( fd, (char*)buffer, maxBytes, MSG_NOSIGNAL );
89 if ( result < 0 )
90 {
91 bytesWritten = 0;
92 eosFlag = true;
93 return false;
94 }
95 if ( result == 0 )
96 {
97 bytesWritten = 0;
98 eosFlag = true;
99 return false;
100 }
101 bytesWritten = result;
102 eosFlag = false;
103 return true;
104 }

119public:
120 /** The method attempts to read up to 'numBytes' from the file descriptor. The
121 actual number of bytes read is returned via 'bytesRead'
122
123 The 'eosFlag' is set to true if End-of-Stream was encountered, else it is
124 cleared (i.e. set to false)
125
126 The method blocks until at least one byte is read from the file descriptor.
127
128 Returns true if successful; else false on error.
129 */
130
131 static bool read( int fd, bool& eosFlag, void* buffer, ByteCount_T numBytes, ByteCount_T& bytesRead ) noexcept
132 {
133 KIT_SYSTEM_ASSERT( buffer != nullptr );
134
135 // Throw an error if the socket had already been closed
136 if ( fd == INVALID_FD )
137 {
138 return false;
139 }
140
141 // Ignore read requests of ZERO bytes
142 if ( numBytes == 0 )
143 {
144 bytesRead = 0;
145 return true;
146 }
147
148 // perform the read
149 ByteCount_T result = recv( fd, (char*)buffer, numBytes, 0 );
150 if ( result < 0 )
151 {
152 bytesRead = 0;
153 eosFlag = true;
154 return false;
155 }
156
157 // Connection closed gracefully
158 if ( result == 0 )
159 {
160 bytesRead = 0;
161 eosFlag = true;
162 return false;
163 }
164
165 // Success!
166 bytesRead = result;
167 eosFlag = false;
168 return true;
169 }

172 /** Returns true if there is data available to be read from the file descriptor
173 */
174 inline static bool available( int fd ) noexcept
175 {
176 // Trap that the stream has been CLOSED!
177 if ( fd == INVALID_FD )
178 {
179 return false;
180 }
181
182 int nbytes;
183 ioctl( fd, FIONREAD, &nbytes );
184 return nbytes > 0;
185 }

188 /** Creates the "listening" socket. It will attempt N times to binding the
189 listening port before giving up. Upon success a valid file descriptor
190 is returned (>=0). On error a negative value is returned
191 */
192 static int createListeningSocket( int port,
193 unsigned numRetries,
194 uint32_t delayBetweenRetriesMs ) noexcept
195 {
196 struct sockaddr_in local;
197 int one = 1;
198 int result;
199
200 // Create the Socket to listen with
201 int fd = socket( AF_INET, SOCK_STREAM, 0 );
202 if ( fd < 0 )
203 {
204 return fd;
205 }
206
207 // Set Options on the socket
208 setsockopt( fd, SOL_SOCKET, SO_REUSEADDR, &one, sizeof( one ) );
209
210 // Set the "address" of the socket
211 unsigned retry = numRetries;
212 while ( retry )
213 {
214 memset( &local, 0, sizeof( local ) );
215 local.sin_family = AF_INET;
216 local.sin_addr.s_addr = htonl( INADDR_ANY );
217 local.sin_port = htons( port );
218 if ( ( result = bind( fd, (struct sockaddr*)&local, sizeof( local ) ) ) >= 0 )
219 {
220 break;
221 }
222
223 // Delay between retry attempts
224 Kit::System::sleep( delayBetweenRetriesMs );
225 if ( --retry == 0 )
226 {
227 return result;
228 }
229 }
230
231 // Create a queue to hold connection requests
232 if ( ( result = ::listen( fd, SOMAXCONN ) ) < 0 )
233 {
234 close( fd );
235 return result;
236 }
237
238 // Success!
239 return fd;
240 }

242 /** This method is to accept an incoming connection request. Returns
243 a valid file descriptor for the accepted connection, or < 0 if an error
244 occurred. When successful, the 'client_addr' structure is populated
245 with information about the remote Host.
246 */
247 inline static int acceptConnection( int listeningFd, struct sockaddr_in& client_addr ) noexcept
248 {
249 // Wait on the 'accept'
250 socklen_t client_len = sizeof( client_addr );
251 return accept( listeningFd, (struct sockaddr*)&client_addr, &client_len );
252 }

254 /** This method enables the SO_KEEPALIVE option on the specified socket.
255 Returns >=0 if successful, else on error < 0 is returned
256 */
257 inline static int enableKeepAlive( int fd ) noexcept
258 {
259 int bOptVal = 1;
260 int bOptLen = sizeof( int );
261 return setsockopt( fd, SOL_SOCKET, SO_KEEPALIVE, (char*)&bOptVal, bOptLen );
262 }

265 /** This resolved the specific network address string and port number (IPv4
266 or IPv6) to a socket address info structure. The method returns true if
267 successful, else false is returned.
268
269 When successful, the method returns one or MORE address info structure
270 candidates via the 'results' argument. In addition on success, the caller
271 is required to freeAddresses() method to free the memory associated
272 with the 'results' argument.
273 */
274 inline static bool resolveAddress( const char* remoteHostName, int portNumToConnectTo, struct addrinfo*& results ) noexcept
275 {
276 Kit::Text::FString<5> port( portNumToConnectTo );
277 struct addrinfo hints;
278 memset( &hints, 0, sizeof( hints ) );
279 hints.ai_family = AF_UNSPEC;
280 hints.ai_socktype = SOCK_STREAM;
281 hints.ai_protocol = IPPROTO_TCP;
282 return getaddrinfo( remoteHostName, port, &hints, &results ) == 0;
283 }
284
285 /** This method frees the memory associated with the 'results' argument
286 that was populated by a previous call to the resolveAddress() method.
287 */
288 inline static void freeAddresses( struct addrinfo* results ) noexcept
289 {
290 freeaddrinfo( results );
291 }
292
293 /** Create a file descriptor for the specified socket address info.
294 Returns a valid file descriptor if successful, else < 0 is returned.
295 */
296 inline static int createSocket( struct addrinfo* addr ) noexcept
297 {
298 return socket( addr->ai_family, addr->ai_socktype, addr->ai_protocol );
299 }
