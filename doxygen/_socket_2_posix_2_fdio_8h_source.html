30/** This static class provides a collection of functions for operating on POSIX
31 file descriptors for BSD sockets.
32 */
33class Fdio
34{
35public:
36 /// Used to indicate an invalid file descriptor
37 static constexpr int INVALID_FD = -1;
38
39 /** Closes the file descriptor 'fd'. If 'fd' is already closed (i.e. INVALID_FD),
40 the method is a NOP. Upon a successful close, 'fd' is set to INVALID_FD.
41 */
42 static void close( int& fd ) noexcept;
43
44public:
45 /** Attempts to write up to 'maxBytes' from 'buffer' to the file descriptor
46 'fd'. The actual number of bytes written is returned in 'bytesWritten'.
47
48 The 'eosFlag' is set to true if End-of-Stream was encountered, else it is
49 cleared (i.e. set to false)
50
51 The method blocks until at least one byte is written to the file descriptor.
52
53 Returns true if successful; else false on error.
54 */
55 static bool write( int fd, bool& eosFlag, const void* buffer, ByteCount_T maxBytes, ByteCount_T& bytesWritten ) noexcept;
56
57 /// Flushes the file descriptor 'fd'
58 static void flush( int fd ) noexcept;
59
60 /// Returns true if the file descriptor is in the open state
61 static bool isOpened( int fd ) noexcept;
62
63public:
64 /** The method attempts to read up to 'numBytes' from the file descriptor. The
65 actual number of bytes read is returned via 'bytesRead'
66
67 The 'eosFlag' is set to true if End-of-Stream was encountered, else it is
68 cleared (i.e. set to false)
69
70 The method blocks until at least one byte is read from the file descriptor.
71
72 Returns true if successful; else false on error.
73 */
74 static bool read( int fd, bool& eosFlag, void* buffer, ByteCount_T numBytes, ByteCount_T& bytesRead ) noexcept;
75
76
77 /// Returns true if there is data available to be read from the file descriptor
78 static bool available( int fd ) noexcept;
79
80public:
81 /** Creates the "listening" socket. It will attempt N times to binding the
82 listening port before giving up. Upon success a valid file descriptor
83 is returned (>=0). On error a negative value is returned
84 */
85 static int createListeningSocket( int port,
86 unsigned numRetries,
87 uint32_t delayBetweenRetriesMs ) noexcept;
88
89 /** This method is to accept an incoming connection request. Returns
90 a valid file descriptor for the accepted connection, or < 0 if an error
91 occurred. When successful, the 'client_addr' structure is populated
92 with information about the remote Host.
93 */
94 static int acceptConnection( int listeningFd, struct sockaddr_in& client_addr ) noexcept;
95
96 /** This method enables the SO_KEEPALIVE option on the specified socket.
97 Returns >=0 if successful, else on error < 0 is returned
98 */
99 static int enableKeepAlive( int fd ) noexcept;
100
101public:
102 /** This resolved the specific network address string and port number (IPv4
103 or IPv6) to a socket address info structure. The method returns true if
104 successful, else false is returned.
105
106 When successful, the method returns one or MORE address info structure
107 candidates via the 'results' argument. In addition on success, the caller
108 is required to freeAddresses() method to free the memory associated
109 with the 'results' argument.
110 */
111 static bool resolveAddress( const char* remoteHostName, int portNumToConnectTo, struct addrinfo*& results ) noexcept;
112
113 /** This method frees the memory associated with the 'results' argument
114 that was populated by a previous call to the resolveAddress() method.
115 */
116 static void freeAddresses( struct addrinfo* results ) noexcept;
117
118 /** Create a file descriptor for the specified socket address info.
119 Returns a valid file descriptor if successful, else < 0 is returned.
120 */
121 static int createSocket( struct addrinfo* addr ) noexcept;
122};
