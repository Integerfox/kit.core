37/** This class defines methods for forcibly terminating the application.
38 How gracefully (or not) the shutdown is dependent on the actual platform
39 implementation. In addition, what 'shutdown' means is also platform
40 dependent, e.g. on an embedded platform it may invoke a reboot and/or
41 restart of the application.
42
43 Since this is a forcibly shutdown there are no guaranties with respect to
44 releasing resources.
45
46 The interface does provide for callback method(s) to be called on shutdown.
47 These methods are guaranteed to be called (or at least attempted to be
48 called in the case of true code failure).
49 */
50class Shutdown
51{
52public:
53 /** Reserved exit codes for the KIT Library */
54 enum : int
55 {
56 eSUCCESS = OPTION_KIT_SYSTEM_SHUTDOWN_BEGIN_EXIT_CODES, //!< Application terminated successfully
57 eFAILURE, //!< Application terminated due to a failure. This is a generic failure
58 eFATAL_ERROR, //!< Application detected a fatal error and self terminated. This is a generic failure
59 eASSERT, //!< Application terminated due to an assertion failure
60 eOSAL, //!< Application terminated due to an OSAL failure (e.g. error detected by a module in the Kit::System namespace)
61 eDATA_MODEL, //!< Application terminated due to a Data Model failure
62 eCONTAINER, //!< Application terminated due to a Container failure (e.g. crossed linked intrusive containers)
63 eSTREAMIO, //!< Application terminated due to a stream I/O failure. This includes errors in File IO, Sockets, Serial ports, etc.
64 eMEMORY, //!< Application terminated due to a memory failure (e.g. failed memory allocation)
65 eDRIVER, //!< Application terminated due to a driver failure. This is a generic failure
66 eITC, //!< Application terminated due to an inter-thread communication failure. This is a generic failure
67 eFSM_EVENT_OVERFLOW, //!< Application terminated due to a finite state machine event overflow condition (i.e. a FSM event was-dropped/not-processed)
68 eWATCHDOG //!< Application terminated due to a watchdog timeout (e.g. watchdog configuration errors, intentional WDOG trips)
69 };
70
71public:
72 /** This call defines the callback interface that is used when the
73 application is shutdown
74 */
75 class IHandler : public Kit::Container::ListItem
76 {
77 public:
78 /** This method is called when the application is shutting down. The
79 method is passed 'exit_code' which is the exit code provided by the
80 application when called the Shutdown interface. The return value
81 from the handler will be used as the application shutdown exit code.
82 To leave the exit code unaltered - return the passed in 'exit_code'.
83
84 Notes:
85 o The final exit code is the serialized 'sum' of the all of the
86 shutdown handlers.
87 o The notify() method is called in the thread context that
88 initiated the shutdown.
89 */
90 virtual int notify( int exitCode ) noexcept = 0;
91
92 public:
93 /// Ensure the destructor is virtual
94 virtual ~IHandler() {}
95 };
96
97
98public:
99 /** This function will force a shutdown of the application with a
100 'success' exit code. What 'forced' means is dependent on the
101 actual platform. All registered callback methods will be called
102 before exiting the application.
103 */
104 static int success() noexcept;
105
106 /** This function will force a shutdown of the application with a 'failure'
107 exit code. The caller can optional specify an exit code. What 'forced'
108 means is dependent on the actual platform. All registered callback
109 methods will be called before exiting the application. The returned
110 value - assuming the method actually returns - will be the exit code
111 returned by the last shutdown handler.
112
113 Note: The recommended approach for exiting the application due to an
114 error is to use the Cpl::System::FatalError interface so that
115 the 'why' of the failure has the potential for being captured.
116 */
117 static int failure( int exitCode = eFAILURE ) noexcept;
118
119
120public:
121 /** This method is used to register a callback method that will be called
122 when success() or failure() is called and before the application exits.
123
124 A LIFO is used when registering handlers. This means that the first
125 handler registered, is the last handler run.
126
127 During shutdown, the 'notify()' method of 'instanceToRegister'
128 is called.
129 */
130 static void registerHandler( Shutdown::IHandler& instanceToRegister ) noexcept;
131
132
133public:
134 /** This COMPONENT Scoped method. The application SHOULD not directly
135 access this method (unless you are REALLY REALLY REALLY sure you
136 know what you are doing).
137
138 Helper method used to notify registered shutdown handlers
139 */
140 static int notifyShutdownHandlers( int exitCode ) noexcept;
141};
