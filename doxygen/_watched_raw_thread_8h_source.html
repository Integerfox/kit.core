71/** RawThread - A concrete implementation for raw threads that provides
72 wrapper methods for manual watchdog management via the Supervisor.
73 */
74class RawThread : public WatchedThread
75{
76public:
77 /** Constructor
78 @param wdogTimeoutMs The watchdog timeout period for this thread in milliseconds
79 */
80 RawThread( uint32_t wdogTimeoutMs ) noexcept
81 : WatchedThread( wdogTimeoutMs )
82 {
83 }
84
85 /** Starts watchdog monitoring for this thread by registering it with the Supervisor.
86 */
87 void startWatching() noexcept
88 {
89 Supervisor::beginWatching( *this );
90 }
91
92 /** Stops watchdog monitoring for this thread by unregistering it from the Supervisor.
93 */
94 void stopWatching() noexcept
95 {
96 Supervisor::endWatching( *this );
97 }
98
99 /** Kicks/reloads the watchdog timer for this thread via the Supervisor.
100 This should be called periodically to indicate the thread is healthy.
101 */
102 void reload() noexcept
103 {
104 Supervisor::reloadThread( *this );
105 }
106};
