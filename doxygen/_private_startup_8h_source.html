22/** This class defines a start-up handler/hook that allows a KIT Library
23 sub-system to register to be called when the Api::initialize()
24 method is called.
25
26 The intended usage of this interface is that a sub-system creates a
27 sub-class of this class and then creates a static instance of its sub-class.
28 During the init process, the instance's 'notifyInit' method will be called.
29
30 NOTE: This class/interface SHOULD ONLY be used by KIT Library interfaces,
31 the application should NEVER register a startup hook!
32 */
33class IStartupHook : public Kit::Container::ListItem
34{
35public:
36 /** This enum defines 'init_levels'. The init levels are processed
37 in order (lowest value to highest), i.e. callbacks for lowest
38 numbered init level are called first and the callback for highest
39 numbered init level are called last. Within a init level there is
40 no guaranteed order to the callbacks.
41 */
42 enum InitLevel
43 {
44 TEST_INFRA=0, /// Initialized first
45 SYSTEM,
46 MIDDLE_WARE,
47 APPLICATION
48 }; /// Initialized last
49
50
51public:
52 /** This method is called as part of the KIT Library initialize()
53 process. The 'initLevel' informs the client what initialize
54 level context the notifyInit() method is being called.
55 */
56 virtual void notify( InitLevel initLevel ) = 0;
57
58
59protected:
60 /** Base Class constructor -->performs the callback/init-level
61 registration
62 */
63 IStartupHook( InitLevel myInitLevel ) noexcept;
64
65public:
66 /// Ensure the destructor is virtual
67 virtual ~IStartupHook() {}
68
69
70public:
71 /** This is used to register instance of this class. This method is
72 NOT implemented by the 'client sub-system' - it is a singleton that
73 is implemented Kit::System::Startup classes.
74
75 NOTE: This is the ONE KIT Library method that can be called BEFORE
76 Kit::System::Api::initialize() is called.
77 */
78 static void registerHook( IStartupHook& callbackInstance, InitLevel initOrder ) noexcept;
79
80 /** This method is intended to be USED ONLY by the Kit::System::Api::init()
