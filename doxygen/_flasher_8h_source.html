26/** This concrete class flashes an LED based on a series of off/on cycle times
27 contained in linked list.
28
29 This class is not thread safe.
30 */
31class Flasher
32{
33public:
34 /// Constructor
35 Flasher()
36 {
37 Bsp_turn_off_debug1();
38 }
39
40 /** Method to run a flash pattern. The pattern is defined by a list of
41 Period objects that specify the on/off state and duration of each period
42 in the pattern.
43 */
44 void runPattern( Kit::Container::DList<Period>& pattern ) noexcept
45 {
46 // Print out the flash pattern. The list content is NOT changed
47 KIT_SYSTEM_TRACE_MSG( "main", "Flasher::runPattern()" );
48 uint32_t totalDurationMs = 0;
49 Period* item = pattern.first(); // Get the first item in the pattern list without modifying the list
50 while ( item )
51 {
52 KIT_SYSTEM_TRACE_MSG( "main", " %s period: %" PRIu32 " ms", item->getLEDOnOffState() ? "ON " : "OFF", item->getDurationMs() );
53 totalDurationMs += item->getDurationMs();
54 // Skip to the next item in the pattern list without modifying the list
55 item = pattern.next( *item );
56 }
57 KIT_SYSTEM_TRACE_MSG( "main", "Total duration: %" PRIu32 " ms", totalDurationMs );
58
59 // Execute the flash pattern.
60 // NOTE: The list is empty as the items are removed from the list as they are executed.
61 item = pattern.get(); // Get the first item in the pattern list (the item IS removed from list)
62 while ( item )
63 {
64 if ( item->getLEDOnOffState() )
65 {
66 Bsp_turn_on_debug1();
67 }
68 else
69 {
70 Bsp_turn_off_debug1();
71 }
72 Kit::System::sleep( item->getDurationMs() );
73
74 // Get the next item in the pattern list (the item IS removed from list)
75 item = pattern.get();
76 }
77
78 // Once the pattern is done, make sure the LED is off
79 Bsp_turn_off_debug1();
80 }
81};
