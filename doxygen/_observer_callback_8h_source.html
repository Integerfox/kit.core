20/** This template class implements a Observer that uses a callback function
21 (or similar callable) for change notifications. This approach avoids the
22 need for inheritance when an associated class needs to subscribe to multiple
23 Model Points.
24
25 Usage:
26 class MyClass
27 {
28 ObserverCallback<Kit::Dm::Mp::Uint32> m_sub1;
29
30 void onM1(Kit::Dm::Mp::Uint32& mp, Kit::Dm::IObserver& sub) { ... }
31
32 MyClass(Kit::EventQueue::IChangeNotification& eventQueServer)
33 : m_sub1(eventQueServer)
34 {
35 m_sub1.setCallback<MyClass, &MyClass::onM1>(this);
36 }
37 };
38
39 Template Arguments:
40 MP - The concrete Model Point Type
41 */
42template <class MP>
43class ObserverCallback : public Observer<MP>
44{
45public:
46 /** Define the Callback function signature.
47 */
48 typedef void ( *CallbackFunc_T )( void* context, MP& modelPointThatChanged, IObserver& clientObserver );
49
50protected:
51 /// Context pointer (passed to the callback function)
52 void* m_context;
53
54 /// The callback function
55 CallbackFunc_T m_callback;
56
57public:
58 /// Constructor
59 ObserverCallback( Kit::EventQueue::IChangeNotification& myEventLoop )
60 : Observer<MP>( myEventLoop )
61 , m_context( 0 )
62 , m_callback( 0 )
63 {
64 }
65
66 /// Default Constructor
67 ObserverCallback()
68 : Observer<MP>()
69 , m_context( 0 )
70 , m_callback( 0 )
71 {
72 }
73
74public:
75 /// Type safe change notification. See Kit::Dm::IObserver
76 void modelPointChanged( MP& modelPointThatChanged, IObserver& clientObserver ) noexcept override
77 {
78 if ( m_callback )
79 {
80 m_callback( m_context, modelPointThatChanged, clientObserver );
81 }
82 }
83
84public:
85 /** Sets the callback to a Member Function of the 'CONTEXT' class type.
86 This provides a type-safe wrapper that converts the Context pointer
87 to the appropriate class pointer.
88 */
89 template <class CONTEXT, void ( CONTEXT::*FUNC )( MP&, IObserver& )>
90 void setCallback( CONTEXT* context ) noexcept
91 {
92 m_context = context;
93 m_callback = &entryPoint<CONTEXT, FUNC>;
94 }
95
96 /** Sets the callback to a generic C-style function pointer.
97 */
98 void setCallback( CallbackFunc_T callbackFunction, void* context = 0 ) noexcept
99 {
100 m_context = context;
101 m_callback = callbackFunction;
102 }
103
104protected:
105 /// Internal helper method to adapt Member Function to C-style callback
106 template <class CONTEXT, void ( CONTEXT::*FUNC )( MP&, IObserver& )>
107 static void entryPoint( void* context, MP& modelPointThatChanged, IObserver& clientObserver )
108 {
109 CONTEXT* obj = static_cast<CONTEXT*>( context );
110 ( obj->*FUNC )( modelPointThatChanged, clientObserver );
111 }
112};
