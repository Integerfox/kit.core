21/** This concrete class implements manages a list of Software Timers. When an
22 individual timer's Count object reaches zero, its callback method is invoked
23 and it is remove from the Active List. The Active List is designed to be
24 very efficient with respect to decrement counts when there are many Timers
25 in the list, i.e. the amount of time to decrement the individual Timers is
26 NOT a function of the number of active Timers.
27
28 The Timer Manager requires that the Timer Manager instances, all Timer
29 instances, and the Timer's Context (i.e. the code that executes the
30 timer expired callbacks) all execute in the SAME thread.
31 */
32class TimerManager
33{
34public:
35 /// Constructor
36 TimerManager();
37
38public:
39 /** This method starts the Timer Manager. This method should be called
40 only once just before the Main/Event loop processing begins.
41
42 This method must be called from the same thread that the Timer Manager
43 executes in.
44 */
45 virtual void startManager() noexcept;
46
47 /** This method processes the current active timer lists. For each timer
48 that has expired, the timer's context callback method is called.
49 */
50 virtual void processTimers() noexcept;
51
52 /// Returns true if there is at least one active timer
53 virtual bool areActiveTimers() const noexcept;
54
55
56public:
57 /// Register for notification
58 virtual void attach( ICounter& clientToCallback ) noexcept;
59
60 /** Unregister for notification. Returns true if the client was currently
61 attached/registered; else false is returned.
62 */
63 virtual bool detach( ICounter& clientToCallback ) noexcept;
64
65 /// This method converts the requested duration in milliseconds to counter ticks.
66 virtual uint32_t msecToCounts( uint32_t durationInMsecs ) const noexcept;
67
68
69protected:
70 /// Helper method.
71 virtual void addToActiveList( ICounter& clientToCallback ) noexcept;
72
73 /** This method is intended to be call by a the timing source and each
74 call to this method represents that one tick has expired, i.e. decrement
75 the active Counter objects' by one.
76 */
77 virtual void tick( uint32_t milliseconds = 1 ) noexcept;
78
79 /** This method is used by the Tick source to information the Timer Manager
80 that there are no more ticks for the timing source's current tick cycle
81 */
82 virtual void tickComplete() noexcept;
83
84protected:
85 /// List of active counters. Points to a physical list.
86 Kit::Container::DList<ICounter> m_counters;
87
88 /// List of Pending-to-attach counters (this happens when timers attach from the timer-expired-callbacks). Points to a physical list.
89 Kit::Container::DList<ICounter> m_pendingAttach;
90
91 /// Elapsed time of the previous processing cycle
92 uint32_t m_timeMark;
93
94 /// Elapsed time of the current processing cycle
95 uint32_t m_timeNow;
96
97 /// Flag to tracks when I am actively processing/consuming ticks
98 bool m_inTickCall;
99};
