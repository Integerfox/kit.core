24/** This abstract class defines interface for managing 'entries' from a
25 a collection of entries (i.e. read an entry from a Journal EntryRecord).
26
27 See the README.md file for more details about "Journal Records".
28
29 NOTE: This interface/class is NOT THREAD SAFE and should only be 'used' from
30 the Record Server's thread.
31*/
32class IEntry : public Kit::Persistence::Record::IDataRecord
33{
34public:
35 /** This structure define an 'marker' that identifies an entry's location
36 in persistent media.
37 */
38 struct Marker_T
39 {
40 uint64_t timestamp; //!< The timestamp value for the entry (Note: this is free-running counter, i.e. is NOT elapsed based)
41 Size_T mediaOffset; //!< Offset, within the IMedia storage to the start of the Entry
42
43 /// Constructor
44 Marker_T()
45 : timestamp( 0 )
46 , mediaOffset( 0 )
47 {
48 }
49 };
50
51public:
52 /** This method reads/retrieves the latest entry (from the list of Journal
53 Entries) stored in the persistent media. The method is synchronous in
54 that the method does not return until the entry has been 'read' from
55 the persistent media.
56
57 Returns true if the latest entry is valid and has been read into 'dst';
58 else false is returned.
59
60 NOTE: When getLatest() returns false, this means there are NO entries
61 stored.
62
63 NOTE: 'dst' is ALWAYS updated EVEN if no entry was 'found', basically 'dst'
64 is used as a work buffer when traversing the list.
65 */
66 virtual bool getLatest( IPayload& dst, Marker_T& entryMarker ) noexcept = 0;
67
68public:
69 /** This method walks the entire 'list of entries' and returns the next
70 newer entry as specified by the 'newerThan' value. The traversal
71 starts with the 'beginHereMarker' The method is synchronous in that the
72 method does not return until the entry has been 'read' from the persistent
73 media.
74
75 Returns true if the 'next' entry was found and has been read into 'dst';
76 else false is returned.
77
78 NOTE: 'dst' is ALWAYS updated EVEN if no entry was 'found', basically 'dst'
79 is used as a work buffer when traversing the list.
80
81 */
82 virtual bool getNext( uint64_t newerThanTimestamp,
83 const Marker_T beginHereMarker,
84 IPayload& dst,
85 Marker_T& entryMarker ) noexcept = 0;
86
87 /** This method is similar to getNext(), except that it returns the next
88 oldest entry.
89 */
90 virtual bool getPrevious( uint64_t olderThanTimestamp,
91 const Marker_T beginHereMarker,
92 IPayload& dst,
93 Marker_T& entryMarker ) noexcept = 0;
94
95public:
96 /** This method can be used to read an entry by its 'entry index'. The
97 entry index is a zero based index.
98
99 Returns true if the entry at 'entryIndex' is a valid entry; else false
100 is returned. Note: if 'entryIndex' is out of range, false is returned.
101
102 NOTE: 'dst' is ALWAYS updated EVEN if no valid entry was 'found',
103 basically 'dst' is used as a work buffer when traversing the list.
104 */
105 virtual bool getByEntryIndex( Size_T entryIndex,
106 IPayload& dst,
107 Marker_T& entryMarker ) noexcept = 0;
108
109
110 /** This method returns the maximum allowed 'entryIndex' when calling
111 getByEntryIndex().
112 */
113 virtual Size_T getMaxIndex() const noexcept = 0;
114
115public:
116 /** This method appends an entry to the list of Journal Entries. The method
117 is synchronous in that the method does not return until the entry has
118 been 'written' the persistent media.
119
120 Returns true on success; else if an error occurred (e.g. IO error while
121 writing) false is returned.
122 */
123 virtual bool addEntry( const IPayload& src ) noexcept = 0;
124
125public:
126 /** This method resets the head pointer and the timestamp to zero. It is
127 essentially a logical erase of the entries. The method returns true if
128 the reset was successful; else false is returned when an error occurred.
129
130 NOTE: With respect to 'make room' for more entries, there is no actual
131 need to either logical or physical the entries, since the the
132 design always guarantees that the latest N entries are persistently
133 stored.
134 */
135 virtual bool resetHead() noexcept = 0;
