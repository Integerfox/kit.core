21/** This 'implementation' class provide the base functionality for a ring buffer.
22
23 The Ring buffer is ISR/Thread safe as long as following constraints are met:
24 - Writing to an 'unsigned' data type is an atomic machine instruction.
25 - There is SINGLE producer (add) and a SINGLE consumer (remove)
26
27 NOTE: This class is not intended to be used directly. A child class is
28 required to provide the ring buffer memory and type safety.
29 */
30class RingBufferBase
31{
32protected:
33 /// Constructor
34 RingBufferBase( unsigned numMaxElements ) noexcept;
35
36public:
37 /// This method returns true if the Ring Buffer is empty
38 bool isEmpty() const noexcept
39 {
40 return m_readIdx == m_writeIdx;
41 }
42
43 /// This method returns true if the Ring Buffer is full
44 bool isFull() const noexcept
45 {
46 return ( ( m_writeIdx + 1 ) % m_elements ) == m_readIdx;
47 }
48
49 /// This method returns the maximum number of items that can be stored in the Ring buffer.
50 unsigned getMaxItems() const noexcept
51 {
52 return m_elements - 1;
53 }
54
55 /// This method returns the CURRENT number of elements in the Ring Buffer
56 unsigned getNumItems() const noexcept
57 {
58 return ( m_writeIdx - m_readIdx + m_elements ) % m_elements;
59 }
60
61public:
62 /** This method resets the underlying read/write indexes, i.e. effectively
63 clearing the Ring Buffer contents. This method can ONLY BE USED when
64 the Ring Buffer is NOT in use. It is the application's responsibility
65 for ensuring this condition is met.
66 */
67 void clearTheBuffer() noexcept
68 {
69 m_readIdx = m_writeIdx = 0;
70 }
71
72protected:
73 /** Removes the first item in the Ring buffer. The contents of the
74 removed item will be copied into the 'elemPtr' argument. The method
75 returns true if the operation was successful; else false is
76 returned, i.e. the Ring buffer is/was empty.
77 */
78 bool remove( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept;
79
80 /** Add a new item to the Ring Buffer as the last item in the buffer. The
81 contents of the source item (aka 'elemPtr') will be copied into the
82 Ring Buffer. Returns true if the operation was successful; else false
83 is returned, i.e. the Buffer was full prior to the attempted add().
84 */
85 bool add( const void* elemPtr, size_t elemSize, void* dstRingBufMemory ) noexcept;
86
87protected:
88 /** This method inspects the head element of the Ring buffer without removing it.
89 The contents of the head element will be copied into the 'elemPtr' argument.
90 The method returns true if the operation was successful; else false is
91 returned, i.e. the Ring buffer is empty.
92 */
93 bool peekHead( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept;
94
95 /** This method inspects the tail element of the Ring buffer without removing it.
96 The contents of the tail element will be copied into the 'elemPtr' argument.
97 The method returns true if the operation was successful; else false is
98 returned, i.e. the Ring buffer is empty.
99 */
100 bool peekTail( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept;
101
102protected:
103 /** This method returns a pointer to the next item to be removed. In addition
104 it returns the number of elements that can be removed as linear/flat
105 buffer (i.e. without wrapping around raw buffer memory).
106
107 The returned pointer and element count are only valid till the next
108 read/remove operation.
109
110 If the Ring buffer is empty, a null pointer is returned
111 */
112 void* peekNextRemoveItems( unsigned& dstNumFlatElements, size_t elemSize, const void* srcRingBufMemory ) const noexcept;
113
114 /** This method 'removes' N elements - that were removed using the
115 pointer returned from peekNextRemoveItems - from the ring buffer.
116 Basically it updates the head pointer to reflect items removed using
117 direct memory access.
118
119 'numElementsToRemove' be less than or equal to the 'dstNumFlatElements'
120 returned from peekNextRemoveItems().
121
122 CAUTION: IF YOU DON'T UNDERSTAND THE USE CASE FOR THIS METHOD - THEN
123 DON'T USE IT. If this method is used improperly, it WILL
124 CORRUPT the Ring Buffer!
125 */
126 void removeElements( unsigned numElementsToRemove ) noexcept
127 {
128 // By definition/design - I simply have to update the read index
129 m_readIdx = ( m_readIdx + numElementsToRemove ) % m_elements;
130 }
131
132protected:
133 /** This method returns a pointer to the next item to be added. In addition
134 it returns the number of elements that can be added as linear/flat
135 buffer (i.e. without wrapping around raw buffer memory)
136
137 If the Ring buffer is full, a null pointer is returned
138 */
139 void* peekNextAddItems( unsigned& dstNumFlatElements, size_t elemSize, const void* srcRingBufMemory ) const noexcept;
140
141 /** This method 'adds' N elements - that were populated using the
142 pointer returned from peekNextAddItems - to the ring buffer. Basically
143 its updates the tail pointer to reflect items added using direct
144 memory access.
145
146 'numElementsAdded' be less than or equal to the 'dstNumFlatElements'
147 returned from peekNextAddItems().
148
149 CAUTION: IF YOU DON'T UNDERSTAND THE USE CASE FOR THIS METHOD - THEN
150 DON'T USE IT. If this method is used improperly, it WILL
151 CORRUPT the Ring Buffer!
152 */
153 void addElements( unsigned numElementsAdded ) noexcept
154 {
155 // By definition/design - I simply have to update the write index
156 m_writeIdx = ( m_writeIdx + numElementsAdded ) % m_elements;
157 }
158
159protected:
160 /** Index of the next item to be read from the ring. This variable is only
161 updated when a read operation is performed. In addition, the variable
162 is atomically updated.
163 */
164 volatile unsigned m_readIdx;
165
166 /** Index of the next item to be added to the ring. This variable
167 is only updated when a add operation is performed. In addition,
168 the variable is atomically updated.
169 */
170 volatile unsigned m_writeIdx;
171
172 /// Maximum number of element in the 'ring memory
173 const unsigned m_elements;
174
175private:
176 /// Prevent access to the copy constructor -->Ring Buffers can not be copied!
177 RingBufferBase( const RingBufferBase& m ) = delete;
178
179 /// Prevent access to the assignment operator -->Ring Buffers can not be copied!
180 RingBufferBase& operator=( const RingBufferBase& m ) = delete;
181
182 /// Prevent access to the move constructor -->Ring Buffers can not be implicitly moved!
183 RingBufferBase( RingBufferBase&& m ) = delete;
184
185 /// Prevent access to the move assignment operator -->Ring Buffers can not be implicitly moved!
186 RingBufferBase& operator=( RingBufferBase&& m ) = delete;
187};
