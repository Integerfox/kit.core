21/** This concrete template class extends the DecoderReader where a frame
22 consider one line of PRINTABLE ASCII text followed by a newline character.
23
24 ANY printable ASCII character is accepted as start-of-frame. And end-of-frame
25 is a newline character (`\r` or `\n`). There is NO escape sequence since the
26 escape sequence is need to embedded EOF characters - but in this case EOF
27 (newline) is NOT a printable ASCII character -->so an escape sequence is
28 not meaningful.
29
30 This class is NOT thread safe.
31
32 NOTE: If a non-printable ASCII character is encounter within a frame, the
33 current frame is aborted AND a newline character is required before
34 accepting/detecting a new SOF character(s).
35
36 */
37class LineDecoder : public DecoderReader
38{
39public:
40 /** Constructor.
41 NOTES:
42 - The size of the workBuffer determines how big of 'chunks' data is
43 read from the "input source", i.e. it is a working buffer and
44 does NOT have to be the size of the maximum possible input frame.
45
46 - If the 'convertTabs' argument is set to a value OTHER than a tab
47 character, then any tab characters encounter will be converted to
48 the value of 'convertTabs'. If 'convertTabs' is set to a tab
49 character, then tabs are not converted - which means that if tabs
50 are encountered, they will be treated as a non-printable character
51 and cause the current frame to be aborted.
52 */
53 LineDecoder( ISource& source,
54 uint8_t* workBuffer,
55 Kit::Type::SSize_T sizeOfWorkBuffer,
56 char convertTabs = '\t' )
57 : DecoderReader( source, workBuffer, sizeOfWorkBuffer )
58 , m_frameDetected( false )
59 , m_convertTabs( convertTabs )
60 {
61 KIT_SYSTEM_ASSERT( convertTabs == '\t' || ( convertTabs >= ' ' && convertTabs <= '~' ) );
62 }
63
64protected:
65 /// See Kit::Framing::IDecoder
66 bool isStartOfFrame( uint8_t byte ) noexcept override;
67
68 /// See Kit::Framing::IDecoder
69 bool isEndOfFrame( uint8_t byte ) noexcept override;
70
71 /// See Kit::Framing::IDecoder
72 bool isEscapeByte( uint8_t byte ) noexcept override;
73
74 /// See Kit::Framing::IDecoder
75 bool isLegalByte( uint8_t byte ) noexcept override;
76
77 /// See Kit::Framing::DecoderReader
78 void initializeScan() noexcept override;
79
80protected:
81 /// Helper method to convert tabs when requested
82 void convertTabs( uint8_t& incomingByte ) noexcept;
83
84protected:
85 /// Track if I have encountered SOF character(s) without an EOF character
86 bool m_frameDetected;
87
88 /// Remember my tabs option
89 char m_convertTabs;
90};
