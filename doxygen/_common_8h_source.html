36/** This partial concrete class provides common implementations, utility methods,
37 etc. for concrete child classes that implement the IChunk interface. The
38 class ASSUMES the following:
39
40 - All child classes will use a checksum to detect corrupted data.
41
42 - There is sufficiently large temporary buffer to hold a Record's data plus
43 the additional meta data persistent stored. Note: The private_.h header
44 file provides a default buffer. See private_.h for additional details
45 about the default buffer.
46
47 - Some of helper methods (e.g. pushToRecord() and pullFromRecord()) are
48 virtual wrapper methods. By having a virtual wrapper - pushing/pulling
49 to/from the data record facilitates future child classes to extend existing
50 chunk classes with the ability to decrypt/encrypt the record data
51 */
52class Common : public IChunk
53{
54protected:

59 Size_T workBufferSize ) noexcept

61 , m_crc( edc )
62 , m_workBuffer( workBuffer )
63 , m_workBufferSize( workBufferSize )
64 {
65 }

67protected:
68 /** Helper method to read a Size_T value from the persistence storage. It
69 has the following behavior/side-effects:
70
71 - The 'value' is accumulated into the CRC calculation (i.e. m_crc.accumulate()
72 is called)
73
74 - The 'offset' is updated to point to the next byte after the Size_T
75 value that was read
76
77 - The default behavior is to use m_workBuffer as the buffer to read the
78 Size_T value into RAM. However, a different buffer can be provided via
79 the workBuffer parameter.
80
81 Returns true if the read was successful, else false is returned.
82 */
83 bool readSizeT( Size_T& value, Size_T& offset, IMedia& media ) noexcept;
84
85 /// Same as readSizeT() but reads a uint64_t value
86 bool readUint64( uint64_t& value, Size_T& offset, IMedia& media ) noexcept;
87
88 /** Helper method to read the application record data into m_workBuffer. It
89 has the following behavior/side-effects:
90
91 - The method advances the 'offset' to point to the next byte after the
92 'datalen'
93
94 - The method also accumulates the data into the CRC calculation.
95
96 Returns true if the read was successful, else false is returned.
97 */
98 bool readRecordData( Size_T dataLen, Size_T& offset, IMedia& media ) noexcept;
99
100protected:
101 /** Helper method to write a Size_T value to persistent storage. It has the
102 following behavior/side-effects:
103
104 - Converts the 'value' from the Host/MCU Endianess to the media's Endianess
105 before writing to persistent storage
106
107 - The 'value' is accumulated into the CRC calculation (i.e. m_crc.accumulate()
108 is called). The accumulation is done AFTER any endianess conversion
109 is done.
110
111 - The 'offset' is updated to point to the next byte after the Size_T
112 value that was written
113
114 Returns true if the write was successful, else false is returned.
115 */
116 bool writeSizeT( Size_T value, Size_T& offset, IMedia& media ) noexcept;
117
118 /// Same as writeSizeT() but writes a uint64_t value
119 bool writeUint64( uint64_t value, Size_T& offset, IMedia& media ) noexcept;
120
121 /** Helper method to write the application record data stored in m_workBuffer.
122 It has the following behavior/side-effects:
123
124 - The method advances the 'offset' to point to the next byte after the
125 'datalen'
126
127 - The method also accumulates the data into the CRC calculation.
128
129 Returns true if the write was successful, else false is returned.
130 */
131 bool writeRecordData( Size_T dataLen, Size_T& offset, IMedia& media ) noexcept;
132
133protected:
