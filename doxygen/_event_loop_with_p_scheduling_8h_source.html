22/** This class extends the EventLoop class to add periodic scheduling to an
23 event based 'thread'. The timing resolution of the periodic scheduling is a
24 determined by the 'timingTickInMsec' argument value in the class's constructor.
25 For example if 'timingTickInMsec' is 10ms then no interval should have an
26 execution frequency faster than 100Hz (i.e. 10ms).
27
28 The order of processing is:
29
30 1. The Event Loop signal, Event Flags, and Software Timers are processed.
31 2. The Periodic scheduler's executeScheduler() method is called
32 3. The optional 'Idle' function is called.
33 4. The thread blocks and wait for any of the asynchronous actions in step 1
34 to wake up the thread.
35 */
36class EventLoopWithPScheduling : public EventLoop, public PeriodicScheduler
37{
38public:
39 /** Constructor. The argument 'timingTickInMsec' specifies the timing
40 resolution that will be used for Kit::System::Timers AND for the
41 periodic scheduling.
42
43 The 'watchdog' parameter is optional and can be used to provide
44 watchdog monitoring for this event loop with periodic scheduling.
45 */
46 EventLoopWithPScheduling( Interval_T intervals[],
47 Hook_T beginThreadProcessing = nullptr,
48 Hook_T endThreadProcessing = nullptr,
49 ReportSlippageFunc_T slippageFunc = nullptr,
50 NowFunc_T nowFunc = ElapsedTime::millisecondsEx,
51 IdleFunc_T idleFunc = nullptr,
52 uint32_t timeOutPeriodInMsec = OPTION_KIT_SYSTEM_EVENT_LOOP_TIMEOUT_PERIOD,
53 Kit::Container::SList<IEventFlag>* eventFlagsList = nullptr,
54 IWatchedEventLoop* watchdog = nullptr ) noexcept;
55
56
57public:
58 /// See Kit::System::IRunnable
59 void entry() noexcept override;
60
61protected:
62 /// Cache the Idle function pointer
63 IdleFunc_T m_idleFunc;
64};
