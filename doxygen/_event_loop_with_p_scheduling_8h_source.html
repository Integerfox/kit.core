22/** This class extends the EventLoop class to add periodic scheduling to an
23 event based 'thread'. The timing resolution of the periodic scheduling is a
24 determined by the 'timingTickInMsec' argument value in the class's constructor.
25 For example if 'timingTickInMsec' is 10ms then no interval should have a
26 interval duration less than 10ms.
27
28 The order of processing is:
29
30 1. The Event Loop signal, Event Flags, and Software Timers are processed.
31 2. The Periodic scheduler's executeScheduler() method is called
32 3. The optional 'Idle' function is called.
33 4. The loop is repeated until there are no expired timers, and no event
34 flags - at which point the thread blocks and wait for any of the above
35 asynchronous actions to wake up the thread.
36 */
37class EventLoopWithPScheduling : public EventLoop, public PeriodicScheduler
38{
39public:
40 /** Constructor. The argument 'timingTickInMsec' specifies the timing
41 resolution that will be used for Kit::System::Timers AND for the
42 periodic scheduling.
43 */
44 EventLoopWithPScheduling( Interval_T intervals[],
45 Hook_T beginThreadProcessing = nullptr,
46 Hook_T endThreadProcessing = nullptr,
47 ReportSlippageFunc_T slippageFunc = nullptr,
48 NowFunc_T nowFunc = ElapsedTime::millisecondsEx,
49 IdleFunc_T idleFunc = nullptr,
50 uint32_t timeOutPeriodInMsec = OPTION_KIT_SYSTEM_EVENT_LOOP_TIMEOUT_PERIOD,
51 Kit::Container::SList<IEventFlag>* eventFlagsList = nullptr ) noexcept;
52
53
54public:
55 /// See Kit::System::IRunnable
56 void entry() noexcept override;
57
58protected:
59 /// Cache the Idle function pointer

61};
