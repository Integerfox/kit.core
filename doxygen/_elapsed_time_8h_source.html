21/** This class defines the interface for accessing the elapsed time since power
22 up and/or reset of the platform. Elapsed time has millisecond precision.
23 Elapsed time is kept in a free running counter - which will roll
24 over after its data type has been exceeded
25
26 There are two set of methods for accessing elapsed time:
27 1. The default 'API' that uses uint32_t to represent elapsed time. The
28 32bit counter rolls over after 49.7 days.
29 2. Extended 'API' that uses a 64bit data type to represent elapsed time.
30 The 64bit counter rolls over after 584 million years.
31
32 NOTE: The two APIs share the same underlying timing source, i.e. the returned
33 values are 'in-sync'
34 */
35class ElapsedTime
36{
37public:
38 /** This method returns the elapsed time, in milliseconds, since the system
39 was powered on and/or reset. The maximum value is 49.7 days before
40 the internal counter rolls over.
41 */
42 static uint32_t milliseconds() noexcept;
43
44 /** This method returns the delta time, in milliseconds, between the
45 specified 'startTime' and 'endTime'. 'endTime' is defaulted to NOW
46 (i.e. a call to milliseconds(). The calculation properly handles the
47 scenario of when the has been 'roll over' between the two times.
48 */
49 inline static uint32_t deltaMilliseconds( uint32_t startTime, uint32_t endTime = milliseconds() ) noexcept
50 {
51 return endTime - startTime;
52 }
53
54 /** This method returns true if the specified amount of time has elapsed
55 since the 'timeMarker'. The calculation properly handles the scenario
56 of when the has been 'roll over' between the 'timeMarker' and NOW.
57 */
58 inline static bool expiredMilliseconds( uint32_t timeMarker,
59 uint32_t duration,
60 uint32_t currentTime = milliseconds() ) noexcept
61 {
62 return deltaMilliseconds( timeMarker, currentTime ) >= duration;
63 }
64
65public:
66 /** This method returns the elapsed time, in seconds, since the system
67 was powered on and/or reset. The maximum value is 49.7 days before
68 the internal counter rolls over.
69 */
70 static inline uint32_t seconds() noexcept
71 {
72 return milliseconds() / 1000L;
73 }
