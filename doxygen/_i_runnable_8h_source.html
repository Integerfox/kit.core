23/** This mostly an abstract class defines the interface for an object that is
24 "executed" when a Thread object is created.
25 */
26class IRunnable
27{
28public:
29 /** This method is called when the Thread is started. If this function
30 returns, the Thread will be terminated. A child class is responsible for
31 implementing the entry() method.
32 */
33 virtual void entry() noexcept = 0;
34
35 /** This method is a request to have the runnable object terminate itself.
36 This is only a request, it is not a requirement of a runnable object to
37 oblige the request. Some runnable objects are just not 'built' to
38 handle a such a request. On the other hand, a runnable object such as a
39 EventServer is, and should make every attempt to comply with the request.
40
41 NOTES:
42
43 1) This a low-level terminate request in that it does not understand
44 the "semantics" of what the thread is currently doing. It simply
45 attempts to exit the entry() method. The system must first shutdown
46 the "application usage" of the thread BEFORE calling this method.
47 2) This method is an attempted work-around for terminating threads,
48 by having the thread run itself to completion. Not all OSes
49 support a polite way (i.e. reclaiming resource, memory, etc.) of
50 killing threads.
51 */
52 virtual void pleaseStop() noexcept {}
53
54 /** This method returns a pointer to the Thread object that is executing
55 the IRunnable instance. The pointer will be nullptr if the thread has
56 not started execution (i.e. before entry() is called) AND after the
57 thread has terminated (i.e. after entry() returns)
58 */
59 virtual Thread* getThread() noexcept
60 {
61 return m_parentThreadPtr_;
62 }
63
64public:
65 /// Virtual destructor
66 virtual ~IRunnable() noexcept = default;
67
68protected:
69 /// Constructor
70 IRunnable() noexcept
71 : m_parentThreadPtr_( nullptr )
72 {
73 }
74
75protected:
76 /// Allow the Thread access to the IRunnable object
77 friend class Thread;
78
79 /** This method allows the Thread class to provide the IRunnable object a
80 reference to the thread it is executing in.
81
82 NOTE: This method is intended for internal use by the Thread class ONLY.
83 */
84 virtual void setThread( Thread* thread ) noexcept
85 {
86 m_parentThreadPtr_ = thread;
87 }
88
89protected:
90 /// A back-reference to the thread that the runnable object is executing in
91 Thread* m_parentThreadPtr_;
92};
