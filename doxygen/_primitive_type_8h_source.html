147public:
148 /** Atomic Read and then clear bits operation.
149
150 NOTES:
151 1. The return data value is BEFORE the clear operation.
152 2. If the MP is invalid then the model point is set to the valid
153 state AND 'dstData' is set to zero.
154 3. The returned sequence number is the AFTER the clear operation.
155 */
156 inline uint16_t readThenClearBits( ELEMTYPE& dstData,
157 ELEMTYPE maskToClear,
158 bool forceChgNotification = false,
159 Kit::Dm::IModelPoint::LockRequest_T lockRequest = Kit::Dm::IModelPoint::eNO_REQUEST ) noexcept
160 {
161 Kit::System::Mutex::ScopeLock criticalSection( Kit::Dm::ModelPointBase::m_modelDatabase.getMutex_() );
162 dstData = this->m_data;
163 ELEMTYPE newData = dstData & ~maskToClear;
164 return Kit::Dm::ModelPointBase::writeData( &newData, sizeof( ELEMTYPE ), forceChgNotification, lockRequest );
165 }
166
167 /// Atomic Read and then clear (i.e. set to 0) operation.
168 inline uint16_t readThenClear( ELEMTYPE& dstData,
169 bool forceChgNotification = false,
170 Kit::Dm::IModelPoint::LockRequest_T lockRequest = Kit::Dm::IModelPoint::eNO_REQUEST ) noexcept
171 {
172 return readThenClearBits( dstData, (ELEMTYPE)-1, forceChgNotification, lockRequest );
173 }
174
175 /// Atomic bitwise OR operation
176 inline uint16_t bitwiseOR( ELEMTYPE maskToOR,
177 bool forceChgNotification = false,
178 Kit::Dm::IModelPoint::LockRequest_T lockRequest = Kit::Dm::IModelPoint::eNO_REQUEST ) noexcept
179 {
180 Kit::System::Mutex::ScopeLock criticalSection( Kit::Dm::ModelPointBase::m_modelDatabase.getMutex_() );
181 return this->write( this->m_data | maskToOR, forceChgNotification, lockRequest );
182 }
183
184 /// Atomic bitwise XOR operation
185 inline uint16_t bitwiseXOR( ELEMTYPE maskToXOR,
186 bool forceChgNotification = false,
187 Kit::Dm::IModelPoint::LockRequest_T lockRequest = Kit::Dm::IModelPoint::eNO_REQUEST ) noexcept
188 {
189 Kit::System::Mutex::ScopeLock criticalSection( Kit::Dm::ModelPointBase::m_modelDatabase.getMutex_() );
190 return this->write( this->m_data ^ maskToXOR, forceChgNotification, lockRequest );
191 }
192
193 /// Atomic bitwise AND operation
194 inline uint16_t bitwiseAND( ELEMTYPE maskToAND,

199 return this->write( this->m_data & maskToAND, forceChgNotification, lockRequest );

203 inline uint16_t bitwiseClearAndSet( ELEMTYPE maskToClear,

207 {

209 return this->write( ( this->m_data & ~maskToClear ) | maskToSet, forceChgNotification, lockRequest );
210 }
211
212public:
213 /// Atomic operation to set the zero indexed bit to a 1.
214 inline uint16_t setBit( uint8_t bitPosition,
215 bool forceChgNotification = false,
216 Kit::Dm::IModelPoint::LockRequest_T lockRequest = Kit::Dm::IModelPoint::eNO_REQUEST ) noexcept
217 {
218 Kit::System::Mutex::ScopeLock criticalSection( Kit::Dm::ModelPointBase::m_modelDatabase.getMutex_() );
219 return this->write( this->m_data | ( 1 << bitPosition ), forceChgNotification, lockRequest );
220 }
221
222 /// Atomic operation to set the zero indexed bit to a 0.
223 inline uint16_t clearBit( uint8_t bitPosition,
224 bool forceChgNotification = false,

226 {

228 return this->write( this->m_data & ~( 1 << bitPosition ), forceChgNotification, lockRequest );
229 }

237 return this->write( this->m_data ^ ( 1 << bitPosition ), forceChgNotification, lockRequest );
