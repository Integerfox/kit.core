25/** This class provides a concrete implementation for a Point who's data is a
26 void pointer.
27
28 On general principle there should NOT be model point type that is generic
29 (it goes against the whole 'model-points-are-type-safe' thingy). HOWEVER,
30 there are some cases where it is desirable to have MP contain a pointer -
31 where the actual definition of what that pointer points to is 'not visible'
32
33 Shorter version: DO NOT USE THIS MODEL POINT TYPE UNLESS YOU HAVE NO
34 ALTERNATIVE. Not wanting to take the time to create a new model point type
35 (with test code) is NOT an acceptable excuse for using this model point
36 type.
37
38 The toJSON() format is:
39 \code
40
41 { name:"<mpname>", type:"<mptypestring>", valid:true|false, seqnum:nnnn, locked:true|false,
42 val:"0x<hexValue" }
43
44 \endcode
45
46 The "val" format for the fromJSON() format is below. NOTE: setting a Pointer
47 value from JSON is not something that is likely to be useful in general (i.e
48 DON'T do it), but it is supported for completeness.
49 \code
50
51 val:"0x<hexValue>"
52
53 \endcode
54
55 NOTE: All methods in this class ARE thread Safe unless explicitly
56 documented otherwise.
57 */
58class Void : public PrimitiveType<uintptr_t, Void>
59{
60public:
61 /// Constructor. Invalid MP.
62 Void( Kit::Dm::IModelDatabase& myModelBase, const char* symbolicName )
63 : PrimitiveType<uintptr_t, Void>( myModelBase, symbolicName )
64 {
65 }
66
67 /// Constructor. Valid MP. Requires an initial value
68 Void( Kit::Dm::IModelDatabase& myModelBase, const char* symbolicName, void* initialValue )
69 : PrimitiveType<uintptr_t, Void>( myModelBase, symbolicName, (uintptr_t)initialValue )
70 {
71 }
72
73public:
74 /** Type safe read. See Kit::Dm::IModelPoint. Note: the underlying data is
75 a uintptr_t, but the read/write methods use void* for type safety.
76 */
77 inline bool read( void*& dstData, uint16_t* seqNumPtr = nullptr ) const noexcept
78 {
79 return Kit::Dm::ModelPointBase::readData( &dstData, sizeof( uintptr_t ), seqNumPtr );
80 }
81
82 /** Type safe write. See Kit::Dm::IModelPoint. Note: the underlying data is
83 a uintptr_t, but the read/write methods use void* for type safety.
84 */
85 inline uint16_t write( void* newValue,
86 bool forceChgNotification = false,
87 Kit::Dm::IModelPoint::LockRequest_T lockRequest = Kit::Dm::IModelPoint::eNO_REQUEST ) noexcept
88 {
89 return Kit::Dm::ModelPointBase::writeData( &newValue, sizeof( uintptr_t ), forceChgNotification, lockRequest );
90 }
91
92public:
93 /// See Kit::Dm::ModelPoint.
94 const char* getTypeAsText() const noexcept override
95 {
96 return "Kit::Dm::Mp::Void*";
97 }
98
99protected:
100 /** See Kit::Dm::Point. Note: NO loading '0x' prefix
101 \code
102 Output: val:"0x<hexadecimal value>"
103 \endcode
104 */
105 bool setJSONVal( JsonDocument& doc ) noexcept override
106 {
107 Kit::Text::FString<20> hexString;
108 hexString.format( "0x%" PRIXPTR, this->m_data );
109 doc["val"] = (char*) hexString.getString();
110 return true;
111 }
112
113public:
114 /** See Kit::Dm::Point.
115 \code
116 Input: val:"0x<hexvalue>"
117 \endcode
118 */
119 bool fromJSON_( JsonVariant& src, LockRequest_T lockRequest, uint16_t& retSequenceNumber, Kit::Text::IString* errorMsg ) noexcept override
120 {
121 // Parse as a hex string (e.g. "0xA12F")
122 if ( src.is<const char*>() )
123 {
124 // Try to parse the string as a number
125 uintptr_t val;
126 if ( Kit::Text::StringTo::unsignedInt( val, src.as<const char*>(), 16 ) )
127 {
128 retSequenceNumber = PrimitiveType<uintptr_t, Void>::write( val, false, lockRequest );
129 return true;
130 }
131 }
132 if ( errorMsg )
133 {
134 *errorMsg = "Invalid syntax for the 'val' key/value pair";
135 }
136 return false;
137 }
138};
