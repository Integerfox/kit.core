24/** This class defines the interface for a platform independent Non-volatile
25 storage driver. Only basic read/write operations are defined. How the NV
26 storage is used is left up to the application and/or upper layers
27
28 The Interface assumes byte-level erasing/writing semantics with respect to
29 persistent storage. In addition, the interface provides the client with a
30 page model, where an individual page represents the maximum amount of data
31 that can be written in a single physical update of the NV storage. That
32 said - the client is NOT required to write in page sized chunks. The page
33 information is provided to allow clients to optimize (if desired) their
34 writes to the NV storage.
35
36 The interface itself is NOT thread safe. It is the responsibility of
37 the users/clients of the driver to handle any threading issues.
38 */
39class IApi : public virtual Kit::Driver::IStart, public virtual Kit::Driver::IStop
40{
41public:
42 /** This method writes N bytes from 'srcData' starting at the storage offset
43 of 'dstOffset'. The application is responsible for ensuring that the
44 size 'srcData' is at least 'numBytesToWrite' in length.
45
46 This write() method itself is synchronous, but the physically writing of
47 the data to the NV storage is not guaranteed to synchronous, i.e. the
48 physical write operation can completed AFTER the write() method returns.
49 If a second write() call is made and the physical storage is still being
50 updated from the first write() call, the second call will wait until
51 the first physical write operation completes. There is no defined
52 semantics for the waiting (i.e. it could be a blocking wait or a busy
53 wait)
54
55 The method returns true if operation was successful; else false is
56 returned.
57 */
58 virtual bool write( size_t dstOffset,
59 const void* srcData,
60 size_t numBytesToWrite ) noexcept = 0;
61
62 /** This method reads N bytes starting at the storage offset of 'srcOffset'
63 and stores the bytes into 'dstData'.
64
65 The method returns true if operation was successful; else false is
66 returned. Note: If 'sizeDstData' is less than 'numBytesToRead' then
67 the method fails immediately.
68 */
69 virtual bool read( size_t srcOffset,
70 void* dstData,
71 size_t sizeDstData,
72 size_t numBytesToRead ) noexcept = 0;
73
74
75public:
76 /** This method returns the total number of NV storage pages. A Page is
77 defined as the boundary/maximum amount of data that can be written in
78 a single physical update of the NV storage
79
80 NOTE: ALWAYS use the getTotalSize() method for querying/determining the
81 total available storage size.
82 */
83 virtual size_t getNumPages() const noexcept = 0;
84
85 /** This method returns the NV storage page size. See getNumPages() for
86 more details about the definition of a page.
87
88 NOTE: ALWAYS use the getTotalSize() method for querying/determining the
89 total available storage size.
90 */
91 virtual size_t getPageSize() const noexcept = 0;
92
93 /** The method that returns the total storage size in bytes.
94
95 Note: This is canonical source of truth for the total available storage,
96 i.e, if the driver instance is a 'Gang' driver instance then
97 the method CAN return a different result from: getNumPages() * getPageSize()
98 */
99 virtual size_t getTotalSize() const noexcept = 0;
100
101public:
102 /// Virtual destructor
103 virtual ~IApi() noexcept;
104};
