21/** This abstract class defines an interface of the output destination for an
22 Encoder instance. The concrete instance determines the 'destination', e.g. a
23 Stream, RAM buffer, etc.
24 */
25class IDestination
26{
27public:
28 /** Used to initialize the "frame output sequence".
29 Returns true if successful, false if no space is available
30 in the destination, or an error was encountered.
31 */
32 virtual bool startOutput() noexcept = 0;
33
34 /** Outputs the next N bytes to the destination.
35 Returns true if successful, false if 'no space' is available
36 in the destination, or an error was encountered.
37 */
38 virtual bool appendOutput( const void* srcBuffer, Kit::Type::SSize_T numBytes ) noexcept = 0;
39
40 /** Convenience/Alias method to output a single byte at a time */
41 inline bool appendOutput( uint8_t srcByte ) noexcept
42 {
43 return appendOutput( &srcByte, 1 );
44 }
45
46 /** Used to finalize the "frame output sequence".
47 Returns true if successful, false if 'no space' is available
48 in the destination, or an error was encountered.
49
50 NOTE: There is NO guarantee that endOutput() will always be called
51 before starting a new frame. What is guaranteed is that startOutput()
52 will be called with each new attempted frame output sequence.
53 */
54 virtual bool endOutput() noexcept = 0;
55
56public:
57 /// Virtual Destructor
58 virtual ~IDestination() = default;
59};
