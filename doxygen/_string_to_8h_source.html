29 /** This template method converts the specified string to an signed integer.
30 The method returns true if the conversion was successful. When false is
31 returned, the 'convertedValue' argument is NOT updated.
32
33 By default the conversion assumes a base 10 number and that the 'end-of-number'
34 is end-of-string.
35
36 If endptr is specified and the conversation was successful, a pointer
37 to the first character 'after' the number is returned.
38
39 NOTE: Only base 10 conversions are supported.
40
41 The template argument 'T' must be a signed integer type.
42 */
43 template <typename T>
44 static bool signedInt( T& convertedValue,
45 const char* string,
46 const char* validStopChars = nullptr,
47 const char** endptr = nullptr ) noexcept
48 {
49 // Static assertion to enforce that T is a signed integer type
50 static_assert( std::is_signed<T>::value, "T must be a signed integer type" );
51
52 long long dstVal;
53 bool result = a2ll( dstVal, string, 10, validStopChars, endptr );
54
55 // Check if the value is within the range of type T
56 if ( result && ( dstVal >= std::numeric_limits<T>::min() && dstVal <= std::numeric_limits<T>::max() ) )
57 {
58 convertedValue = static_cast<T>( dstVal );
59 return true;
60 }
61 return false;
62 }
