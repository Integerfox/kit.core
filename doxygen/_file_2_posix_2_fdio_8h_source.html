30/** This static class provides a collection of functions for operating on POSIX
31 file descriptors.
32 NOTE: All file name/paths MUST be in native format (i.e. use '/' as the
33 directory separator)
34 */
35class Fdio : public Kit::Io::Stdio::Posix::Fdio
36{
37public:
38 /** Opens a file. The default arguments are set for:
39 o Read/Write access
40 o Do NOT create the file if the file does not exist
41 o Do NOT truncate the file contents
42 */
43 static KitIoFileHandle_T open( const char* fileEntryName, bool readOnly = true, bool forceCreate = false, bool forceEmptyFile = false ) noexcept;
44
45public:
46 /** Returns the length, in bytes, of the file. If there is an error than
47 false is returned. Note: the implementation of this method is brute force
48 in that it sets the file position indicator to EOF and uses the EOF byte
49 offset as the length. The file position indicator is restored to its
50 original position before returning.
51 */
52 static bool length( KitIoFileHandle_T fd, ByteCount_T& length ) noexcept;
53
54public:
55 /** Returns the current file pointer offset, in bytes, from the top of the
56 file. If there is an error than false is returned.
57 */
58 static bool currentPos( KitIoFileHandle_T fd, ByteCount_T& currentPos ) noexcept;
59
60 /** Adjusts the current pointer offset by the specified delta (in bytes).
61 Returns true if successful, else false (i.e. setting the pointer
62 past/before the file boundaries).
63 */
64 static bool setRelativePos( KitIoFileHandle_T fd, ByteCount_T deltaOffset ) noexcept;
65
66 /** Sets the file pointer to the absolute specified offset (in bytes).
67 Returns true if successful, else false (i.e. setting the
68 pointer past the end of the file).
69 */
70 static bool setAbsolutePos( KitIoFileHandle_T fd, ByteCount_T newoffset ) noexcept;
71
72 /** Sets the file pointer to End-Of-File. Returns true if successful, else
73 false if an error occurred.
74 */
75 static bool setToEof( KitIoFileHandle_T fd ) noexcept;
76
77
78public:
79 /** Returns information about the file system entry. If there is any
80 error, the function returns false; else true is returned.
81 */
82 static bool getInfo( const char* fsEntryName, struct stat& statOut ) noexcept;
83
84 /** Creates a new, empty file. If the file already exists, the call
85 fails. Returns true if successful.
86 */
87 static bool createFile( const char* fileName ) noexcept;
88
89 /** Creates a new directory. If the directory already exists, the call fails.
90 Returns true if successful.
91 */
92 static bool createDirectory( const char* dirName ) noexcept;
93
94public:
95 /** Renames and/or moves a file. Returns true if successful.
96 */
97 static bool move( const char* oldName, const char* newName ) noexcept;
98
99 /** Removes a file or directory. The directory must be empty to be removed.
100 Returns true if successful.
101 */
102 static bool remove( const char* fsEntryName );
103
104public:
105 /** Opens a directory - for 'walking' its contents. Each successful 'open'
106 call must be followed (at some point) by a call to closeDirectory()
107 */
108 static bool openDirectory( KitIoFileDirectory_T& hdl, const char* dirName ) noexcept;
109
110 /** Closes a directory. Can only be called after a successful call to
111 openDirectory()
112 */
113 static void closeDirectory( KitIoFileDirectory_T& hdl ) noexcept;
114
115 /** Reads 'next' entry in the directory. The entry name is copied into
116 'dstEntryName'. If there are no more entries in the directory, then
117 'dstEntryName' is set to an empty string. The 'maxEntryNameSize'
118 argument specifies the size - which needs to include space for the null
119 terminator - of the 'dstEntryName' buffer.
120
121 Returns true if successful; else false if a file system error was
122 encountered.
123 */
124 static bool readDirectory( KitIoFileDirectory_T& hdl, NameString& dstEntryName ) noexcept;
125};
