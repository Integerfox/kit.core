24/** This template class represents a client response message, which is posted
25 to the client's mailbox after the corresponding server request message is
26 returned to the client via the server message returnToSender interface. The
27 purpose of this class is to provide a standard model for confirmed
28 asynchronous messages. The SERVICE template argument is the type of the
29 server interface that is used by the corresponding server request message.
30 The CLIENT class type is the interface which is invoked by the response
31 message when the client thread invokes the process routine inherited from
32 the Message class.The CLIENT interface must contain a function named
33 "response", which returns nothing (void) and takes a single argument, which
34 is a reference to this template class type.
35 */
36template <class CLIENT, class SERVICE, class PAYLOAD>
37class ResponseMessage : public IMessage
38{
39public:
40 /// Constructor
41 ResponseMessage( CLIENT& client, EventQueue::IQueue& clientsEventQueue, SERVICE& server, PAYLOAD& payload )
42 : m_client( &client ), m_rh( clientsEventQueue, *this ), m_request( server, payload, m_rh )
43 {
44 }
45
46 /// Constructor
47 ResponseMessage( CLIENT& client, EventQueue::IQueue& clientsEventQueue, SAP<SERVICE>& serverSap, PAYLOAD& payload )
48 : m_client( &client ), m_rh( clientsEventQueue, *this ), m_request( serverSap, payload, m_rh )
49 {
50 }
51
52public:
53 /** This method is use to ABANDON the response message. In this context 'abandoned'
54 means the client's 'response(msg)' method will NEVER be called. It does
55 NOT mean that request processing is aborted.
56
57 This method should only be used when the client is 'going out of scope'
58 (i.e. called in its destructor). If there is outstanding request/response
59 transaction when the client's destructor is called, failing to call the
60 abandon() method will result in a callback to a non-existing object.
61
62 NOTE: In a perfect world, the client would always wait for all outstanding
63 transactions to complete before its destructor is called. However, in the
64 real world, this is not always possible. Therefore, this method provides
65 a way for the client to 'opt out' of receiving the response callback.
66
67 This method CAN ONLY be called from the CLIENT thread context.
68 */
69 void abandon() noexcept
70 {
71 m_client = nullptr;
72 }
73
74
75 public:
76 /// See Kit::Itc::Message. This call executes in the Client's thread context.
77 void process() noexcept override
78 {
79 if ( m_client )
80 {
81 m_client->response( *this );
82 }
83 }
84
85
86public:
87 /// Returns a reference to the contained server-request-message
88 RequestMessage<SERVICE, PAYLOAD>& getRequestMsg()
89 {
90 return m_request;
91 }
92
93 /// Returns a reference the payload associated with this request/response
94 PAYLOAD& getPayload()
95 {
96 return m_request.getPayload();
97 }
98
99protected:
100 /// Reference to the client interface whose "response" method will be called
101 CLIENT* m_client;
102
103 /// Return handler used to deliver the response
104 AsyncReturnHandler m_rh;
105
106 /// I contain the actual instantiated server request message!
107 RequestMessage<SERVICE, PAYLOAD> m_request;
108};
