21/** This abstract class defines the interface to signal a 'Event Flag'. Event
22 Flags are used to indicate that an 'event' has occurred
23
24 Each EventLoop supports up to 32 individual Event Flags (numbered 0-31).
25
26 Event Flags are used to indicate that an 'event' has occurred. Individual
27 event flags can be viewed as binary semaphores with respect to be signaled/
28 waiting (though waiting is done on the EventLoop's entire set of event flags).
29
30 A EventLoop can wait for at least one event to be signaled. When the EventLoop
31 is waiting on event(s) and it is then signaled - all of the Events that were
32 in the signaled state when the EventLoop was unblocked are cleared.
33
34 The IEventManager interface is inherently thread safe.
35 */
36class IEventManager
37{
38public:
39 /** This operation is called by clients to signal one or more Event Flags.
40 Each bit in 'events' is different Event Flag.
41
42 This method can ONLY be called from a thread context.
43 */
44 virtual void signalMultipleEvents( uint32_t events ) noexcept = 0;
45
46 /** This operation is similar to signalEvents(), except that it sets one
47 event flags. The 'eventNumber' is the bit number (zero based) of the
48 Event Flag to set.
49
50 This method can ONLY be called from a thread context.
51 */
52 virtual void signalEvent( uint8_t eventNumber ) noexcept = 0;
53
54
55public:
56 /** This method is same as signalMultipleEvents() EXCEPT this method can
57 ONLY be called from supervisor mode and/or ISR contexts.
58 */
59 virtual void su_signalMultipleEvents( uint32_t events ) noexcept = 0;
60
61 /** This method is same as signalEvent() EXCEPT this method can ONLY be
62 called from supervisor mode and/or ISR contexts.
63 */
64 virtual void su_signalEvent( uint8_t eventNumber ) noexcept = 0;
65
66public:
67 /// Virtual Destructor
68 virtual ~IEventManager() = default;
69
70};
