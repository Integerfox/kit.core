38/** This concrete class is a 'server' that flashes the LED. It responds to
39 rate requests from a 'client'. A software timer is used to manage the LED
40 flashing.
41 */
42class Server : public RateSync, public Kit::System::Timer, public Kit::Itc::OpenCloseSync
43{
44public:
45 /// Constructor
46 Server( Kit::EventQueue::IQueue& myEventQueue,
47 uint32_t initialFlashRateMs = 500 ) noexcept // default to 1Hz
48 : RateSync( myEventQueue )
49 , Timer( myEventQueue )
50 , Kit::Itc::OpenCloseSync( myEventQueue )
51 , m_flashRateMs( initialFlashRateMs )
52 , m_opened( false )
53 {
54 }
55
56public:
57 /// See Itc::Synchronous::IRateRequest
58 void request( RateMsg& msg ) noexcept override
59 {
60 // NOTE: Make sure to have a *reference* to the payload (instead of a copy)
61 // since we need to update the 'success' field in the payload to
62 // indicate pass/fail of the request
63 auto& payload = msg.getPayload();
64
65 // Validate the flash rate request
66 if ( ( payload.flashRateMs < OPTION_ITC_SYNCHRONOUS_SERVER_MIN_FLASH_RATE_MS ) ||
67 ( payload.flashRateMs > OPTION_ITC_SYNCHRONOUS_SERVER_MAX_FLASH_RATE_MS ) )
68 {
69 // Invalid flash rate request. Set the response value to false to indicate failure and return
70 payload.success = false;
71 }
72
73 // Update the Flash rate
74 {
75 payload.success = true;
76 m_flashRateMs = payload.flashRateMs;
77 expired(); // Expire the timer immediately to update the flash rate without waiting for the current flash cycle to complete
78 }
79
80 // Return the message to the sender since we are done processing the request
81 msg.returnToSender();
82 }
83
84public:
85 /// See Kit::Itc::OpenCloseSync
86 void request( OpenMsg& msg ) noexcept override
87 {
88 // Only 'open' when not already opened
89 if ( !m_opened )
90 {
91 // Housekeeping
92 m_opened = true;
93 m_timeMarker = Kit::System::ElapsedTime::milliseconds();
94
95 // Start the timer to begin flashing the LED
96 // Call the software timer expired function to start the flashing
97 expired();
98 }
99
100 // Return the message to the sender since we are done processing the request
101 msg.returnToSender();
102 }
103
104 /// See Kit::Itc::OpenCloseSync
105 void request( CloseMsg& msg ) noexcept override
106 {
107 // Only 'close' when already opened
108 if ( m_opened )
109 {
110 // Housekeeping
111 m_opened = false;
112 Timer::stop();
113 }
114
115 // Return the message to the sender since we are done processing the request
116 msg.returnToSender();
117 }
118
119public:
120 /// See Kit::System::ICounter, i.e. software timer expired callback
121 void expired() noexcept override
122 {
123 // Toggle the LED
124 Bsp_toggle_debug1();
125 uint32_t now = Kit::System::ElapsedTime::milliseconds();
126 KIT_SYSTEM_TRACE_MSG( "main", "LED Flash. delay=%" PRIu32 " ms", m_flashRateMs );
127
128 // Restart the timer for the next flash cycle. Account for the trace output latency
129 // by adjusting the next duration to the next m_flashRateMs boundary
130 uint32_t nextDuration = m_flashRateMs;
131 uint32_t elapsed = Kit::System::ElapsedTime::deltaMilliseconds( m_timeMarker, now );
132 if ( elapsed >= m_flashRateMs )
133 {
134 nextDuration = m_flashRateMs - ( elapsed - m_flashRateMs );
135 }
136 Timer::start( nextDuration );
137 m_timeMarker = now;
138 }
139
140
141protected:
142 /// Time marker used to account for 'printf' latency in the timer expiration callback.
143 uint32_t m_timeMarker;
144
145 /// Current flash rate
146 uint32_t m_flashRateMs;
147
148 /// Track my open/closed
149 bool m_opened;
150};
