26/** This partially abstract class defines a interface for operating on an
27 output stream (example of a stream is 'stdout' or a socket connection).
28
29 All Write calls on the stream are 'blocking' - i.e. the calls do not
30 return until one or more bytes have been written to the stream.
31
32 Note: There is really only just one write() method (the one the returns
33 'bytesWritten'). All of the other write() are convenience methods
34 and as such a default implementation is provided for these methods.
35 */
36class IOutput : virtual public IClose, virtual public IEos
37{
38public:
39 /** Writes a single byte to the stream. Returns true if successful,
40 or false if End-of-Stream was encountered.
41 */
42 virtual bool write( char c ) noexcept;
43
44 /** Writes a string to the stream. The string's trailing '\0' is stripped
45 off and not outputted to the stream. Returns true if successful, or false
46 if End-of-Stream was encountered. The method does not return until all
47 characters in the string have been written to the Output stream.
48 */
49 virtual bool write( const char* string ) noexcept;
50
51 /** Writes a string to the stream. The string's trailing '\0' is stripped
52 off and not outputted to the stream. Returns true if successful, or false
53 if End-of-Stream was encountered. The method does not return until all
54 characters in the string have been written to the Output stream.
55 */
56 virtual bool write( const Kit::Text::IString& string ) noexcept;
57
58 /** Formatted write to the stream. The formatting syntax/semantics is the
59 same as printf(). The string's trailing '\0' is stripped off and not
60 outputted to the stream. The client is required to provide storage for
61 a temporary buffer used to format the outgoing data. The contents of
62 'formatBuffer' will match what was written to the stream. Returns true
63 if successful, or false if End-of-Stream was encountered. The method
64 does not return until the all of the characters in the 'formatBuffer'
65 (after the contents have been 'formatted') is written to the Output
66 stream.
67 */
68 KIT_SYSTEM_PRINTF_CHECKER(3, 4)
69 virtual bool write( Kit::Text::IString& formatBuffer, const char* format, ... ) noexcept;
70
71 /** Same as write( String& formatBuffer, const char* format,...), except that
72 it is called with a va_list instead of a variable number of arguments.
73 */
74 virtual bool vwrite( Kit::Text::IString& formatBuffer, const char* format, va_list ap ) noexcept;
75
76 /** Writes the content of the buffer to the stream. Returns true if
77 successful, or false if End-of-Stream as encountered. The method does
78 not return until 'numBytes' have been written to the Output stream.
79 */
80 virtual bool write( const void* buffer, ByteCount_T numBytes ) noexcept;
81
82 /** Writes the content of the buffer to the stream. At most 'maxBytes'
83 will be outputted. The actual number of bytes written is returned
84 via 'bytesWritten'. Returns true if successful, or false if End-of-Stream
85 was encountered.
86 */
87 virtual bool write( const void* buffer, ByteCount_T maxBytes, ByteCount_T& bytesWritten ) noexcept = 0;
88
89 /** Forces all buffered data (if any) to be written to the stream media.
90 The actual behavior of this method for how/when the data is flushed -
91 is stream-media and platform dependent.
92 */
93 virtual void flush() noexcept = 0;
94
95
96public:
97 /// Lets the make the destructor virtual
98 virtual ~IOutput() noexcept = default;
99
100};
