31/** This concrete class is a IRunnable object that provides a event driven
32 execution model for a thread. The thread will remaining blocked until
33 an "event" occurs. The following "events" are supported:
34
35 1) The EventLoop Semaphore was signaled (this is separate from the Thread Semaphore)
36 2) A Event Flag was signaled
37 3) A Software Timer expiring
38
39 To receive a callbacks when Event Flags are signaled, the application must
40 provide a list of Event Flag instances to the Event Loop constructor.
41
42 Child classes can be used to extend the 'event processing' to include other
43 types of events, e.g. Inter-Thread-Communication messaging.
44
45 NOTE: The EventLoop does NOT use/consume the Thread Semaphore.
46 */
47class EventLoop : public IRunnable, public IEventManager, public ISignable, public TimerManager
48{
49public:
50 /** Constructor. The 'timeOutPeriodInMsec' parameter specifies how
51 long the EventLoop will wait for an event before timing out and
52 processing the software timers. The value of 'timeOutPeriodInMsec' is
53 the resolution of the timing source for the Software Timers, i.e. the
54 event loop will 'wake-up' at least every 'timeOutPeriodInMsec'
55 milliseconds.
56
57 A fatal error is generated if 'timeOutPeriodInMsec' is set to zero.
58
59 When the 'eventFlagsList' parameter is null then the EventLoop will not
60 monitor any event flags.
61
62 WARNING: The application CANNOT modify the eventFlagsList after the
63 EventLoop instance is created.
64 */
65 EventLoop( uint32_t timeOutPeriodInMsec = OPTION_KIT_SYSTEM_EVENT_LOOP_TIMEOUT_PERIOD,
66 Kit::Container::SList<IEventFlag>* eventFlagsList = nullptr ) noexcept;
67
68 /// Virtual destructor
69 virtual ~EventLoop() = default;
70
71
72protected:
73 /** This method is used to initialize the Event Loop's thread has started
74 to executed.
75
76 This method is intended to be used by child classes that are extending
77 the Event Loop. For this use case - this method MUST be called once on
78 the beginning of the entry() method and BEFORE the "main loop" for the
79 entry() method is entered.
80 */
81 virtual void startEventLoop() noexcept;
82
83 /** This method is used to wait (and process) the next event(s). This
84 method returns after being woken up and processing all timer expired
85 and pending Event Flag events. This method should always be wrapped
86 in a loop (typically a forever loop).
87
88 If the 'skipWait' argument is set to true then the method does not
89 wait for event - it simply processing the current event flags and
90 timers. The use scenario for skipping the wait is for child classes
91 to execute the event loop multiple times to drain their message/notifications
92 queues one message/notification at a time (i.e. give equal time/priority
93 to all types of events)
94
95 The method typically returns true. The method only returns false if
96 the pleaseStop() method was called on the Event Loop instance.
97
98 This method is intended to be used by child classes that are extending
99 the Event Loop. For this use case - this method MUST be called inside
100 the child class's "main loop" in its the appRun() method.
101
102 Example appRun() for a child class extending the Event Loop:
103 @code
104
105 void ChildEventLoop::entry()
106 {
107 startEventLoop();
108 <child specific initialization>
109
110 bool run = true;
111 while( run )
112 {
113 run = waitAndProcessEvents();
114 if ( run )
115 {
116 <child specific event processing>
117 }
118 }
119 stopEventLoop()
120 }
121
122 @endcode
123 */
124 virtual bool waitAndProcessEvents( bool skipWait = false ) noexcept;
125
126 /** This method is used to clean-up the Event Loop's when the thread is
127 being stopped.
128
129 This method is intended to be used by child classes that are extending
130 the Event Loop. For this use case - this method MUST be called once
131 AFTER the event-processing loop has exited.
132 */
133 virtual void stopEventLoop() noexcept;
