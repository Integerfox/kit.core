38 /** Closes the file descriptor 'fd'. If 'fd' is already closed (i.e. INVALID_SOCKET),
39 the method is a NOP. Upon a successful close, 'fd' is set to INVALID_SOCKET.
40 */
41 static void close( SOCKET& fd ) noexcept;

43public:
44 /** Attempts to write up to 'maxBytes' from 'buffer' to the file descriptor
45 'fd'. The actual number of bytes written is returned in 'bytesWritten'.
46
47 The 'eosFlag' is set to true if End-of-Stream was encountered, else it is
48 cleared (i.e. set to false)
49
50 The method blocks until at least one byte is written to the file descriptor.
51
52 Returns true if successful; else false on error.
53 */
54 static bool write( SOCKET fd, bool& eosFlag, const void* buffer, ByteCount_T maxBytes, ByteCount_T& bytesWritten ) noexcept
55 {
56 KIT_SYSTEM_ASSERT( buffer != nullptr );
57
58 // Throw an error if the socket had already been closed
59 if ( fd == INVALID_SOCKET )
60 {
61 return false;
62 }
63
64 // Ignore write requests of ZERO bytes
65 if ( maxBytes == 0 )
66 {
67 bytesWritten = 0;
68 return true;
69 }
70
71 // perform the write
72 int result = send( fd, (char*)buffer, maxBytes, 0 );
73 if ( result == SOCKET_ERROR || result == 0 ) // Error or connection closed gracefully
74 {
75 bytesWritten = 0;
76 eosFlag = true;
77 return false;
78 }
79
80 // Success!
81 bytesWritten = static_cast<ByteCount_T>(result);
82 eosFlag = false;
83 return true;
84 }

157 /** Creates the "listening" socket. It will attempt N times to binding the
158 listening port before giving up. Upon success a valid socket descriptor
159 is returned. On error INVALID_SOCKET is returned
160 */
161 static SOCKET createListeningSocket( int port,
162 unsigned numRetries,
163 uint32_t delayBetweenRetriesMs ) noexcept
164 {
165 struct sockaddr_in local;
166 int one = 1;
167
168 // Create the Socket to listen with
169 SOCKET fd = socket( AF_INET, SOCK_STREAM, 0 );
170 if ( fd == INVALID_SOCKET )
171 {
172 return INVALID_SOCKET;
173 }
174
175 // Set Options on the socket
176 setsockopt( fd, SOL_SOCKET, SO_REUSEADDR, (char*)&one, sizeof( one ) );
177
178 // Set the "address" of the socket
179 unsigned retry = numRetries;
180 while ( retry )
181 {
182 memset( &local, 0, sizeof( local ) );
183 local.sin_family = AF_INET;
184 local.sin_addr.s_addr = htonl( INADDR_ANY );
185 local.sin_port = htons( port );
186 if ( bind( fd, (struct sockaddr*)&local, sizeof( local ) ) != SOCKET_ERROR )
187 {
188 break;
189 }
190
191 // Delay between retry attempts
192 Kit::System::sleep( delayBetweenRetriesMs );
193 if ( --retry == 0 )
194 {
195 return INVALID_SOCKET;
196 }
197 }
198
199 // Create a queue to hold connection requests
200 if ( ::listen( fd, SOMAXCONN ) == SOCKET_ERROR )
201 {
202 return INVALID_SOCKET;
203 }
204
205 // Success!
206 return fd;
207 }

209 /** This method is to accept an incoming connection request. Returns
210 a valid socket descriptor for the accepted connection, or INVALID_SOCKET
211 if an error occurred. When successful, the 'client_addr' structure is

214 inline static SOCKET acceptConnection( SOCKET listeningFd, struct sockaddr_in& client_addr ) noexcept
215 {
216 // Wait on the 'accept'
217 socklen_t client_len = sizeof( client_addr );

219 }

221 /** This method enables the SO_KEEPALIVE option on the specified socket.
222 Returns 0 if successful, else on error SOCKET_ERROR is returned

224 inline static int enableKeepAlive( SOCKET fd ) noexcept
225 {
226 int bOptVal = 1;
227 int bOptLen = sizeof( int );

232 /** This resolved the specific network address string and port number (IPv4
233 or IPv6) to a socket address info structure. The method returns true if
234 successful, else false is returned.
235
236 When successful, the method returns one or MORE address info structure
237 candidates via the 'results' argument. In addition on success, the caller
238 is required to freeAddresses() method to free the memory associated
239 with the 'results' argument.
240 */
241 inline static bool resolveAddress( const char* remoteHostName, int portNumToConnectTo, struct addrinfo*& results ) noexcept
242 {
243 Kit::Text::FString<5> port( portNumToConnectTo );
244 struct addrinfo hints;
245 memset( &hints, 0, sizeof( hints ) );
246 hints.ai_family = AF_UNSPEC;
247 hints.ai_socktype = SOCK_STREAM;
248 hints.ai_protocol = IPPROTO_TCP;
249 return getaddrinfo( remoteHostName, port, &hints, &results ) == 0;
250 }
251
252 /** This method frees the memory associated with the 'results' argument
253 that was populated by a previous call to the resolveAddress() method.
254 */
255 inline static void freeAddresses( struct addrinfo* results ) noexcept
256 {
257 freeaddrinfo( results );
258 }
259
260 /** Create a file descriptor for the specified socket address info.
261 Returns a valid socket descriptor if successful, else INVALID_SOCKET is returned.
262 */
263 inline static SOCKET createSocket( struct addrinfo* addr ) noexcept
264 {
265 return socket( addr->ai_family, addr->ai_socktype, addr->ai_protocol );
266 }
