25/** This partially abstract class defines a interface for operating on an
26 input stream (example of a stream is 'stdin' or a socket connection).
27 All Read calls on the stream are 'blocking' - i.e. the calls do not
28 return until one or more bytes have been read from the stream.
29
30 Note: There is really only just one read() method (the one the returns
31 'bytesRead'). All of the other read() are convenience methods
32 and as such a default implementation is provided for these methods.
33
34 */
35class IInput : virtual public IClose, virtual public IEos
36{
37public:
38 /** Reads a single byte from the stream. Returns true if successful,
39 or false if End-of-Stream was encountered.
40
41 If the method fails, the contents of the 'c' is undefined.
42 */
43 virtual bool read( char& c ) noexcept;
44
45 /** Reads N bytes into the String's internal buffer. The number of
46 bytes read will be less or equal to the String's max length. The
47 String is guaranteed to be terminated by a '\0'. The placement
48 of the '\0' is determined by the number of bytes read from the
49 stream (e.g. buffer[bytesRead] = '\0'). Returns true if successful,
50 or false if End-of-Stream was encountered.
51
52 If the method fails, the contents of the 'destString' is undefined.
53
54 NOTE: NO FILTERING of 'non-printable' characters is done! Therefore
55 it is up to the client application to deal with the problem!
56 */
57 virtual bool read( Kit::Text::IString& destString ) noexcept;
58
59 /** This method is similar to read( Kit::Text::IString& destString ), expect
60 that it will NOT return until 'numBytesToRead' have been read.
61
62 If the method fails, the contents of the 'destString' is undefined.
63 */
64 virtual bool read( Kit::Text::IString& destString, ByteCount_T numBytesToRead ) noexcept;
65
66 /** Attempts to read the specified number of bytes from the stream in the
67 supplied buffer. The actual number of bytes read is returned via
68 'bytesRead'. Returns true if successful, or false if End-of-Stream
69 was encountered.
70
71 If the method fails, the contents of the 'buffer' is undefined.
72
73 The caller is responsible for providing a buffer that is large
74 enough to hold the requested number of bytes.
75 */
76 virtual bool read( void* buffer, ByteCount_T numBytes, ByteCount_T& bytesRead ) noexcept = 0;
77
78 /** This method is similar to read( void* buffer, int numBytes, int& bytesRead ),
79 except that it will NOT return until 'numBytesToRead' have been read.
80
81 If the method fails, the contents of the 'buffer' is undefined.
82 */
83 virtual bool read( void* buffer, ByteCount_T numBytesToRead ) noexcept;
84
85 /** Returns true if there data available to be read from the stream.
86
87 NOTE: The implementation of this method is VERY PLATFORM dependent! If
88 your code uses it - it may not be portable to all platforms.
89 If a platform does not/can not support this method it is
90 guaranteed to return 'true'
91 */
92 virtual bool available() noexcept = 0;
93
94
95public:
96 /// Lets the make the destructor virtual
97 virtual ~IInput() noexcept = default;
98
99};
