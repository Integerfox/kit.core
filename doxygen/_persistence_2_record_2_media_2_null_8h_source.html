26/** This concrete class provides a 'null' implementation of the IMedia
27 interface. The implementation is a 'happy path' implementation that does

30 storage available on the platform.
31 */

41public:
42 /// See Kit::Persistence::Record::IMedia

44
45 /// See Kit::Persistence::Record::IMedia
46 void stop() noexcept override {};
47
48public:
49 /// See Kit::Persistence::Record::IMedia.
50 bool write( Size_T offset, const void* srcData, Size_T srcLen ) noexcept override
51 {
52 Size_T endOffset = offset + srcLen;
53 if ( srcData == nullptr || endOffset > m_maxSize || endOffset < offset )
54 {
55 return false;
56 }
57 return true;
58 }
59
60 /// See Kit::Persistence::Record::IMedia.
61 Size_T read( Size_T offset, void* dstBuffer, Size_T bytesToRead ) noexcept override
62 {
63 Size_T endOffset = offset + bytesToRead;
64 if ( dstBuffer == nullptr || endOffset > m_maxSize || endOffset < offset )
65 {
66 return KIT_PERSISTENCE_SIZE_MAX;
67 }
68
69 memset( dstBuffer, 0, bytesToRead );
70 return bytesToRead;
71 }
72
73public:

75 Size_T getMaxSize() const noexcept override { return m_maxSize; }
