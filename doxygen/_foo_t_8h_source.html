21/** Data structure used to exchange data between a Client and Server for the
22 'Foo' transaction.
23
24 The following are needed because the structure is used as a 'data source'
25 for a Model Point:
26 ----------------------------------------------------------------------------
27 - The comparison operation
28 - The copy constructor and copy operator are implemented to ensure that
29 any 'pad' bytes in the structure get zero'd out. This is desired because
30 the default 'comparison operation' in the DM framework uses memcmp. A
31 more sophisticated comparison approach can be used - but it requires the
32 concrete Model Point to supply/implement the comparison method.
33 */
34struct Foo_T
35{
36 int32_t upperLimit; //!< IN: Upper limit for the random number generator
37 int32_t lowerLimit; //!< IN: Lower limit for the random number generator
38 int32_t randomNumber; //!< OUT: Random number generated by the Server and sent to the Client
39 bool isValid; //!< OUT: Indicates whether the 'randomNumber' value is valid (true) or not (false)
40
41
42 /// Constructor (to ensure any pad bytes get zero'd -->needed for brute force compare operation)
43 Foo_T()
44 {
45 memset( (void*)this, 0, sizeof( Foo_T ) );
46 }
47
48 /// Copy Constructor (to ensure any pad bytes get zero'd)
49 Foo_T( const Foo_T& other )
50 {
51 memcpy( (void*)this, (void*)&other, sizeof( Foo_T ) );
52 }
53
54 /// Constructor - all arguments
55 Foo_T( int32_t upperLimit,
