24/** This abstract class defines the interface that a contained object must
25 support if it has comparable key associated with it.
26 */
27class Key
28{
29public:
30 /** Key Compare function. Returns <0, 0, >0 if this key is less than,
31 equal, or greater than respectively to the specified key!
32 The actual type of the 'key' is up to the client sub-class
33 that implements this interface. It is the responsibility of
34 the sub-class to correctly define/interpret the data type of the key.
35
36 CAUTION: If the key being compared is NOT a compatible type or is not
37 a valid key, then INT_MIN is returned. The caller of the method
38 is responsible for checking for INT_MIN
39
40 */
41 virtual int compareKey( const Key& key ) const noexcept = 0;
42
43
44 /** Returns the object's length (in bytes) and a pointer to the start of
45 key data. If 'returnRawKeyLenPtr' is null, then no length is
46 returned.
47 */
48 virtual const void* getRawKey( unsigned* returnRawKeyLenPtr = 0 ) const noexcept = 0;
49
50public:
51 /// Ensure a Virtual destructor
52 virtual ~Key() {}
53};

57/** This template class is used to generate Key classes for most of the C/C++
58 primitive data types. This class is NOT intended for general use - it is
59 used create the typedefs that follow.
60 */
61template <class DATATYPE>
62class KeyPlainType : public Key
63{
64public:
65 /// Constructor
66 KeyPlainType( DATATYPE initialValue = 0 ) noexcept
67 : m_keyData( initialValue )
68 {
69 }
70
71
72public:
73 /// Updates the Key's content value
74 void setValue( DATATYPE newValue ) noexcept
75 {
76 m_keyData = newValue;
77 }
78
79 /// Returns the Key's content value
80 DATATYPE getKeyValue() const noexcept
81 {
82 return m_keyData;
83 }
84
85public:
86 /// See Kit::Container::Key
87 int compareKey( const Key& key ) const noexcept override
88 {
89 unsigned len = 0;
90 DATATYPE* ptr = static_cast<DATATYPE*>(const_cast<void*>(key.getRawKey( &len )));
91 if ( len != sizeof( DATATYPE ) )
92 {
93 return INT_MIN; // Not a valid key
94 }
95
96 if ( m_keyData < *ptr )
97 {
98 return -1;
99 }
100 else if ( m_keyData > *ptr )
101 {
102 return 1;
103 }
104
105 return 0;
106 }
107
108 /// See Kit::Container::Key
109 const void* getRawKey( unsigned* returnRawKeyLenPtr = 0 ) const noexcept override
110 {
111 if ( returnRawKeyLenPtr != 0 )
112 {
113 *returnRawKeyLenPtr = sizeof( DATATYPE );
114 }
115
116 return &m_keyData;
117 }
118
119
120protected:
121 /// Storage for the key
122 DATATYPE m_keyData;
123};

189 return m_stringKeyPtr;
