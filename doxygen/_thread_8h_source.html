69/** This mostly abstract class defines the operations that can be performed on
70 a thread.
71
72 NOTE: The thread contains a built in counting semaphore that is usable by
73 the developer/application. This semaphore can be used to assist in
74 implementing inter-thread-communications and/or ISR-to-thread
75 communications.
76
77 NOTE: There are default implementations provided for several member and class
78 methods. They are contained in the Thread.cpp file in this directory.
79 If your Operating System port requires a different implementation, then
80 omit compiling the Thread.cpp file (in this directory) and provided
81 your own implementation.
82 */
83class Thread : public ISignable, public Kit::Container::ListItem

89 /** This method returns the task's unique identifier. The returned identifier

92 NOTE: The returned value can NOT be used a handle/reference/pointer
93 to a Thread object instance.
94 */
95 virtual KitSystemThreadID_T getId() const noexcept;

97 /** This method returns the 'executing' state of the thread. If the method
98 returns false, the underlying thread has terminated (i.e. the IRunnable
99 object's entry() method has completed) and then the Thread object/instance
100 can be safely deleted using the destroy() method below.
101
102 NOTE: Default implementation is provided in the Thread.cpp file.
103 */
104 virtual bool isActive() const noexcept;

106 /** This method returns a reference to the thread's IRunnable object

109 virtual IRunnable& getRunnable() const noexcept;

120 /** Special version of getCurrent(). getCurrent() will trigger a FATAL error
121 when called from a NON CPL thread. tryGetCurrent() will return a nullptr

127 static Thread* tryGetCurrent() noexcept;

129 /** This method causes the CURRENT thread to wait until its 'thread

144 static bool timedWait( uint32_t timeoutInMsec ) noexcept;

146 /** This method is the similar to the isActive() method, but it operates

151 */
152 static bool isActiveThread( Thread* threadPtr ) noexcept;

175 /** This abstract class defines the client interface for walking the list of
176 threads, i.e. defines the callback method for when walking/traversing
177 the list of active threads.
178 */
179 class ITraverser
180 {
181 public:
182 /// Virtual Destructor
183 virtual ~ITraverser() {}
184
185 public:
186 /** This method is called once for ever item in the "traversee's" list.
187 The return code from the method is used by the traverser to continue
188 the traversal (eCONTINUE), or abort the traversal (eABORT).
189 */
190 virtual Kit::Type::TraverserStatus item( Thread& nextThread ) noexcept = 0;

193 /** Internal Iterator that allows the Client to traverse the list
194 active threads.
195
196 NOTES:
197 o There is difference between an "active" thread and a Thread object
198 instance. The underlying platform/OS determines if the a thread is
199 "active", i.e. the thread is created and running. The Thread object
200 gets created BEFORE the thread is active and exists AFTER thread
201 terminates/end. Why the explanation, this method only includes
202 "active" threads. Threads that have terminated, but that their
203 associated Thread object instance has not been delete/destroyed
204 will NOT show up in the active list.
205 o This call is mutex protected. It will prevent new threads from
206 being created/deleted until it has completed.
207 o A C++ iterator was purposefully not used because access to the list
208 of threads has to be mutex protected. By using an
209 custom-iterator-with-a-callback approach - the Thread class can
210 enforce proper mutex/critical-section handling.
211
212 NOTE: Default implementation is provided in the Thread.cpp file.
213 */
214 static void traverse( Thread::ITraverser& client ) noexcept;
215

218 /** This method creates a thread according to the specified parameters. The
219 thread/task begins execution immediately upon its creation.
220
221 NOTE: Not all platform support all options, e.g. The C++11 multi threading
222 support does not directly support thread priorities.
223
224
225 @returns A pointer to the Thread created, OR 0 if the was an error.
226
227 @param runnable a runnable object which is started when the task is created
228 @param name a null terminated string that represent that thread's name
229 @param priority a priority ranging from highest priority (KIT_SYSTEM_THREAD_PRIORITY_HIGHEST)
230 to lowest priority (KIT_SYSTEM_THREAD_PRIORITY_LOWEST).
231 Priorities are defined in relative 'levels' to the max/normal/min
232 settings because there is no standardized priority settings
233 across Operating Systems. If the priority value exceeds the
234 upper/lower bounds the platform's priority range, the thread's
235 priority will be clamped to the upper/lower bound of the priority
236 range.
237 @param stackSize Requested size, in bytes, of the stack for the thread. A value of
238 of zero will size the thread's stack per the platform's default
239 stack size setting.
240 @param stackPtr Pointer to the stack memory to be used for the thread's stack. If
241 zero is passed, then the underlying platform/OS will allocate the
242 stack memory.
243 @param allowSimTicks With this argument is true the thread will be created to
244 use/support simulated time when 'Simulate ticks' are enabled.
245 A value false will force the thread to always be a 'real time'
246 thread. See Cpl::System::SimTicks for more details about
247 simulated time.
248 */
249 static Thread* create( IRunnable& runnable,
250 const char* name,
251 int priority = KIT_SYSTEM_THREAD_PRIORITY_NORMAL,
252 int stackSize = 0,
253 void* stackPtr = 0,
254 bool allowSimTicks = true ) noexcept;
255

257 /** This method is used to destroy a thread that was created by the Thread
258 Factory. If you are going to delete a thread that was created by the
259 Thread Factory, you MUST use the Factory's destroy method to delete it
260 to properly insure that the memory used to create the task is released
261 correctly.
262
263 When the 'delayTimeMsToWaitIfNotStopped' argument is NOT zero, AND the
264 thread is still active. This method will call pleaseStop() on the IRunnable
265 object associated with the thread. After the delay time (in milliseconds)
266 has expired - will attempt to unconditionally terminate the thread.
267
268 NOTES:
269
270 o The application should only delete/destroy a thread AFTER the
271 thread's IRunnable object.run() method has ended/terminated. If
272 the thread is destroy/delete before run() has completed, there is
273 NO GUARANTEE with respect to whether or not the thread has
274 released all acquired resources!
275 o NOT ALL Platforms support brute-force termination of threads,
276 e.g. in C++11 there is no method/interface to forcibly terminate
277 a thread. Once again -->it is a HIGHLY DISCOURAGED to design
278 your application where you need to forcibly terminate threads.
279 o This method only deletes/destroys the Thread instance -- it does
280 NOT delete/destroy the associated IRunnable instance.
281 */
282 static void destroy( Thread& threadToDestroy, uint32_t delayTimeMsToWaitIfActive = 0 ) noexcept;
