23/** This class implements a synchronous ReturnHandler. The assumption is that
24 the client sends a message to a server mailbox, and then blocks on the
25 thread's semaphore. When the server thread invokes the rts() method, the
26 thread/semaphore is signaled, and thus the client is notified that it has
27 ownership of the message and that the server has completed processing the
28 message.
29
30 The expected typical usage is to created a SyncReturnHandler on the stack
31 in the client's thread just before the mailbox.postSync() method is called.
32 */
33class SyncReturnHandler : public IReturnHandler
34{
35public:
36 /** The constructor initializes the handler with a reference to the
37 thread semaphore of the current thread.
38 */
39 SyncReturnHandler( Kit::System::ISignable& clientSemaphore = Kit::System::Thread::getCurrent() ) noexcept;
40
41public:
42 /// See ReturnHandler
43 void rts() noexcept override;
44
45protected:
46 /** Reference to the client thread's thread semaphore which is used
47 to notify the client that the server has completed processing
48 the message.
49 */
50 Kit::System::ISignable& m_waiter;
51};
