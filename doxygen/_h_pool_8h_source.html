24/** This template class defines a concrete Allocator that allocates memory from
25 the HEAP for a memory pool that can create up to N concurrent instances of
26 class T. However, once the initial set of blocks are allocated, no more
27 heap operations are performed. All of the memory is aligned to class "T"
28 alignment boundaries.
29
30 NOTES:
31
32 1) If you only need memory for ONE instance - use AlignedClass structure
33 in Aligned.h instead.
34
35 2) The class is not inherently multi-thread safe.
36
37 3) If the requested number of bytes on the allocate() method is greater
38 than the block size (i.e. sizeof(T)), nullptr is returned if
39 'fatalErrors' is false; else a fatal error is generated.
40
41 4) The class can be deleted. However, it is the responsibility of the
42 Application to properly clean-up/release ALL outstanding block
43 allocations before deleting the HPool instance.
44
45
46 Template args: class "T" is the type of class to allocated
47 */
48
49template <class T>
50class HPool : public Allocator
51{
52protected:
53 /// Allocate memory for BlockInfo_ instances
54 Pool::BlockInfo* m_infoBlocks;
55
56 /// Allocate blocks
57 AlignedClass<T>* m_blocks;
58
59 /// My Pool work object
60 Pool* m_poolPtr;
61
62
63public:
64 /** Constructor. When the 'fatalErrors' argument is set to true, memory errors
65 (e.g. out-of-memory) will generate a Kit::System::FatalError call.
66 */
67 HPool( size_t maxNumBlocks, bool fatalErrors = false )
68 : m_infoBlocks( new Pool::BlockInfo[maxNumBlocks]() )
69 , m_blocks( new AlignedClass<T>[maxNumBlocks] )
70 , m_poolPtr( new Pool( m_infoBlocks, sizeof( AlignedClass<T> ), maxNumBlocks, m_blocks, fatalErrors ) )
71 {
72 KIT_SYSTEM_ASSERT( m_poolPtr != nullptr );
73 }
74
75
76
77 /// Destructor.
78 ~HPool()
79 {
80 delete m_poolPtr;
81 delete[] m_blocks;
82 delete[] m_infoBlocks;
83 }
84
85
86public:
87 /// See Kit::Memory::Allocator
88 void* allocate( size_t numbytes ) noexcept { return m_poolPtr->allocate( numbytes ); }
89
90 /// See Kit::Memory::Allocator
91 void release( void* ptr ) noexcept { m_poolPtr->release( ptr ); }
92
93 /// See Kit::Memory::Allocator
94 size_t wordSize() const noexcept { return m_poolPtr->wordSize(); }
95
96private:
97 /// Prevent access to the copy constructor -->HPools can not be copied!
98 HPool( const HPool& m );
99
100 /// Prevent access to the assignment operator -->HPools can not be copied!
101 const HPool& operator=( const HPool& m );
102};
