22/** This template class implements a Stack that has a fixed depth and stores
23 copies of the data items.
24
25 Template Args:
26 ITEM:= Type of the data stored in the Stack
27 */
28template <class ITEM>
29class Stack
30{
31public:
32 /** Constructor. The application is responsible for providing the memory
33 for the Stack. The argument 'maxElements' is the number of items that
34 will fit in the memory allocated by 'memoryForElements' - it is NOT
35 the number of bytes of 'memoryForElements'.
36 */
37 Stack( unsigned maxElements, ITEM memoryForElements[] ) noexcept
38 : m_elements( memoryForElements )
39 , m_count( 0 )
40 , m_maxItems( maxElements )
41 {
42 }
43
44
45public:
46 /** Adds an item to the top of the stack. Returns true if successful;
47 else false is returned (e.g. on stack overflow).
48
49 NOTE: A copy of the item is what is pushed on the stack
50 */
51 bool push( const ITEM& src ) noexcept
52 {
53 if ( isFull() )
54 {
55 return false;
56 }
57
58 m_elements[m_count++] = src;
59 return true;
60 }
61
62
63 /** Removes the top item of the stack. If the stack is empty then the
64 method returns false and no value is returned.
65 */
66 bool pop( ITEM& dst ) noexcept
67 {
68 if ( isEmpty() )
69 {
70 return false;
71 }
72
73 dst = m_elements[--m_count];
74 return true;
75 }
76
77 /** Removes AND discards the top item of the stack. If the stack is empty
78 then the method returns false and no value is returned.
79 */
80 bool pop() noexcept
81 {
82 if ( isEmpty() )
83 {
84 return false;
85 }
86
87 --m_count;
88 return true;
89 }
90
91 /** Returns a copy of the item on the top of the Stack. The state of the
92 stack is unchanged. If the stack is empty then the method returns false
93 and no value is returned.
94 */
95 bool peekTop( ITEM& dst ) const noexcept
96 {
97 if ( isEmpty() )
98 {
99 return false;
100 }
101
102 dst = m_elements[m_count - 1];
103 return true;
104 }
105
106 /** Returns a POINTER to the top item of the Stack. The state of the stack
107 is unchanged. If the stack is empty then the method returns a null
108 pointer. The pointer is valid until the next push or pop operation.
109
110 NOTE: The application CAN modify the item in place on the Stack. Use
111 this power wisely!
112 */
113 ITEM* peekTop() noexcept
114 {
115 if ( isEmpty() )
116 {
117 return nullptr;
118 }
119
120 return &( m_elements[m_count - 1] );
121}
122
123public:
124 /// This method returns true if the Stack is empty
125 bool isEmpty() const noexcept
126 {
127 return m_count == 0;
128 }
129
130 /// This method returns true if the Stack is full
131 bool isFull() const noexcept
132 {
133 return m_count == m_maxItems;
134 }
135
136 /// This method returns the current number of items in the Stack
137 unsigned getNumItems() const noexcept
138 {
139 return m_count;
140 }
141
142 /// This method returns the maximum number of items that can be stored in the Stack.
143 unsigned getMaxItems() const noexcept
144 {
145 return m_maxItems;
146 }
147
148public:
149 /// Empties the Stack. All references to the item(s) in the stack are lost.
150 void clearTheStack() noexcept
151 {
152 m_count = 0;
153 }
154
155
156protected:
157 /// Memory for the Elements
158 ITEM* m_elements;
159
160 /// Current number of items in the stack
161 unsigned m_count;
162
163 /// Maximum number of items in that the stack can hold
164 unsigned m_maxItems;
165
166
167private:
168 /// Prevent access to the copy constructor -->Containers can not be copied!
169 Stack( const Stack& m ) = delete;
170
171 /// Prevent access to the assignment operator -->Containers can not be copied!
172 Stack& operator=( const Stack<ITEM>& m ) = delete;
173
174 /// Prevent access to the move constructor -->Containers can not be implicitly moved!
175 Stack( Stack&& m ) = delete;
176
177 /// Prevent access to the move assignment operator -->Containers can not be implicitly moved!
178 Stack<ITEM>& operator=( Stack<ITEM>&& m ) = delete;
179};
