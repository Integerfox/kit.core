22/** This internal concrete class provides common methods that are used for
23 extracting and inserting data from/to a Memory Buffer. Because there are no
24 guaranties on the address alignment of data in the Memory buffer - this class
25 uses memcpy() operations to read/write data from/to the Memory buffer to avoid
26 memory address alignment errors when operating on multibyte data.
27
28 A 'public' child class is required to provide the 'Endian' semantics for the
29 memory buffer - i.e. BIG ENDIAN or LITTLE ENDIAN - when reading/writing. It
30 is the public child class that provides the methods to read/write multi-byte
31 data types (e.g. uint16_t, uint32_t, float, double, etc).
32
33 Usage:
34 1. Create an cursor instance specifying a memory buffer and optionally a
35 starting offset into the buffer.
36 2. Use the various methods to read/write data from/to the
37 memory Buffer. The cursor maintains a SINGLE internal buffer pointer
38 that is advanced by the size of each read/write operation.
39 3. The cursor will not 'overrun' the maximum memory buffer size. If an
40 overrun is attempted - the cursor enters an error state and all
41 subsequent read/write operations will fail. The error state can
42 be cleared by calling setPosition() with a valid offset.
43
44 */
45class CursorBase
46{
47protected:
48 /** Constructor. 'srcBuffer' must be a valid pointer to a memory buffer and
49 'srcBufferSize' must be less than SIZE_MAX
50 */
51 CursorBase( void* srcBuffer, size_t srcBufferSize, size_t startingOffset );
52
53
54public:
55 /** Returns true if the cursor is in an error state (e.g. an 'overrun' was
56 attempted). Once the cursor is in an error state - it can only be
57 'clear' by calling setPosition() with a valid 'newOffset'
58 */
59 bool errorOccurred() const noexcept;
