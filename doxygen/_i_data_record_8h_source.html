24/** This abstract extends the IRecord interface to include semantics for an
25 unmanaged 'data store', i.e. combines the IPayload interface with the
26 IRecord interface.
27
28 The data is only read/written from Persistent storage 'on demand' from
29 the application.
30
31 NOTE: This interface/class is NOT THREAD SAFE and should only be 'used' from
32 the Record Server's thread.
33*/
34class IDataRecord : public IRecord, public IPayload
35{
36public:
37 /** This method is used by to write record's data to persistent storage.
38 This method calls the IPayload's getData() method for the data to write
39 to persistent storage.
40
41 The 'index' offset can be used to 'index into' the IMedia instead
42 of starting at offset zero. This argument should only be used when
43 multiple 'entries' are being stored in a single Record.
44
45 The method returns true on success; else false if an error occurred.
46
47 NOTE: This method is NOT THREAD SAFE and should only be called from the
48 Record Server's thread.
49 */
50 virtual bool writeToMedia( Size_T index = 0 ) noexcept = 0;
51
52 /** This method is used by the application to read the DataRecord's data
53 from persistent storage. Note: This method calls the DataRecord's putData()
54 method with the data read from persistent storage.
55
56 The 'index' offset can be used to 'index into' the IMedia instead
57 of starting at offset zero. This argument should only be used when
58 multiple 'entries' are being stored in a single Record.
59
60 The method returns true on success; else false if an error occurred
61
62 NOTE: This method is NOT THREAD SAFE and should only be called from the
63 Record Server's thread.
64 */
65 virtual bool readFromMedia( Size_T index = 0 ) noexcept = 0;
66
67public:
68 /** This method is responsible for updating the 'data store' in RAM to
69 default values. This method is called when there is NO valid data when
70 reading the record's data from persistence storage.
71
72 The method returns true if the 'data store' is in an acceptable state
73 and the end of the call; else if there is error or there is no acceptable
74 state of the 'data store' then false is returned
75 */
76 virtual bool processNoValidData() noexcept = 0;
77};
