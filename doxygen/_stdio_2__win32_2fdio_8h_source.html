42/** This static class provides a collection of functions for operating on Win32 handles.
43 */
44class Win32IO
45{
46public:
47 /** Attempts to write up to 'maxBytes' from 'buffer' to the file descriptor
48 'fd'. The actual number of bytes written is returned in 'bytesWritten'.
49
50 The 'eosFlag' is set to true if End-of-Stream was encountered, else it is
51 cleared (i.e. set to false)
52
53 The method blocks until at least one byte is written to the file descriptor.
54
55 Returns true if successful; else false on error.
56 */
57 static bool write( HANDLE fd, bool& eosFlag, const void* buffer, ByteCount_T maxBytes, ByteCount_T& bytesWritten ) noexcept
58 {
59 // Trap that the stream has been CLOSED!
60 if ( fd == INVALID_HANDLE_VALUE )
61 {
62 return false;
63 }
64
65 // Ignore write requests of ZERO bytes
66 if ( maxBytes == 0 )
67 {
68 bytesWritten = 0;
69 return true;
70 }
71
72 // perform the write
73 unsigned long work;
74 BOOL result = WriteFile( fd, buffer, maxBytes, &work, 0 );
75 bytesWritten = (int)work;
76 DWORD lastError = GetLastError();
77 eosFlag = ( result != 0 || bytesWritten > 0 ) ? false
78 : lastError == ERROR_HANDLE_EOF || lastError == ERROR_BROKEN_PIPE || bytesWritten == 0 ? true
79 : false;
80 return result != 0 && bytesWritten > 0;
81 }

100 /** Closes the file descriptor 'fd'. If 'fd' is already closed (i.e. INVALID_HANDLE_VALUE),
101 the method is a NOP. Upon a successful close, 'fd' is set to INVALID_HANDLE_VALUE.
102 */
103 static void close( HANDLE& fd ) noexcept
104 {
105 if ( fd != INVALID_HANDLE_VALUE )

107 CloseHandle( fd );

150 /** Returns true if there is data available to be read from STDIN
151
152 THE PROBLEM with STDIN on Windoze:
153 Windows consider mouse events, change in focus, etc. as events that get
154 routed to the StdIn Handle. The ReadFile() method ignores these 'non-character'
155 events, but WaitForSingleObject() does NOT. This means that WaitForSingleObject()
156 can/will return false positive (that data is available) when in fact the
157 ReadFile() call will block.
158
159 AND to make matters worse - the console hack ONLY WORKS where is a 'live'
160 terminal window, i.e. when the terminal's stdin is piped from a file the
161 hack DOES NOT WORK! Not a very good solution :(, but it solved my
162 immediate needs.
163 */
164 static bool availableStdin( HANDLE fd ) noexcept
165 {
166 // Trap that the stream has been CLOSED!
167 if ( fd == INVALID_HANDLE_VALUE )
168 {
169 return false;
170 }
171 ISAVAILBLE( fd );
172 }
173
174 /// Returns true if there is data available to be read from the file descriptor
175 static bool available( HANDLE fd ) noexcept
176 {
177 // Trap that the stream has been CLOSED!
178 if ( fd == INVALID_HANDLE_VALUE )
179 {
180 return false;
181 }
182
183 DWORD signaled = WaitForSingleObject( fd, 0 );
184 return signaled == WAIT_OBJECT_0;
185 }
