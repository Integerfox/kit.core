23/** This abstract class defines a interface for a stream line writer. A
24 line writer allows the client to write lines to a stream. The interface
25 also handles the output of the 'new line' character(s).
26 */
27class ILineWriter
28{
29public:
30 /** Outputs the specified string to the stream. No newline character(s)
31 are written. Returns true if successful, or false if End-of-Stream was
32 encountered.
33 */
34 virtual bool print( const char* srcstring ) noexcept = 0;
35
36 /** Outputs the specified string to the stream and then appends the
37 newline character(s) to the stream. Returns true if successful, or
38 false if End-of-Stream was encountered.
39 */
40 virtual bool println( const char* srcstring ) noexcept = 0;
41
42 /** Outputs the first 'numbytes' bytes of the specified string to the
43 stream. No newline character(s) are written. Returns true if
44 successful, or false if End-of-Stream was encountered.
45 */
46 virtual bool print( const char* srcstring, int numbytes ) noexcept = 0;
47
48 /** Outputs the first 'numbytes' bytes of the specified string to the
49 stream and then appends the newline character(s) to the stream.
50 Returns true if successful, or false if End-of-Stream was
51 encountered.
52 */
53 virtual bool println( const char* srcstring, int numbytes ) noexcept = 0;
54
55
56 /** Outputs the newline character(s) to the stream.
57 */
58 virtual bool println() noexcept = 0;
59
60 /** Formatted output to the stream. The formatting syntax/semantics is the
61 same as printf(). The number of characters actually outputted to
62 stream is limited by the size of 'formatBuffer'. No newline character(s)
63 are written to the stream. Returns true if successful, or false if
64 End-of-Stream was encountered.
65 */
66 KIT_SYSTEM_PRINTF_CHECKER( 3, 4 )
67 virtual bool print( Kit::Text::IString& formatBuffer, const char* format, ... ) noexcept = 0;
68
69 /** Same as above, except newline character(s) are appended to the end of
70 the formatted output.
71 */
72 KIT_SYSTEM_PRINTF_CHECKER( 3, 4 )
73 virtual bool println( Kit::Text::IString& formatBuffer, const char* format, ... ) noexcept = 0;
74
75 /** Same as print( String& formatBuffer, const char* format,...), except that
76 it is called with a va_list instead of a variable number of arguments.
77 */
78 virtual bool vprint( Kit::Text::IString& formatBuffer, const char* format, va_list ap ) noexcept = 0;
79
80 /** Same as println( String& formatBuffer, const char* format,...), except that
81 it is called with a va_list instead of a variable number of arguments.
82 */
83 virtual bool vprintln( Kit::Text::IString& formatBuffer, const char* format, va_list ap ) noexcept = 0;
84
85 /// Forces all buffered data (if any) to be written to the stream media.
86 virtual void flush() noexcept = 0;
87
88 /// Closes the writer and the underlying output stream.
89 virtual void close() noexcept = 0;
90
91
92public:
93 /// Lets the make the destructor virtual
94 virtual ~ILineWriter() noexcept = default;
95};
