22/** This abstract class represents a item that can be contained in
23 an Map (aka a sorted list implemented using an AVL tree). The client
24 sub-class is required to implement the 'getKey()' method (from the
25 parent class KeyedItem).
26
27 NOTE: Map Items CAN be used with SList or DList containers
28 */
29class MapItem : public KeyedItem
30{
31protected:
32 /// Magic values for balance status
33 enum Balance_T : int8_t
34 {
35 eLEFT_ = -1, //!< Left child is deeper than right child
36 eEVEN_ = 0, //!< Left and right children are the same depth
37 eRIGHT_ = 1 //!< Right child is deeper than left child
38 };
39
40protected:
41 /// Constructor
42 MapItem() { initialize( nullptr ); }
43
44 /** Constructor -->special constructor to allow a Map to be
45 statically allocated. Only the Sorted List itself should ever
46 use this constructor -->not intended for Items in the list
47 */
48 MapItem( const char* ignoreThisParameter_usedToCreateAUniqueConstructor )
49 : KeyedItem( ignoreThisParameter_usedToCreateAUniqueConstructor ) {}
50
51
52 /// Initialize the node when being inserted in the tree
53 inline void initialize( MapItem* parent )
54 {
55 m_parentPtr_ = parent;
56 m_is_NOT_root_ = true;
57 m_balance_ = eEVEN_;
58 m_nextPtr_ = nullptr;
59 m_prevPtr_ = nullptr;
60 }
61
62
63protected: // Helper methods
64 /// Get tree connection/pointer
65 inline MapItem* getParent() const { return static_cast<MapItem*>(m_parentPtr_); }
66
67 /// Set tree connection/pointer
68 inline void setParent( MapItem* n ) { m_parentPtr_ = n; }
69
70 /// Get tree connection/pointer
71 inline MapItem* getLeft() const { return static_cast<MapItem*>(m_prevPtr_); };
72
73 /// Set tree connection/pointer
74 inline void setLeft( MapItem* n ) { m_prevPtr_ = n; }
75
76 /// Get tree connection/pointer
77 inline MapItem* getRight() const { return static_cast<MapItem*>(m_nextPtr_); };
78
79 /// Set tree connection/pointer
80 inline void setRight( MapItem* n ) { m_nextPtr_ = n; }
81
82
83protected:
