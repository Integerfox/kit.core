21/** This 'implementation' class provides the base functionality for a ring buffer.
22
23 The Ring buffer is ISR/Thread safe as long as following constraints are met:
24 - Writing to an 'unsigned' data type is an atomic machine instruction.
25 - There is SINGLE producer (add) and a SINGLE consumer (remove)
26
27 NOTE: This class relies on the 'volatile' qualifier to ensure that the
28 read/write indexes are properly updated across threads/ISRs. It should
29 ONLY be used with hardware targets that DO NOT support std::atomic with
30 full lock-free semantics. For example - the Arm Cortex-M0 does NOT
31 have hardware instructions to support full lock-free atomics.
32
33 NOTE: This class is not intended to be used directly. A child class is
34 required to provide the ring buffer memory and type safety.
35 */
36class RingBufferBaseVolatile
37{
38protected:
39 /// Constructor
40 RingBufferBaseVolatile( unsigned numMaxElements ) noexcept;
41
42public:
43 /// This method returns true if the Ring Buffer is empty
44 bool isEmpty() const noexcept
45 {
46 return m_readIdx == m_writeIdx;
47 }
48
49 /// This method returns true if the Ring Buffer is full
50 bool isFull() const noexcept
51 {
52 return ( ( m_writeIdx + 1 ) % m_elements ) == m_readIdx;
53 }
54
55 /// This method returns the maximum number of items that can be stored in the Ring buffer.
56 unsigned getMaxItems() const noexcept
57 {
58 return m_elements - 1;
59 }
60
61 /// This method returns the CURRENT number of elements in the Ring Buffer
62 unsigned getNumItems() const noexcept
63 {
64 return ( m_writeIdx - m_readIdx + m_elements ) % m_elements;
65 }
66
67public:
68 /** This method resets the underlying read/write indexes, i.e. effectively
69 clearing the Ring Buffer contents. This method can ONLY BE USED when
70 the Ring Buffer is NOT in use. It is the application's responsibility
71 for ensuring this condition is met.
72 */
73 void clearTheBuffer() noexcept
74 {
75 m_readIdx = m_writeIdx = 0;
76 }
77
78protected:
79 /** Removes the first item in the Ring buffer. The contents of the
80 removed item will be copied into the 'elemPtr' argument. The method
81 returns true if the operation was successful; else false is
82 returned, i.e. the Ring buffer is/was empty.
83 */
84 bool remove( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept;
85
86 /** Add a new item to the Ring Buffer as the last item in the buffer. The
87 contents of the source item (aka 'elemPtr') will be copied into the
88 Ring Buffer. Returns true if the operation was successful; else false
89 is returned, i.e. the Buffer was full prior to the attempted add().
90 */
91 bool add( const void* elemPtr, size_t elemSize, void* dstRingBufMemory ) noexcept;
92
93protected:
94 /** This method inspects the head element of the Ring buffer without removing it.
95 The contents of the head element will be copied into the 'elemPtr' argument.
96 The method returns true if the operation was successful; else false is
97 returned, i.e. the Ring buffer is empty.
98 */
99 bool peekHead( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept;
100
101 /** This method inspects the tail element of the Ring buffer without removing it.
102 The contents of the tail element will be copied into the 'elemPtr' argument.
103 The method returns true if the operation was successful; else false is
104 returned, i.e. the Ring buffer is empty.
105 */
106 bool peekTail( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept;
107
108protected:
109 /** This method returns a pointer to the next item to be removed. In addition
110 it returns the number of elements that can be removed as linear/flat
111 buffer (i.e. without wrapping around raw buffer memory).
112
113 The returned pointer and element count are only valid till the next
114 read/remove operation.
115
116 If the Ring buffer is empty, a null pointer is returned
117 */
118 void* peekNextRemoveItems( unsigned& dstNumFlatElements, size_t elemSize, const void* srcRingBufMemory ) const noexcept;
119
120 /** This method 'removes' N elements - that were removed using the
121 pointer returned from peekNextRemoveItems - from the ring buffer.
122 Basically it updates the head pointer to reflect items removed using
123 direct memory access.
124
125 'numElementsToRemove' be less than or equal to the 'dstNumFlatElements'
126 returned from peekNextRemoveItems().
127
128 CAUTION: IF YOU DON'T UNDERSTAND THE USE CASE FOR THIS METHOD - THEN
129 DON'T USE IT. If this method is used improperly, it WILL
130 CORRUPT the Ring Buffer!
131 */
132 void removeElements( unsigned numElementsToRemove ) noexcept
133 {
134 // By definition/design - I simply have to update the read index
135 m_readIdx = ( m_readIdx + numElementsToRemove ) % m_elements;
136 }
137
138protected:
139 /** This method returns a pointer to the next item to be added. In addition
140 it returns the number of elements that can be added as linear/flat
141 buffer (i.e. without wrapping around raw buffer memory)
142
143 If the Ring buffer is full, a null pointer is returned
144 */
145 void* peekNextAddItems( unsigned& dstNumFlatElements, size_t elemSize, const void* srcRingBufMemory ) const noexcept;
146
147 /** This method 'adds' N elements - that were populated using the
148 pointer returned from peekNextAddItems - to the ring buffer. Basically
149 its updates the tail pointer to reflect items added using direct
150 memory access.
151
152 'numElementsAdded' be less than or equal to the 'dstNumFlatElements'
153 returned from peekNextAddItems().
154
155 CAUTION: IF YOU DON'T UNDERSTAND THE USE CASE FOR THIS METHOD - THEN
156 DON'T USE IT. If this method is used improperly, it WILL
157 CORRUPT the Ring Buffer!
158 */
159 void addElements( unsigned numElementsAdded ) noexcept
160 {
161 // By definition/design - I simply have to update the write index
162 m_writeIdx = ( m_writeIdx + numElementsAdded ) % m_elements;
163 }
164
165protected:
166 /** Index of the next item to be read from the ring. This variable is only
167 updated when a read operation is performed. In addition, the variable
168 is atomically updated.
169 */
170 volatile unsigned m_readIdx;
171
172 /** Index of the next item to be added to the ring. This variable
173 is only updated when a add operation is performed. In addition,
174 the variable is atomically updated.
175 */
176 volatile unsigned m_writeIdx;
177
178 /// Maximum number of element in the 'ring memory
179 const unsigned m_elements;
180
181private:
182 /// Prevent access to the copy constructor -->Ring Buffers can not be copied!
183 RingBufferBaseVolatile( const RingBufferBaseVolatile& m ) = delete;
184
185 /// Prevent access to the assignment operator -->Ring Buffers can not be copied!
186 RingBufferBaseVolatile& operator=( const RingBufferBaseVolatile& m ) = delete;
187
188 /// Prevent access to the move constructor -->Ring Buffers can not be implicitly moved!
189 RingBufferBaseVolatile( RingBufferBaseVolatile&& m ) = delete;
190
191 /// Prevent access to the move assignment operator -->Ring Buffers can not be implicitly moved!
192 RingBufferBaseVolatile& operator=( RingBufferBaseVolatile&& m ) = delete;
193};
