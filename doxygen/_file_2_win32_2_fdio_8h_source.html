36/** This static class provides a collection of functions for operating on Windows

45 /** Opens a file. The default arguments are set for:
46 o Read/Write access
47 o Do NOT create the file if the file does not exist
48 o Do NOT truncate the file contents
49 */
50 static KitIoFileHandle_T open( const char* fileEntryName, bool readOnly = true, bool forceCreate = false, bool forceEmptyFile = false ) noexcept
51 {
52 KIT_SYSTEM_ASSERT( fileEntryName != nullptr );
53
54 // Set open flags as requested
55 DWORD createOpt = OPEN_EXISTING;
56 DWORD accessOpt = readOnly ? GENERIC_READ : GENERIC_READ | GENERIC_WRITE;
57 if ( forceCreate )
58 {
59 createOpt = OPEN_ALWAYS;
60 }
61 if ( forceEmptyFile )
62 {
63 createOpt = CREATE_ALWAYS;
64 }
65
66 // Open the file
67 KitIoFileHandle_T fd( CreateFile( fileEntryName,
68 accessOpt,
69 FILE_SHARE_READ | FILE_SHARE_WRITE,
70 0,
71 createOpt,
72 FILE_ATTRIBUTE_NORMAL,
73 0 ) );
74 return fd;
75 }

77public:
78 /** Returns the length, in bytes, of the file. If there is an error than
79 false is returned. Note: the implementation of this method is brute force
80 in that it sets the file position indicator to EOF and uses the EOF byte
81 offset as the length. The file position indicator is restored to its
82 original position before returning.
83 */
84 static bool length( KitIoFileHandle_T fd, ByteCount_T& length ) noexcept
85 {
86 if ( fd == INVALID_HANDLE_VALUE )
87 {
88 return false;
89 }
90
91 DWORD len = GetFileSize( fd, 0 );
92
93 // GetFileSize returns INVALID_FILE_SIZE on error, but this is also a valid
94 // file size (4GB - 1). Must check GetLastError() to distinguish.
95 if ( len == INVALID_FILE_SIZE )
96 {
97 if ( GetLastError() != NO_ERROR )
98 {
99 return false;
100 }
101 }
102
103 length = static_cast<ByteCount_T>( len );
104 return true;
105 }

108 /** Returns the current file pointer offset, in bytes, from the top of the
109 file. If there is an error than false is returned.
110 */
111 inline static bool currentPos( KitIoFileHandle_T fd, ByteCount_T& currentPos ) noexcept
112 {
113 if ( fd == INVALID_HANDLE_VALUE )
114 {
115 return false;
116 }
117
118 DWORD pos = SetFilePointer( fd, 0, 0, FILE_CURRENT );
119
120 // SetFilePointer returns INVALID_SET_FILE_POINTER on error, but this is also
121 // a valid position. Must check GetLastError() to distinguish.
122 if ( pos == INVALID_SET_FILE_POINTER )
123 {
124 if ( GetLastError() != NO_ERROR )
125 {
126 return false;
127 }
128 }
129
130 currentPos = static_cast<ByteCount_T>( pos );
131 return true;
132 }

134 /** Adjusts the current pointer offset by the specified delta (in bytes).
135 Returns true if successful, else false (i.e. setting the pointer
136 past/before the file boundaries).

138 inline static bool setRelativePos( KitIoFileHandle_T fd, ByteCount_T deltaOffset ) noexcept
139 {
140 if ( fd == INVALID_HANDLE_VALUE )
141 {
142 return false;
143 }
144
145 DWORD result = SetFilePointer( fd, deltaOffset, 0, FILE_CURRENT );
146
147 // SetFilePointer returns INVALID_SET_FILE_POINTER on error, but this is also

209 /** Creates a new, empty file. If the file already exists, the call
210 fails. Returns true if successful.
211 */
212 static bool createFile( const char* fileName ) noexcept
213 {
214 KIT_SYSTEM_ASSERT( fileName != nullptr );
215
216 // Set open flags to only open+create if the file does not already exist
217 DWORD createOpt = CREATE_NEW;
218 DWORD accessOpt = GENERIC_READ | GENERIC_WRITE;
219
220 // Open the file to create it
221 KitIoFileHandle_T fd( CreateFile( fileName,
222 accessOpt,
223 FILE_SHARE_READ | FILE_SHARE_WRITE,
224 0,
225 createOpt,
226 FILE_ATTRIBUTE_NORMAL,
227 0 ) );
228 bool result = ( fd != INVALID_HANDLE_VALUE );
229 Fdio::close( fd );
230 return result;
231 }

243 /** Renames and/or moves a file. Returns true if successful.
244 Note: On Windoze, the rename() command does NOT move directories,
245 but it will move files. The rename() command will fail with 'EACCES'
246 when attempting to move a directory.
247 */
248 static bool move( const char* oldName, const char* newName ) noexcept
249 {
250 KIT_SYSTEM_ASSERT( oldName != nullptr );
251 KIT_SYSTEM_ASSERT( newName != nullptr );
252 return ::rename( oldName, newName ) == 0;
253 }
254
255 /** Removes a file or directory. The directory must be empty to be removed.
256 Returns true if successful.
257 */
258 static bool remove( const char* fsEntryName )
259 {
260 KIT_SYSTEM_ASSERT( fsEntryName != nullptr );
261 if ( ::remove( fsEntryName ) != 0 ) // NOTE: remove() on Windoze does NOT delete directories!
262 {
263 return _rmdir( fsEntryName ) == 0;
264 }
265 return true;
266 }

269 /** Returns the first entry in the directory. Each successful 'find-first'
270 call must be followed (at some point) by a call to closeDirectory().
271
272 The entry name is copied into 'dstEntryName'. The 'maxEntryNameSize'
273 argument specifies the size - which needs to include space for the null
274 terminator - of the 'dstEntryName' buffer. If there are no entries in
275 the directory, then 'dstEntryName' is set to an empty string.
276
277 Returns true if successful; else false if a file system error was
278 encountered.
279 */
280 static bool findFirstDirEntry( KitIoFileDirectory_T& hdl,
281 const char* dirName,
282 NameString& dstEntryName ) noexcept
283 {
284 KIT_SYSTEM_ASSERT( dirName != nullptr );
285
286 WIN32_FIND_DATA fdata;
287 hdl = FindFirstFile( dirName, &( fdata ) );
288 if ( hdl != INVALID_HANDLE_VALUE )
289 {
290 dstEntryName = fdata.cFileName;
291 return true;
292 }
293
294 // Check if the error is 'no files found'
295 if ( GetLastError() == ERROR_FILE_NOT_FOUND )
296 {
297 dstEntryName.clear();
298 return true;
299 }
300
301 // File system error
302 return false;
303 }
304
305 /** Closes a directory. Can only be called after a successful call to
306 findFirstDirEntry()
307 */
308 static void closeDirectory( KitIoFileDirectory_T& hdl ) noexcept
309 {
310 FindClose( hdl );
311 }
312
313 /** Reads 'next' entry in the directory. The entry name is copied into
314 'dstEntryName'. If there are no more entries in the directory, then
315 'dstEntryName' is set to an empty string. The 'maxEntryNameSize'
316 argument specifies the size - which needs to include space for the null
317 terminator - of the 'dstEntryName' buffer.
318
319 This method can only be called after a successful call to
320 findFirstDirEntry().
321
322 Returns true if successful; else false if a file system error was
323 encountered.
324 */
325 static bool findNextDirEntry( KitIoFileDirectory_T& hdl, NameString& dstEntryName ) noexcept
326 {
327 WIN32_FIND_DATA fdata;
328 if ( FindNextFile( hdl, &fdata ) )
329 {
330 dstEntryName = fdata.cFileName;
331 return true;
332 }
333
334 // Check if the error is 'no files found'
335 if ( GetLastError() == ERROR_NO_MORE_FILES )
336 {
337 dstEntryName.clear();
338 return true;
339 }
340
341 // File system error
342 return false;
343 }
