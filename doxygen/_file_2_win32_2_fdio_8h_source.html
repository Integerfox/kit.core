28/** This static class provides a collection of functions for operating on Windows

31 NOTE: All file name/paths MUST be in native format (i.e. use '\' as the

44public:
45 /** Returns the length, in bytes, of the file. If there is an error than

50 */

53public:

59 /** Adjusts the current pointer offset by the specified delta (in bytes).

64
65 /** Sets the file pointer to the absolute specified offset (in bytes).

68 */
69 static bool setAbsolutePos( KitIoFileHandle_T fd, ByteCount_T newoffset ) noexcept;
70
71 /** Sets the file pointer to End-Of-File. Returns true if successful, else

74 static bool setToEof( KitIoFileHandle_T fd ) noexcept;
75
76public:
77 /** Returns information about the file system entry. If there is any

82 /** Creates a new, empty file. If the file already exists, the call

92public:
93 /** Renames and/or moves a file. Returns true if successful.

97 */
98 static bool move( const char* oldName, const char* newName ) noexcept;
99
100 /** Removes a file or directory. The directory must be empty to be removed.

102 */
103 static bool remove( const char* fsEntryName );
104
105public:
106 /** Returns the first entry in the directory. Each successful 'find-first'
107 call must be followed (at some point) by a call to closeDirectory().
108
109 The entry name is copied into 'dstEntryName'. The 'maxEntryNameSize'
110 argument specifies the size - which needs to include space for the null

113
114 Returns true if successful; else false if a file system error was
115 encountered.
116 */
117 static bool findFirstDirEntry( KitIoFileDirectory_T& hdl,
118 const char* dirName,
119 NameString& dstEntryName ) noexcept;
120
121 /** Closes a directory. Can only be called after a successful call to

123 */
124 static void closeDirectory( KitIoFileDirectory_T& hdl ) noexcept;
125
126 /** Reads 'next' entry in the directory. The entry name is copied into
127 'dstEntryName'. If there are no more entries in the directory, then
128 'dstEntryName' is set to an empty string. The 'maxEntryNameSize'
129 argument specifies the size - which needs to include space for the null
130 terminator - of the 'dstEntryName' buffer.
131
132 This method can only be called after a successful call to
133 findFirstDirEntry().
134
135 Returns true if successful; else false if a file system error was
136 encountered.
137 */
138 static bool findNextDirEntry( KitIoFileDirectory_T& hdl, NameString& dstEntryName ) noexcept;
