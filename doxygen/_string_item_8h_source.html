23/** This partially concrete provides a wrapper class that allows a IString
24 object to be stored directly in one the KIT Containers. A StringItem
25 can be stored in a SList, DList, and Map.
26
27 Notes:
28
29 o The allocation of the actual IString is deferred to child classes.
30
31 o When storing a StringItem in a 'keyed' container (i.e. Map) the
32 application MUST NOT change the value of the internal string since it
33 is the 'key' value that is used to determine its placement in the Map.
34 Modifying the internal IString while the StringItem is in a Map will
35 CORRUPT the Container.
36
37 o StringItem instances CANNOT be copied. This is to avoid potential
38 Container corruption since the individual Items contain the memory
39 for the Containers' linkage, i.e. a copied container Item is
40 analogous to storing a single Item in more than one Container at the
41 same time.
42 */
43class StringItem : public Kit::Container::MapItem
44{
45protected:
46 /// Constructor
47 StringItem( IString& string ) noexcept
48 : m_myString( string ) {}
49
50public:
51 /// Returns a reference to the actual IString
52 IString& get() const noexcept { return m_myString; }
53
54 /// Returns a pointer to the internal C string
55 const char* getString() const noexcept { return m_myString.getString(); }
56
57 /// Cast to a IString reference
58 operator IString&() const noexcept { return m_myString; }
59
60 /// Cast to read-only character string pointer.
61 operator const char*() const noexcept { return m_myString.getString(); }
62
63 /// Returns a Read-only pointer to the "raw" (short-hand for getString())
64 const char* operator()() const noexcept { return m_myString.getString(); }
65
66
67public:
68 /// Return Kit::Container::Key (A IString is-a Key)
69 const Kit::Container::Key& getKey() const noexcept { return m_myString; }
70
71
72protected:

74 IString& m_myString;
