20/** This protected base class implements a NON-TYPE-SAFE a singly linked list
21 which maintains the ordering imposed on it by the application. It is
22 intended to be the base class for a type-safe template class. Having
23 a non-templatized base class reduces the memory footprint when an
24 application has more than one type specific SListBase class
25 */
26class SListBase
27{
28protected:
29 /// Constructor initializes head and tail pointers.
30 SListBase() noexcept;
31
32 /** This is a special constructor for when the list is
33 statically declared (i.e. it is initialized as part of
34 C++ startup BEFORE main() is executed. C/C++ guarantees
35 that all statically declared data will be initialized
36 to zero by default (see r.8.4 in C++ Programming Language,
37 Second Edition). Since the head & tail pointers are
38 initialized to zero - then the C/C++ default is OK. Why
39 do I care about all this? Because if you attempt to build
40 static list(s), then the order of when the list is
41 constructed vs. when the static elements are added to the
42 list is a problem! To avoid this problem, a alternate
43 constructor that does NOT initialize the head & tail pointers
44 is provided. It assumes that the pointers are already set
45 zero because the list is "static". Whew!
46 */
47 SListBase( const char* ignoreThisParameter_usedToCreateAUniqueConstructor ) noexcept
48 {
49 // intentially DO NOTHING
50 }
51
52protected:
