25/** This concrete class is responsible for starting and stopping a list

27 for the IRecords.
28 */
29class Server : public Kit::Itc::OpenCloseSync
30{
31public:
32 /** Constructor. The argument 'timingTickInMsec' specifies the timing
33 resolution that will be used for Cpl::Timer::Local Timers.
34
35 NOTE: 'recordList' is variable length array where the last entry in the
36 array MUST BE a nullptr.
37 */
38 Server( Kit::EventQueue::IQueue& myEventQueue,
39 IRecord* recordList[],
40 unsigned numRecords ) noexcept
41 : OpenCloseSync( myEventQueue )
42 , m_records( recordList )
43 , m_numRecords( numRecords )
44 , m_opened( false )
45 {
46 KIT_SYSTEM_ASSERT( recordList );
47 }
48
49public:
50 /// See Kit::Itc::IOpenRequest
51 void request( OpenMsg& msg ) noexcept override
52 {
53 if ( !m_opened )
54 {
55 m_opened = true;
56
57 // Start each record
58 for ( unsigned i = 0; i < m_numRecords && m_opened; i++ )
59 {
60 m_opened &= m_records[i]->start( m_eventQueue );
61 }
62 }
63
64 msg.returnToSender();
65 }
66
67 /// See Kit::Itc::ICloseRequest
68 void request( CloseMsg& msg ) noexcept override
69 {
70 if ( m_opened )
71 {
72 m_opened = false;
73
74 // Stop each record
75 for ( unsigned i = 0; i < m_numRecords; i++ )
76 {
77 m_records[i]->stop();
78 }
79 }
80 msg.returnToSender();
81 }
82
83protected:
84 /// Variable length list of Records to manage. T
85 IRecord** m_records;
86
87 /// Number of elements in the record list
88 unsigned m_numRecords;
89
90 /// Track my open state
91 bool m_opened;
92};
