50/** This concrete class is a 'client' to request/trigger the 'server' to generate
51 a random number. See the README.md file for more details
52 */
53class Client : public Kit::Itc::OpenCloseSync, public Kit::System::Timer
54{
55public:
56 /// Constructor
57 Client( Kit::EventQueue::IQueue& myMbox,
58 MpFoo& mpFoo )
59 : Kit::Itc::OpenCloseSync( myMbox )
60 , Kit::System::Timer( myMbox )
61 , m_mpFoo( mpFoo )
62 , m_obFoo( myMbox )
63 , m_opened( false )
64 {
65 m_obFoo.setCallback<Client, &Client::triggerChanged>( this );
66 }
67
68public:
69 /// See Kit::Itc::OpenCloseSync
70 void request( OpenMsg& msg ) noexcept override
71 {
72 // Only 'open' when not already opened
73 if ( !m_opened )
74 {
75 // Housekeeping
76 m_opened = true;
77 m_offset = 0;
78 m_iterationCount = 0;
79
80 // Send the first server request
81 expired();
82 }
83
84 // Return the message to the sender since we are done processing the request
85 msg.returnToSender();
86 }
87
88 /// See Kit::Itc::OpenCloseSync
89 void request( CloseMsg& msg ) noexcept override
90 {
91 // Only 'close' when already opened
92 if ( m_opened )
93 {
94 // Housekeeping
95 m_opened = false;
96
97 // Make sure the all of the potential callbacks get cancelled
98 Timer::stop();
99 m_mpFoo.detach( m_obFoo );
100 }
101
102 // Return the message to the sender since we are done processing the request
103 msg.returnToSender();
104 }
105
106
107protected:
108 /// Callback for when the software timer expires
109 void expired() noexcept override
110 {
111 // Check for the Application-Exit condition
112 if ( m_iterationCount >= OPTION_DM_TRANSACTION_CLIENT_NUM_ITERATIONS )
113 {
114 Kit::System::Shutdown::success();
115 return;
116 }
117
118 // Trigger the server request and subscribe for the server's response
119 m_iterationCount++;
120 int32_t lowerLimit = m_offset;
121 int32_t upperLimit = m_offset + OPTION_DM_TRANSACTION_CLIENT_LIMIT_RANGE;
122 uint16_t seqNum = m_mpFoo.triggerRequest( upperLimit, lowerLimit );
123 m_mpFoo.attach( m_obFoo, seqNum );
124 }
125
126 /// Callback when the 'trigger' MP changes
127 void triggerChanged( MpFoo& mpThatChanged, Kit::Dm::IObserver& observer ) noexcept
128 {
129 Foo_T data;
130 if ( mpThatChanged.readAndSync( data, observer ) )
131 {
132 KIT_SYSTEM_TRACE_MSG( "main",
133 "client: RESPONSE. upper=%" PRId32 ", lower=%" PRId32 ", result=%" PRId32 ", isValid=%s",
134 data.upperLimit,
135 data.lowerLimit,
136 data.randomNumber,
137 data.isValid ? "true" : "false" );
138
139 // Cancel the callback subscription since we got the response from the server
140 mpThatChanged.detach( m_obFoo );
141
142 // "reset" the MP for next transaction
143 mpThatChanged.setInvalid();
144
145 // Start the timer for the next transaction.
146 m_offset += OPTION_DM_TRANSACTION_CLIENT_TRANSACTION_DELTA;
147 Timer::start( OPTION_DM_TRANSACTION_CLIENT_REQUEST_INTERVAL_MS );
148 }
149 }
150
151protected:
152 /// MP for the Client/Server transaction
153 MpFoo& m_mpFoo;
154
155 /// Observer callback for the MP change notification
156 Kit::Dm::ObserverCallback<MpFoo> m_obFoo;
157
158 /// Offset for upper/lower limit for the next transaction
159 int32_t m_offset;
160
161 /// Iteration count for the Client/Server transaction
162 unsigned m_iterationCount;
163
164 /// Track my open/closed
165 bool m_opened;
166};
