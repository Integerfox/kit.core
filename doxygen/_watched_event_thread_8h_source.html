24/** This class implements the IWatchedEventLoop interface and provides
25 watchdog monitoring capabilities for event loops. It inherits from
26 WatchedThread to maintain watchdog state and has a dependency relationship
27 with the Supervisor class.
28
29 One instance of WatchedEventThread can be designated as the 'supervisor thread'
30 which calls Supervisor::monitorThreads() every tick/event cycle.
31
32 Event threads have a software notify timer that performs periodic health checks.
33 If the health check fails, the watchdog is immediately tripped. The health check
34 interval must be less than the thread's watchdog timeout interval. The health
35 check is customizable on a per-thread basis via the performHealthCheck() virtual method.
36 */
37class WatchedEventThread : public IWatchedEventLoop, public WatchedThread
38{
39public:
40 /** Constructor
41 @param wdogTimeoutMs The watchdog timeout period for this thread in milliseconds
42 @param healthCheckIntervalMs The health check interval in milliseconds (must be < wdogTimeoutMs)
43 @param isSupervisor Set to true if this thread should act as the supervisor thread
44 */
45 WatchedEventThread( uint32_t wdogTimeoutMs, uint32_t healthCheckIntervalMs, bool isSupervisor = false ) noexcept;
46
47public:
48 /// See IWatchedEventLoop
49 void startWatcher( Kit::System::TimerManager& timingSource ) noexcept override;
50
51 /// See IWatchedEventLoop
52 void stopWatcher() noexcept override;
53
54 /// See IWatchedEventLoop
55 void monitorWdog() noexcept override;
56
57 /// See IWatchedEventLoop
58 bool isSupervisorThread() const noexcept override;
59
60protected:
61 /** Performs a health check for this thread. This is called periodically by the
62 software notify timer. If this method returns false, the watchdog will be
63 immediately tripped to force a system reset.
64
65 The default implementation always returns true (healthy). Derived classes
66 can override this method to provide application-specific health checks.
67
68 @return true if the thread is healthy, false if the watchdog should be tripped
69 */
70 virtual bool performHealthCheck() noexcept;
71
72 /** Timer expired callback - called by the TimerComposer when the health check timer expires.
73 This method performs the health check and restarts the timer.
74 */
75 void healthTimerExpired() noexcept;
76
77protected:
78 /// Timer composer for health check timer
79 Kit::System::TimerComposer<WatchedEventThread> m_timer;
80
81 /// Health check interval in milliseconds
82 uint32_t m_healthCheckIntervalMs;
83
84 /// Flag indicating if this is the supervisor thread
85 bool m_isSupervisor;
86
87 /// Flag indicating if the watcher is currently active
88 bool m_isActive;
89};
