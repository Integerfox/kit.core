53public:
54 /// See Kit::Framing::ISource
55 bool read( void* dstBuffer,
56 Kit::Type::SSize_T numBytes,
57 Kit::Type::SSize_T& bytesRead ) noexcept override
58 {
59 // Validate parameters AND trap end-of-string has already been reached.
60 if ( dstBuffer == nullptr || m_srcPtr == nullptr || m_currentIdx < 0 || numBytes < 0 )
61 {
62 bytesRead = 0;
63 return false;
64 }
65
66 // Determine how many bytes can been read
67 Kit::Type::SSize_T availableBytes = m_length - m_currentIdx;
68
69 // Requesting all of the remaining bytes
70 if ( numBytes >= availableBytes )
71 {
72 bytesRead = availableBytes;
73 memcpy( dstBuffer, m_srcPtr + m_currentIdx, availableBytes );
74 m_currentIdx = -1; // Mark end of string reached
75 return true;
76 }
77
78 // Partial read of the remaining bytes
79 memcpy( dstBuffer, m_srcPtr + m_currentIdx, numBytes );
80 m_currentIdx += numBytes;
81 bytesRead = numBytes;
82 return true;
83 }
84
85protected:
86 /// Underlying input string
87 const char* m_srcPtr;
88
89 /// Current read position within the input string. Negative value indicates the end of the string has been reached.
90 Kit::Type::SSize_T m_currentIdx;
91
92 /// Length of the input string (excluding the null terminator)
93 Kit::Type::SSize_T m_length;
