21/** This template class implements an Map using an AVL Binary tree
22 (i.e. a sorted list with fast searches, inserts, and deletes).
23
24 NOTE: The MAP does not allow entries with duplicate key value (i.e. the
25 insert() method will fail when attempting to add an item with a
26 duplicate key value as existing item in the MAP).
27
28 Template ARGS:
29 ITEM - Data type of the object stored in the Map. 'ITEM' must
30 be a sub-class of the Cpl::Container::MapItem base
31 class.
32 */
33
34template <class ITEM>
35class Map
36{
37private:
38 /// Delegate operations to the generic tree implementation
39 AvlTree_ m_tree;
40
41public:
42 /// Constructor.
43 Map() noexcept = default;
44
45 /** This is a special constructor for when the Map is
46 statically declared (i.e. it is initialized as part of
47 C++ startup BEFORE main() is executed. C/C++ guarantees
48 that all statically declared data will be initialized
49 to zero by default (see r.8.4 in C++ Programming Language,
50 Second Edition). Since the head & tail pointers are
51 initialized to zero - then the C/C++ default is OK. Why
52 do I care about all this? Because if you attempt to build
53 static list(s), then the order of when the Map is
54 constructed vs. when the static elements are added to the
55 list is a problem! To avoid this problem, a alternate
56 constructor that does NOT initialize the head & tail pointers
57 is provided. It assumes that the pointers are already set
58 zero because the list is "static". Whew!
59 */
60 Map( const char* ignoreThisParameter_usedToCreateAUniqueConstructor ) noexcept
61 : m_tree( ignoreThisParameter_usedToCreateAUniqueConstructor )
62 {
63 // intentially DO NOTHING
64 }
65
66public:
67 /** Inserts an item into the tree. If the node is successfully inserted,
68 then the method return true. If the tree already contains a node
69 with the same key, then the method returns false.
70 */
71 bool insert( ITEM& node ) noexcept
72 {
73 return m_tree.insert( node );
74 }
75
76 /** Removes the node (if it exists) that has the matching key. Returns
77 the node removed from the tree or 0 if no key match was found.
78 */
79 ITEM* remove( const Key& keyOfItemToDelete ) noexcept
80 {
81 ITEM* nodePtr = find( keyOfItemToDelete );
82 if ( nodePtr )
83 {
84 return (ITEM*)m_tree.removeItem( *nodePtr );
85 }
86 return nullptr;
87 }
88
89 /** Removes the specified item from the tree. Returns true
90 when the node was found and removed; else false is
91 returned (i.e. node not exists in the tree).
92 */
93 bool removeItem( ITEM& node ) noexcept
94 {
95 return m_tree.removeItem( node ) != nullptr;
96 }
97
98 /** Searches for a item with a matching key. Returns the node that
99 matches, else 0.
100 */
101 ITEM* find( const Key& keyToFind ) const noexcept
102 {
103 return (ITEM*)m_tree.find( keyToFind );
104 }
105
106 /** Returns true if the specified item is in the map; else false
107 is returned.
108 */
109 bool isInMap( ITEM& node ) const noexcept
110 {
111 return node.isInContainer_( this );
112 }
113
114
115 /// Returns the first item in the tree. The item remains in the tree. Returns 0 if tree is empty
116 ITEM* first() const noexcept
117 {
118 return (ITEM*)m_tree.first();
119 }
120
121 /// Returns the last item in the tree. The item remains in the tree. Returns 0 if tree is empty
122 ITEM* last() const noexcept
123 {
124 return (ITEM*)m_tree.last();
125 }
126
127 /// Returns the next item in the tree. The item remains in the tree. Returns 0 if at the end-of-tree
128 ITEM* next( ITEM& current ) const noexcept
129 {
130 return (ITEM*)m_tree.next( current );
131 }
132
133 /// Returns the previous item in the tree. The item remains in the tree. Returns 0 if at the start-of-tree
134 ITEM* previous( ITEM& current ) const noexcept
135 {
136 return (ITEM*)m_tree.previous( current );
137 }
138
139
140 /** Removes the first item in the list. Returns 0 if the list
141 is empty.
142 */
143 ITEM* getFirst() noexcept
144 {
145 ITEM* nodePtr = first();
146 if ( nodePtr )
147 {
148 return (ITEM*)m_tree.removeItem( *nodePtr );
149 }
150 return nullptr;
151 }
152
153 /** Removes the last item in the list. Returns 0 if the list
154 is empty.
155 */
156 ITEM* getLast() noexcept
157 {
158 ITEM* nodePtr = last();
159 if ( nodePtr )
160 {
161 return (ITEM*)m_tree.removeItem( *nodePtr );
162 }
163 return nullptr;
164 }
165
166
167public:
168 /// Moves the content of the this Map to the specified Map.
169 void move( Map<ITEM>& dst )
170 {
171 // clear the destination map
172 dst.clearTheMap();
173
174 // Copy each item (so the debug info is correct)
175 ITEM* nextPtr;
176 while ( ( nextPtr = getFirst() ) )
177 {
178 dst.insert( *nextPtr );
179 }
180 }
181
182 /// Empties the Map. All references to the item(s) in the Map are lost.
183 void clearTheMap()
184 {
185 // Drain map so the debug traps work correctly
186 while ( getFirst() )
187 {
188 ;
189 }
190 }
191
192
193private:
194 /// Prevent access to the copy constructor -->Containers can not be copied!
195 Map( const Map<ITEM>& m ) = delete;
196
197 /// Prevent access to the assignment operator -->Containers can not be copied!
198 Map<ITEM>& operator=( const Map<ITEM>& m ) = delete;
199
200 /// Prevent access to the move constructor -->Containers can not be implicitly moved!
201 Map( Map<ITEM>&& m ) = delete;
202
203 /// Prevent access to the move assignment operator -->Containers can not be implicitly moved!
204 Map<ITEM>& operator=( Map<ITEM>&& m ) = delete;
