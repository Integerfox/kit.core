23/** This concrete class implements a IOutput stream that outputs
24 the data to multiple (up to 4) output streams . The write() methods will
25 return false when at least of the IOutput stream(s) had an error. There
26 are methods provided (firstFailed(), nextFailed()) which returns a list of
27 all currently failed IOutput streams. Once a stream has been marked as failed
28 it will not be added back to the list of "good/active" streams. The user
29 must explicitly delete the stream and then re-add it. NOTE: There is no
30 guaranty of order or 'atomicity' with respect to other streams when
31 writing to multiple streams.
32
33 NOTE: The implementation is NOT thread safe. The application must provide
34 its own locking/critical section logic if calling a TeeOutput instance
35 from multiple threads.
36 */
37class TeeOutput : public IOutput
38{
39public:
40 /// Maximum number of output streams supported
41 static constexpr int MAX_STREAMS = 4;
42
43public:
44 /// Constructor. Starts off with NO output streams
45 TeeOutput() noexcept;
46
47 /// Constructor. Starts off with ONE output stream
48 TeeOutput( IOutput& streamA ) noexcept;
49
50 /// Constructor. Starts off with TWO output streams
51 TeeOutput( IOutput& streamA, IOutput& streamB ) noexcept;
52
53
54public:
55 /** Adds a stream. Return true when successful; else false is returned
56 if no more streams can be added
57 */
58 inline bool add( IOutput& stream ) noexcept { return add( m_streams, stream ); }
59
60 /** Removes a stream. Returns true if the stream was actually
61 removed (i.e. that it was in the list to start with).
62
63 NOTE: If the stream exists it will be removed regardless if it is
64 considered good or failed.
65 */
66 bool remove( IOutput& stream ) noexcept;
67
68
69public:
70 /** Returns the first failed stream. If no failed streams exist nullptr
71 will be returned.
72 */
73 inline IOutput* firstFailed() noexcept { return first( m_failed ); }
74
75 /** Returns the next failed stream in the list. If there are no more
76 failed streams nullptr will be returned.
77 */
78 inline IOutput* nextFailed( IOutput& currentFailedStream ) noexcept { return next( m_failed, currentFailedStream ); }
