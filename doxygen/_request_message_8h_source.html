22/** This concrete template class represents a service request message to a
23 particular server. The SERVICE argument of the template is the abstract
24 interface of the Service for which this Request message is associated with.
25
26 This is the interface that the message uses to invoke the message operation.
27 By definition, the SERVICE class must have a member function named "request"
28 that returns nothing (void) and has a signature whose argument is a reference
29 to the message defined by this template.
30 */
31template <class SERVICE, class PAYLOAD>
32class RequestMessage : public ServerMessage
33{
34public:
35 /// Constructor
36 RequestMessage( SERVICE& srv, PAYLOAD& payload, IReturnHandler& returnHandler ) noexcept
37 : ServerMessage( returnHandler )
38 , m_srv( srv )
39 , m_payload( payload )
40 {
41 }
42
43 /// Constructor
44 RequestMessage( SAP<SERVICE>& sapsrv, PAYLOAD& payload, IReturnHandler& returnHandler ) noexcept
45 : ServerMessage( returnHandler ), m_srv( sapsrv.getService() ), m_payload( payload )
46 {
47 }
48
49public:
50 /// See Kit::Itc::Message. This call executes in the Service's thread context.
51 void process() noexcept override
52 {
53 m_srv.request( *this );
54 }
55
56public:
57 /// Returns the payload associated with this request
58 PAYLOAD& getPayload() noexcept
59 {
60 return m_payload;
61 }
62
63protected:
64 /// Reference to the server interface whose 'request' will be called
65 SERVICE& m_srv;
66
67 /// Reference to the request payload
68 PAYLOAD& m_payload;
69};
