21/** This helper class is used to determine when an entire Json object has
22 been read from an input source.
23
24 This class does not read or store the incoming data - it simply inspect

28 Usage:
29 \code
30
31
32 Kit::Io::File::Input fd("bob.txt");
33 ObjectDetector detector;
34
35 while( fd.eof() == false )

37 uint8_t buffer[256];
38 int bytesRead;

41 size_t startPosition;

45 {
46 // JSON object found.
47 // The initial '{' is at 'startPosition'
48 // The final/closing '}' is at 'endPosition'
49 // DO Something...
50 }
51 }
52 }
53
54 \endcode
55
56
57 */

63public:
64 /** Inspect 'numBytes' and returns true if a complete JSON object was
65 found. The pointers 'startOffset' and 'endOffset' return the byte
66 offset to the start and end of the found JSON object. The offset is
67 based on the number of number consumed since the object was created
68 or reset() was called.
69
70 NOTE: The class does NOT VALIDATE the JSON syntax. It ONLY finds
