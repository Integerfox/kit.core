19/** Compile-time macros for instrumenting EventLoop with watchdog monitoring.
20 These macros can be enabled/disabled at compile time and support both
21 direct function calls and pointer-based calls for flexibility.
22 */
23
24#ifdef USE_KIT_SYSTEM_WATCHDOG
25
26/** Macro to start watchdog monitoring in an EventLoop.
27 Usage: KIT_SYSTEM_WATCHDOG_START_EVENTLOOP(thisPtr, timingSource)
28
29 @param thisPtr The IWatchedEventLoop instance pointer
30 @param timingSource Reference to the TimerManager instance that provides timing
31 */
32#define KIT_SYSTEM_WATCHDOG_START_EVENTLOOP( thisPtr, timingSource ) \
33 do \
34 { \
35 if ( (thisPtr) ) \
36 { \
37 (thisPtr)->startWatcher( timingSource ); \
38 } \
39 } \
40 while ( 0 )
41
42/** Macro to stop watchdog monitoring in an EventLoop.
43 Usage: KIT_SYSTEM_WATCHDOG_STOP_EVENTLOOP(thisPtr)
44
45 @param thisPtr The IWatchedEventLoop instance pointer
46 */
47#define KIT_SYSTEM_WATCHDOG_STOP_EVENTLOOP( thisPtr ) \
48 do \
49 { \
50 if ( (thisPtr) ) \
51 { \
52 (thisPtr)->stopWatcher(); \
53 } \
54 } \
55 while ( 0 )
56
57/** Macro to perform watchdog monitoring in an EventLoop.
58 Usage: KIT_SYSTEM_WATCHDOG_EVENTLOOP_MONITOR(thisPtr)
59
60 @param thisPtr The IWatchedEventLoop instance pointer
61 */
62#define KIT_SYSTEM_WATCHDOG_EVENTLOOP_MONITOR( thisPtr ) \
63 do \
64 { \
65 if ( (thisPtr) && (thisPtr)->isSupervisorThread() ) \
66 { \
67 (thisPtr)->monitorWdog(); \
68 } \
69 } \
70 while ( 0 )
71
72#else
73

76/// Watchdog framework disabled at compile time

95/** This abstract interface defines the API for watching/monitoring an EventLoop
96 for watchdog purposes. The interface provides methods to start/stop monitoring
97 and to perform periodic watchdog checks.
98
99 NOTE: The classes method should ONLY be called/invoked using the above macros
100 */
101class IWatchedEventLoop
102{
103public:
104 /** Starts the watchdog monitoring for this event loop.
105 Should be called when the event loop begins execution.
106
107 @param timingSource Reference to the Timing source (aka EventLoop) for the thread that is being watched
108 */
109 virtual void startWatcher( Kit::System::TimerManager& timingSource ) noexcept = 0;
110
111 /** Stops the watchdog monitoring for this event loop.
112 Should be called when the event loop is shutting down.
113 */
114 virtual void stopWatcher() noexcept = 0;
115
116 /** Performs watchdog monitoring check. This method should be called
117 periodically during event loop execution to indicate the thread
118 is still alive and processing events.
119 */
120 virtual void monitorWdog() noexcept = 0;
121
122 /** Returns true if this is the supervisor thread responsible for
123 monitoring other threads.
124
125 @return true if this is the supervisor thread, false otherwise
126 */
127 virtual bool isSupervisorThread() const noexcept = 0;
128
129 /// Virtual destructor
130 virtual ~IWatchedEventLoop() = default;
131};
