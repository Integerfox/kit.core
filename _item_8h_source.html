20/** This class is used by the Container classes to implement a various types
21 of singly linked containers.
22
23 Note: Client code, i.e. application code that needs to define a
24 'containerized' class only needs to inherit from this
25 interface. The Client code SHOULD/SHALL NOT access any of its
26 members or methods! These members/methods are intended to ONLY be
27 accessible by the container classes.
28 */
29class Item
30{
31protected:
32 /// Constructor

36 /** Constructor used ONLY with the child class MapItem: -->special
37 constructor to allow a Map to be statically allocated. Only the Map
38 itself should ever use this constructor -->not intended for Items in a
39 Map
40 */
41 Item( const char* /* ignoreThisParameter_usedToCreateAUniqueConstructor */ ) {}
42
43
44public:
45 /** Helper method to trap when inserting an item in multiple containers.
46 A fatal error will be generated if 'Item' is attempted to be put into
47 more than on container.
48
49 Notes:
50 o The method returns false when there is error - which is ONLY
51 usefully during unittesting when the invoking a FatalError
52 does NOT terminate the application.
53 */
54 bool insert_( void* newContainerPtr );
55
56 /// Returns 'true' if the instance is in the specified container.
57 inline bool isInContainer_( const void* containerPtr ) const noexcept
58 {
59 if ( m_inListPtr_ != containerPtr )
60 {
61 return false;
62 }
63 return true;
64 }
65
66 /** Verifies that the item is in the list for use when the next() method is
67 called. If the item is not in the list, a fatal error is generated.
68
69 Notes:
70 o The method returns false when there is error - which is ONLY
71 usefully during unittesting when the invoking a FatalError
72 does NOT terminate the application.
73 */
74 bool validateNextOkay_( const void* containerPtr ) const noexcept;
75
76 /** Helper method to do the proper 'clean-up' for the multiple-containers-error-trap
77 when removing an item from a container.
78 */
79 inline static void remove_( Item* itemPtr ) noexcept
80 {
81 if ( itemPtr )
82 {
83 itemPtr->m_inListPtr_ = nullptr;
84 }
85 }
86
87public:
88 /// The link field.
89 void* m_nextPtr_;
90
91 /** Debug field. This member is used to trap when there is an attempt
92 to insert a item into a container when it is already in a container
93 */
94 void* m_inListPtr_;
