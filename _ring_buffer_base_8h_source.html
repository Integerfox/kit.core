23/** This 'implementation' class provide the base functionality for a ring buffer.
24
25 The Ring buffer is ISR/Thread safe as long as following constraints are met:
26 - Writing to an 'unsigned' data type is an atomic machine instruction.
27 - There is SINGLE producer (add) and a SINGLE consumer (remove)
28
29 NOTE: This class is not intended to be used directly. A child class is
30 required to provide the ring buffer memory and type safety.
31 */
32class RingBufferBase
33{
34protected:
35 /// Constructor
36 RingBufferBase( unsigned numMaxElements )
37 : m_readIdx( 0 )
38 , m_writeIdx( 0 )
39 , m_elements( numMaxElements )
40 {
41 }
42
43public:
44 /// This method returns true if the Ring Buffer is empty
45 bool isEmpty() const noexcept
46 {
47 return m_readIdx == m_writeIdx;
48 }
49
50 /// This method returns true if the Ring Buffer is full
51 bool isFull() const noexcept
52 {
53 return ( ( m_writeIdx + 1 ) % m_elements ) == m_readIdx;
54 }
55
56 /// This method returns the maximum number of items that can be stored in the Ring buffer.
57 unsigned getMaxItems() const noexcept
58 {
59 return m_elements - 1;
60 }
61
62 /// This method returns the CURRENT number of elements in the Ring Buffer
63 unsigned getNumItems() const noexcept
64 {
65 return ( m_writeIdx - m_readIdx + m_elements ) % m_elements;
66 }
67
68public:
69 /** This method resets the underlying read/write indexes, i.e. effectively
70 clearing the Ring Buffer contents. This method can ONLY BE USED when
71 the Ring Buffer is NOT in use. It is the application's responsibility
72 for ensuring this condition is met.
73 */
74 void clearTheBuffer() noexcept
75 {
76 m_readIdx = m_writeIdx = 0;
77 }
78
79protected:
80 /** Removes the first item in the Ring buffer. The contents of the
81 removed item will be copied into the 'elemPtr' argument. The method
82 returns true if the operation was successful; else false is
83 returned, i.e. the Ring buffer is/was empty.
84 */
85 bool remove( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept
86 {
87 KIT_SYSTEM_ASSERT( elemPtr != nullptr );
88 KIT_SYSTEM_ASSERT( elemSize > 0 );
89 KIT_SYSTEM_ASSERT( srcRingBufMemory != nullptr );
90
91 // Snapshot of ring buffer's indexes
92 unsigned readIdx = m_readIdx;
93 unsigned writeIdx = m_writeIdx;
94
95 // Trap - the Ring buffer is empty
96 if ( readIdx == writeIdx )
97 {
98 return false;
99 }
100
101 // Copy the element data from the ring buffer to the provided pointer
102 uint8_t* srcPtr = ( (uint8_t*)srcRingBufMemory ) + ( readIdx * elemSize );
103 memcpy( elemPtr, srcPtr, elemSize );
104
105 // Update the read index
106 m_readIdx = ( readIdx + 1 ) % m_elements;
107 return true;
108 }
109
110 /** Add a new item to the Ring Buffer as the last item in the buffer. The
111 contents of the source item (aka 'elemPtr') will be copied into the
112 Ring Buffer. Returns true if the operation was successful; else false
113 is returned, i.e. the Buffer was full prior to the attempted add().
114 */
115 bool add( const void* elemPtr, size_t elemSize, void* dstRingBufMemory ) noexcept
116 {
117 KIT_SYSTEM_ASSERT( elemPtr != nullptr );
118 KIT_SYSTEM_ASSERT( elemSize > 0 );
119 KIT_SYSTEM_ASSERT( dstRingBufMemory != nullptr );
120
121 // Snapshot of ring buffer's indexes
122 unsigned readIdx = m_readIdx;
123 unsigned writeIdx = m_writeIdx;
124
125 // Trap - the Ring buffer is full
126 if ( ( writeIdx + 1 ) % m_elements == readIdx )
127 {
128 return false;
129 }
130
131 // Copy the element data to the provided pointer to the ring buffer
132 uint8_t* dstPtr = ( (uint8_t*)dstRingBufMemory ) + ( writeIdx * elemSize );
133 memcpy( dstPtr, elemPtr, elemSize );
134
135 // Update the write index
136 m_writeIdx = ( writeIdx + 1 ) % m_elements;
137 return true;
138 }
139
140protected:
141 /** This method inspects the head element of the Ring buffer without removing it.
142 The contents of the head element will be copied into the 'elemPtr' argument.
143 The method returns true if the operation was successful; else false is
144 returned, i.e. the Ring buffer is empty.
145 */
146 bool peekHead( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept
147 {
148 KIT_SYSTEM_ASSERT( elemPtr != nullptr );
149 KIT_SYSTEM_ASSERT( elemSize > 0 );
150 KIT_SYSTEM_ASSERT( srcRingBufMemory != nullptr );
151
152 // Snapshot of ring buffer's indexes
153 unsigned readIdx = m_readIdx;
154 unsigned writeIdx = m_writeIdx;
155
156 // Trap - the Ring buffer is empty
157 if ( readIdx == writeIdx )
158 {
159 return false;
160 }
161
162 // Copy the element data from the ring buffer to the provided pointer
163 uint8_t* srcPtr = ( (uint8_t*)srcRingBufMemory ) + ( readIdx * elemSize );
164 memcpy( elemPtr, srcPtr, elemSize );
165
166 return true;
167 }
168
169 /** This method inspects the tail element of the Ring buffer without removing it.
170 The contents of the tail element will be copied into the 'elemPtr' argument.
171 The method returns true if the operation was successful; else false is
172 returned, i.e. the Ring buffer is empty.
173 */
174 bool peekTail( void* elemPtr, size_t elemSize, const void* srcRingBufMemory ) noexcept
175 {
176 KIT_SYSTEM_ASSERT( elemPtr != nullptr );
177 KIT_SYSTEM_ASSERT( elemSize > 0 );
178 KIT_SYSTEM_ASSERT( srcRingBufMemory != nullptr );
179
180 // Snapshot of ring buffer's indexes
181 unsigned readIdx = m_readIdx;
182 unsigned writeIdx = m_writeIdx;
183
184 // Trap - the Ring buffer is empty
185 if ( readIdx == writeIdx )
186 {
187 return false;
188 }
189
190 // Get previous tail index
191 unsigned prevTailIdx = writeIdx == 0 ? ( m_elements - 1 ) : ( writeIdx - 1 );
192
193 // Copy the element data from the ring buffer to the provided pointer
194 uint8_t* srcPtr = ( (uint8_t*)srcRingBufMemory ) + ( prevTailIdx * elemSize );
195 memcpy( elemPtr, srcPtr, elemSize );
196
197 return true;
198 }
199
200protected:
201 /** This method returns a pointer to the next item to be removed. In addition
202 it returns the number of elements that can be removed as linear/flat
203 buffer (i.e. without wrapping around raw buffer memory).
204
205 The returned pointer and element count are only valid till the next
206 read/remove operation.
207
208 If the Ring buffer is empty, a null pointer is returned
209 */
210 void* peekNextRemoveItems( unsigned& dstNumFlatElements, size_t elemSize, const void* srcRingBufMemory ) const noexcept
211 {
212 KIT_SYSTEM_ASSERT( srcRingBufMemory != nullptr );
213
214 // Snapshot of ring buffer's indexes
215 unsigned readIdx = m_readIdx;
216 unsigned writeIdx = m_writeIdx;
217
218 // Check for empty
219 if ( readIdx == writeIdx )
220 {
221 dstNumFlatElements = 0;
222 return nullptr;
223 }
224
225 // Number of elements that can be removed without wrapping around the memory buffer
226 unsigned currentElems = ( writeIdx - readIdx + m_elements ) % m_elements;
227 dstNumFlatElements = m_elements - readIdx;
228 if ( dstNumFlatElements > currentElems )
229 {
230 dstNumFlatElements = currentElems;
231 }
232
233 uint8_t* srcPtr = ( (uint8_t*)srcRingBufMemory ) + ( readIdx * elemSize );
234 return srcPtr;
235 }
236
237 /** This method 'removes' N elements - that were removed using the
238 pointer returned from peekNextRemoveItems - from the ring buffer.
239 Basically it updates the head pointer to reflect items removed using
240 direct memory access.
241
242 'numElementsToRemove' be less than or equal to the 'dstNumFlatElements'
243 returned from peekNextRemoveItems().
244
245 CAUTION: IF YOU DON'T UNDERSTAND THE USE CASE FOR THIS METHOD - THEN
246 DON'T USE IT. If this method is used improperly, it WILL
247 CORRUPT the Ring Buffer!
248 */
249 void removeElements( unsigned numElementsToRemove ) noexcept
250 {
251 // By definition/design - I simply have to update the read index
252 m_readIdx = ( m_readIdx + numElementsToRemove ) % m_elements;
253 }
254
255protected:
256 /** This method returns a pointer to the next item to be added. In addition
257 it returns the number of elements that can be added as linear/flat
258 buffer (i.e. without wrapping around raw buffer memory)
259
260 If the Ring buffer is full, a null pointer is returned
261 */
262 void* peekNextAddItems( unsigned& dstNumFlatElements, size_t elemSize, const void* srcRingBufMemory ) const noexcept
263 {
264 // Snapshot of ring buffer's indexes
265 unsigned readIdx = m_readIdx;
266 unsigned writeIdx = m_writeIdx;
267
268 // Check for full
269 if ( ( ( writeIdx + 1 ) % m_elements ) == readIdx )
270 {
271 dstNumFlatElements = 0;
272 return nullptr;
273 }
274
275 // Number of elements that can be added without wrapping around the memory buffer
276 unsigned currentElems = ( writeIdx - readIdx + m_elements ) % m_elements;
277 unsigned totalAvailElems = m_elements - 1 - currentElems;
278 dstNumFlatElements = m_elements - writeIdx;
279 if ( dstNumFlatElements > totalAvailElems )
280 {
281 dstNumFlatElements = totalAvailElems;
282 }
283
284 // Get a pointer to next 'available' address to add an element
285 uint8_t* srcPtr = ( (uint8_t*)srcRingBufMemory ) + ( writeIdx * elemSize );
286 return srcPtr;
287 }
288
289 /** This method 'adds' N elements - that were populated using the
290 pointer returned from peekNextAddItems - to the ring buffer. Basically
291 its updates the tail pointer to reflect items added using direct
292 memory access.
293
294 'numElementsAdded' be less than or equal to the 'dstNumFlatElements'
295 returned from peekNextAddItems().
296
297 CAUTION: IF YOU DON'T UNDERSTAND THE USE CASE FOR THIS METHOD - THEN
298 DON'T USE IT. If this method is used improperly, it WILL
299 CORRUPT the Ring Buffer!
300 */
301 void addElements( unsigned numElementsAdded ) noexcept
302 {
303 // By definition/design - I simply have to update the write index
304 m_writeIdx = ( m_writeIdx + numElementsAdded ) % m_elements;
305 }
306
307protected:
308 /** Index of the next item to be read from the ring. This variable is only
309 updated when a read operation is performed. In addition, the variable
310 is atomically updated.
311 */
312 volatile unsigned m_readIdx;
313
314 /** Index of the next item to be added to the ring. This variable
315 is only updated when a add operation is performed. In addition,
316 the variable is atomically updated.
317 */
318 volatile unsigned m_writeIdx;
319
320 /// Maximum number of element in the 'ring memory
321 const unsigned m_elements;
322
323protected:
324 /// Prevent access to the copy constructor -->Ring Buffers can not be copied!
325 RingBufferBase( const RingBufferBase& m ) = delete;
326
327 /// Prevent access to the assignment operator -->Ring Buffers can not be copied!
328 RingBufferBase& operator=( const RingBufferBase& m ) = delete;
329
330 /// Prevent access to the move constructor -->Ring Buffers can not be implicitly moved!
331 RingBufferBase( RingBufferBase&& m ) = delete;
332
333 /// Prevent access to the move assignment operator -->Ring Buffers can not be implicitly moved!
334 RingBufferBase& operator=( RingBufferBase&& m ) = delete;
335};
